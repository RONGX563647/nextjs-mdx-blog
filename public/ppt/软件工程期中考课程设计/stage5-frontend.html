阶段五：前端可视化界面

一、前端界面概述

系统提供多场景可视化操作界面，界面需实现交互操作、数据展示。根据需求文档，主要包含三大界面场景：

界面 功能
登录界面 实现用户账号密码输入、注册跳转、登录验证
区块链交易界面 实现节点管理、交易创建、挖矿、余额查询、链查询、数据验证、节点同步等操作
聊天界面 实现成员消息发送、消息接收、聊天记录展示

1.1 技术栈

  • Vue 3：渐进式 JavaScript 框架
  • Vite：前端构建工具
  • Element Plus：UI 组件库
  • Axios：HTTP 请求库
  • Vue Router：路由管理

<script setup>
import { ref, reactive } from 'vue'
import { useRouter } from 'vue-router'
import axios from 'axios'
import { ElMessage } from 'element-plus'

const router = useRouter()
const isLogin = ref(true)

const loginForm = reactive({
  username: '',
  password: ''
})

const registerForm = reactive({
  username: '',
  password: '',
  confirmPassword: ''
})

function toggleMode() {
  isLogin.value = !isLogin.value
}

async function handleLogin() {
  try {
    const response = await axios.post('/api/user/login', {
      username: loginForm.username,
      password: loginForm.password
    })
    if (response.data.status === 'success') {
      // 保存用户信息
      localStorage.setItem('userInfo', JSON.stringify(response.data.user))
      localStorage.setItem('username', response.data.user.username)
      ElMessage.success('登录成功')
      router.push('/dashboard')
    }
  } catch (error) {
    ElMessage.error(error.response?.data?.message || '登录失败')
  }
}

async function handleRegister() {
  if (registerForm.password !== registerForm.confirmPassword) {
    ElMessage.warning('两次密码输入不一致')
    return
  }
  try {
    const response = await axios.post('/api/user/register', {
      username: registerForm.username,
      password: registerForm.password
    })
    if (response.data.status === 'success') {
      ElMessage.success('注册成功，请登录')
      isLogin.value = true
    }
  } catch (error) {
    ElMessage.error(error.response?.data?.message || '注册失败')
  }
}
</script>

3.3 路由配置

// router/index.js
import { createRouter, createWebHistory } from 'vue-router'

const routes = [
  {
    path: '/',
    redirect: '/login'
  },
  {
    path: '/login',
    name: 'Login',
    component: () => import('../views/Login.vue')
  },
  {
    path: '/dashboard',
    component: () => import('../views/Dashboard.vue'),
    children: [
      {
        path: 'blockchain',
        name: 'Blockchain',
        component: () => import('../views/BlockchainView.vue')
      },
      {
        path: 'chat',
        name: 'Chat',
        component: () => import('../views/ChatView.vue')
      },
      {
        path: 'profile',
        name: 'Profile',
        component: () => import('../views/ProfileView.vue')
      }
    ]
  }
]

const router = createRouter({
  history: createWebHistory(),
  routes
})

// 路由守卫
router.beforeEach((to, from, next) => {
  const userInfo = localStorage.getItem('userInfo')
  if (to.path !== '/login' && !userInfo) {
    next('/login')
  } else {
    next()
  }
})

export default router

<script setup>
import { ref, reactive, onMounted } from 'vue'
import axios from 'axios'
import { ElMessage } from 'element-plus'

const nodeInfo = ref({})
const wallet = ref({})
const stats = ref({})
const chain = ref([])
const peers = ref([])
const txDialogVisible = ref(false)
const newPeerAddress = ref('')

const txForm = reactive({
  recipient: '',
  amount: 1
})

// 加载节点信息
async function loadNodeInfo() {
  const res = await axios.get('/api/blockchain/node/info')
  nodeInfo.value = res.data
}

async function loadWallet() {
  const res = await axios.get('/api/blockchain/wallet')
  wallet.value = res.data
}

async function loadStats() {
  const res = await axios.get('/api/blockchain/stats')
  stats.value = res.data
}

async function loadChain() {
  const res = await axios.get('/api/blockchain/chain')
  chain.value = res.data
}

async function loadPeers() {
  const res = await axios.get('/api/blockchain/peers')
  peers.value = res.data
}

// 创建交易
async function submitTransaction() {
  try {
    await axios.post('/api/blockchain/transaction', null, {
      params: {
        senderAddress: wallet.value.address,
        recipientAddress: txForm.recipient,
        amount: txForm.amount
      }
    })
    ElMessage.success('交易创建成功')
    txDialogVisible.value = false
    loadChain()
  } catch (error) {
    ElMessage.error('交易创建失败')
  }
}

// 挖矿
async function handleMine() {
  try {
    await axios.post('/api/blockchain/mine', null, {
      params: { minerAddress: wallet.value.address }
    })
    ElMessage.success('挖矿成功')
    loadChain()
    loadStats()
  } catch (error) {
    ElMessage.error('挖矿失败')
  }
}

// 验证链
async function handleValidate() {
  const res = await axios.get('/api/blockchain/validate')
  ElMessage[res.data.valid ? 'success' : 'error'](
    res.data.valid ? '区块链有效' : '区块链无效'
  )
  loadStats()
}

// 同步网络
async function handleSync() {
  await axios.post('/api/blockchain/sync')
  ElMessage.success('同步完成')
  loadChain()
}

// 添加邻居节点
async function addPeer() {
  if (!newPeerAddress.value) return
  await axios.post('/api/blockchain/peers/add', null, {
    params: { peerAddress: newPeerAddress.value }
  })
  newPeerAddress.value = ''
  ElMessage.success('节点添加成功')
  loadPeers()
}

// 移除邻居节点
async function removePeer(peer) {
  await axios.delete('/api/blockchain/peers/remove', {
    params: { peerAddress: peer }
  })
  ElMessage.success('节点已移除')
  loadPeers()
}

function formatTimestamp(timestamp) {
  return new Date(timestamp).toLocaleString()
}

onMounted(() => {
  loadNodeInfo()
  loadWallet()
  loadStats()
  loadChain()
  loadPeers()
})
</script>

<script setup>
import { ref, nextTick, onMounted, onUnmounted } from 'vue'
import axios from 'axios'
import { ElMessage } from 'element-plus'

const messages = ref([])
const newMessage = ref('')
const currentUser = ref(localStorage.getItem('username') || 'guest')
const messageContainer = ref(null)

// 发送消息
async function sendMessage() {
  if (!newMessage.value.trim()) return
  
  try {
    await axios.post('/api/chat/send', {
      content: newMessage.value,
      sender: currentUser.value
    })
    
    // 添加到本地列表
    messages.value.push({
      messageId: Date.now().toString(),
      content: newMessage.value,
      sender: currentUser.value,
      type: 'CHAT',
      timestamp: new Date().toISOString()
    })
    
    newMessage.value = ''
    scrollToBottom()
  } catch (error) {
    ElMessage.error('消息发送失败')
  }
}

// 滚动到底部
function scrollToBottom() {
  nextTick(() => {
    if (messageContainer.value) {
      messageContainer.value.scrollTop = messageContainer.value.scrollHeight
    }
  })
}

// 格式化时间
function formatTime(timestamp) {
  if (!timestamp) return ''
  const date = new Date(timestamp)
  return date.toLocaleTimeString()
}

// 获取类型标签
function getTypeTag(type) {
  const map = {
    'INFO': 'info',
    'WARNING': 'warning',
    'ERROR': 'danger',
    'ORDER': 'success',
    'NOTIFICATION': '',
    'CHAT': ''
  }
  return map[type] || ''
}

// 定时轮询新消息
let pollInterval = null

onMounted(() => {
  pollInterval = setInterval(async () => {
    // TODO: 通过 WebSocket/SSE 实时获取消息
    // 或使用轮询方式获取最新消息
  }, 2000)
})

onUnmounted(() => {
  if (pollInterval) {
    clearInterval(pollInterval)
  }
})
</script>
