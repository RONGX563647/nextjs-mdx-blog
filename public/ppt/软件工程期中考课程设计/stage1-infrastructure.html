阶段一：项目基础设施

一、项目概述

区块链可视化管理交易系统是一个基于前后端分离架构的综合项目，集成区块链平台、区块链节点管理系统、RabbitMQ即时通讯服务平台，实现用户注册登录、区块链节点管理、交易挖矿、数据验证、成员即时聊天等功能。

1.1 项目目标

  • 区块链基础平台：实现区块生成、交易打包、挖矿、链式存储、防篡改、P2P节点同步
  • 区块链节点管理系统：实现用户-节点绑定、节点信息管理、交易操作、数据验证、节点同步
  • RabbitMQ交流服务平台：实现平台成员即时聊天、交易事宜沟通

4.2 区块链核心模型

Block.java - 区块数据结构：

public class Block {
    private String hash;           // 当前区块哈希值
    private String previousHash;   // 前一区块哈希值
    private List<Transaction> transactions;  // 交易列表
    private long timestamp;        // 时间戳
    private int nonce;             // 工作量证明随机数
    private int index;             // 区块索引

    public Block(int index, String previousHash, List<Transaction> transactions) {
        this.index = index;
        this.previousHash = previousHash;
        this.transactions = transactions != null ? transactions : new ArrayList<>();
        this.timestamp = System.currentTimeMillis();
        this.nonce = 0;
        this.hash = calculateHash();
    }

    public String calculateHash() {
        String data = index + previousHash + timestamp + nonce + transactions.toString();
        return HashUtil.sha256(data);
    }

    public void mineBlock(int difficulty) {
        String target = new String(new char[difficulty]).replace('\0', '0');
        while (!hash.substring(0, difficulty).equals(target)) {
            nonce++;
            hash = calculateHash();
        }
        System.out.println("区块挖掘完成! Hash: " + hash);
    }
}

字段说明：

字段 说明
hash 当前区块的 SHA-256 哈希值
previousHash 链中前一个区块的哈希值，保证链式结构
transactions 该区块包含的所有交易记录
timestamp 区块创建时间戳
nonce 用于工作量证明的随机数
index 区块在链中的索引位置

4.3 Blockchain 核心类

Blockchain.java - 区块链核心逻辑：

public class Blockchain {
    private List<Block> chain;
    private List<Transaction> pendingTransactions;
    private int difficulty;
    private double miningReward;

    public Blockchain() {
        this.chain = new ArrayList<>();
        this.pendingTransactions = new ArrayList<>();
        this.difficulty = 4;
        this.miningReward = 10.0;
        createGenesisBlock();
    }

    private void createGenesisBlock() {
        Block genesisBlock = new Block(0, "0", new ArrayList<>());
        genesisBlock.mineBlock(difficulty);
        chain.add(genesisBlock);
    }

    public void addTransaction(Transaction transaction) {
        if (transaction.getSenderAddress() == null || transaction.getRecipientAddress() == null) {
            throw new IllegalArgumentException("交易地址不能为空");
        }
        if (transaction.getAmount() <= 0) {
            throw new IllegalArgumentException("交易金额必须大于 0");
        }
        pendingTransactions.add(transaction);
    }

    public boolean minePendingTransactions(String minerAddress) {
        if (pendingTransactions.isEmpty()) {
            return false;
        }
        Block newBlock = new Block(chain.size(), getLatestBlock().getHash(), pendingTransactions);
        newBlock.mineBlock(difficulty);
        chain.add(newBlock);
        Transaction rewardTransaction = new Transaction("0", minerAddress, miningReward);
        pendingTransactions = new ArrayList<>();
        pendingTransactions.add(rewardTransaction);
        return true;
    }

    public Block getLatestBlock() {
        return chain.get(chain.size() - 1);
    }

    public double getBalance(String address) {
        double balance = 0;
        for (Block block : chain) {
            for (Transaction transaction : block.getTransactions()) {
                if (transaction.getRecipientAddress().equals(address)) {
                    balance += transaction.getAmount();
                }
                if (transaction.getSenderAddress().equals(address)) {
                    balance -= transaction.getAmount();
                }
            }
        }
        return balance;
    }

    public boolean isChainValid() {
        for (int i = 1; i < chain.size(); i++) {
            Block currentBlock = chain.get(i);
            Block previousBlock = chain.get(i - 1);
            if (!currentBlock.getHash().equals(currentBlock.calculateHash())) {
                return false;
            }
            if (!currentBlock.getPreviousHash().equals(previousBlock.getHash())) {
                return false;
            }
        }
        return true;
    }
}

4.4 P2P 网络节点

Node.java - 网络节点通信：

public class Node {
    private static final Logger logger = LoggerFactory.getLogger(Node.class);

    private String nodeId;
    private String host;
    private int port;
    private List<String> peerNodes;
    private ConcurrentHashMap<String, NodeInfo> knownNodes;
    private Blockchain blockchain;

    public Node(String host, int port, Blockchain blockchain) {
        this.nodeId = UUID.randomUUID().toString();
        this.host = host;
        this.port = port;
        this.peerNodes = new ArrayList<>();
        this.knownNodes = new ConcurrentHashMap<>();
        this.blockchain = blockchain;
        logger.info("节点创建成功：{}:{}", host, port);
    }

    public void addPeer(String peerAddress) {
        if (!peerNodes.contains(peerAddress)) {
            peerNodes.add(peerAddress);
            logger.info("已添加节点：{}", peerAddress);
        }
    }

    public void removePeer(String peerAddress) {
        peerNodes.remove(peerAddress);
        knownNodes.remove(peerAddress);
        logger.info("已移除节点：{}", peerAddress);
    }

    public void broadcastBlock(Block block) {
        for (String peer : peerNodes) {
            try {
                WebClient.create("http://" + peer)
                    .post()
                    .uri("/api/blockchain/block")
                    .bodyValue(block)
                    .retrieve()
                    .toBodilessEntity()
                    .block();
                logger.info("区块已广播到节点：{}", peer);
            } catch (Exception e) {
                logger.error("广播区块到节点 {} 失败：{}", peer, e.getMessage());
                removePeer(peer);
            }
        }
    }

    public void broadcastTransaction(Transaction transaction) {
        for (String peer : peerNodes) {
            try {
                WebClient.create("http://" + peer)
                    .post()
                    .uri("/api/blockchain/transaction/receive")
                    .bodyValue(transaction)
                    .retrieve()
                    .toBodilessEntity()
                    .block();
                logger.info("交易已广播到节点：{}", peer);
            } catch (Exception e) {
                logger.error("广播交易到节点 {} 失败：{}", peer, e.getMessage());
                removePeer(peer);
            }
        }
    }

    public void syncWithNetwork() {
        logger.info("开始与网络同步...");
        
        for (String peer : peerNodes) {
            List<Block> peerChain = requestChainFromPeer(peer);
            
            if (peerChain != null && peerChain.size() > 0) {
                logger.info("从节点 {} 获取到链，长度：{}", peer, peerChain.size());
                
                if (peerChain.size() > blockchain.getChain().size()) {
                    if (isValidChain(peerChain)) {
                        blockchain.setChain(peerChain);
                        logger.info("已同步更长的有效链");
                    }
                }
            }
        }
        logger.info("同步完成");
    }

    public List<Block> requestChainFromPeer(String peerAddress) {
        try {
            return WebClient.create("http://" + peerAddress)
                .get()
                .uri("/api/blockchain/chain")
                .retrieve()
                .bodyToMono(List.class)
                .block();
        } catch (Exception e) {
            logger.error("从节点 {} 获取链失败：{}", peerAddress, e.getMessage());
            removePeer(peerAddress);
            return null;
        }
    }

    private boolean isValidChain(List<Block> chainToValidate) {
        for (int i = 1; i < chainToValidate.size(); i++) {
            Block currentBlock = chainToValidate.get(i);
            Block previousBlock = chainToValidate.get(i - 1);
            
            if (!currentBlock.getHash().equals(currentBlock.calculateHash())) {
                return false;
            }
            if (!currentBlock.getPreviousHash().equals(previousBlock.getHash())) {
                return false;
            }
        }
        return true;
    }
}
