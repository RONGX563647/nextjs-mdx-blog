风电净空监控系统 - 全链路开发教程

📖 第一章：页面功能概述

风轮不平衡评估报告页面是风电净空监控系统的核心功能模块之一，主要用于监测和分析风机叶片的不平衡状态， 通过实时数据采集和统计分析，帮助运维人员及时发现风机运行异常，预防设备故障。

1.1 功能模块

📋 核心功能清单

  • 单风机趋势分析：查看指定风机在不同时间范围内的不平衡趋势
  • 多风机对比分析：对比同一风场下多个风机的不平衡数据
  • 等级配置管理：动态配置不平衡等级的划分标准
  • 实时状态监控：查询风机最新的不平衡等级状态
  • 数据导出：支持 Excel 格式的监测数据导出
  • 数据导入：支持批量导入手动录入的监测数据

2.1.1 获取不平衡等级配置

/**
 * 获取不平衡等级配置
 *
 * @return
 */
@ApiOperation(value="获取不平衡等级配置", notes="获取不平衡等级划分标准配置")
@GetMapping(value = "/thresholds")
public Result<List<Map<String, Object>>> getThresholds() {
    try {
        SysCustomConfig config = sysCustomConfigService.getOne(
            new QueryWrapper<SysCustomConfig>().eq("code", "unbalance_level_threshold")
        );

        if (config == null) {
            // 返回默认配置
            List<Map<String, Object>> defaultLevels = new ArrayList<>();
            defaultLevels.add(createLevelMap(1, 1.0, "无影响", "#52c41a"));
            defaultLevels.add(createLevelMap(2, 2.0, "轻微影响", "#73d13d"));
            defaultLevels.add(createLevelMap(3, 3.0, "中度影响", "#ffc53d"));
            defaultLevels.add(createLevelMap(4, 4.0, "严重影响", "#ff7a45"));
            defaultLevels.add(createLevelMap(5, 5.0, "紧急处理", "#ff4d4f"));
            return Result.OK(defaultLevels);
        }

        Map<String, Object> configMap = JSON.parseObject(
            config.getConfigValue(),
            new TypeReference<Map<String, Object>>() {}
        );

        List<Map<String, Object>> levels = (List<Map<String, Object>>) configMap.get("levels");
        return Result.OK(levels);
    } catch (Exception e) {
        log.error("获取不平衡等级配置失败", e);
        return Result.error("获取配置失败：" + e.getMessage());
    }
}

private Map<String, Object> createLevelMap(int level, double max, String text, String color) {
    Map<String, Object> map = new HashMap<>();
    map.put("level", level);
    map.put("max", max);
    map.put("text", text);
    map.put("color", color);
    return map;
}

2.1.2 查询单风机不平衡趋势数据

/**
 * 查询单风机不平衡趋势数据
 *
 * @param windTurbineCode 风机编号
 * @param timeRange 时间范围：day-日，week-周，month-月
 * @return
 */
@ApiOperation(value="查询单风机不平衡趋势数据", notes="查询单风机不平衡趋势数据")
@GetMapping(value = "/trend")
public Result<List<Map<String, Object>>> queryTrendData(
        @RequestParam(name="windTurbineCode", required=true) String windTurbineCode,
        @RequestParam(name="timeRange", defaultValue="day") String timeRange) {

    try {
        String startTime = null;
        String endTime = null;

        Calendar calendar = Calendar.getInstance();
        SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");

        if ("day".equals(timeRange)) {
            calendar.set(Calendar.HOUR_OF_DAY, 0);
            calendar.set(Calendar.MINUTE, 0);
            calendar.set(Calendar.SECOND, 0);
            startTime = sdf.format(calendar.getTime());
            calendar.add(Calendar.DAY_OF_MONTH, 1);
            endTime = sdf.format(calendar.getTime());
        } else if ("week".equals(timeRange)) {
            calendar.add(Calendar.DAY_OF_MONTH, -6);
            calendar.set(Calendar.HOUR_OF_DAY, 0);
            calendar.set(Calendar.MINUTE, 0);
            calendar.set(Calendar.SECOND, 0);
            startTime = sdf.format(calendar.getTime());
            calendar.add(Calendar.DAY_OF_MONTH, 7);
            endTime = sdf.format(calendar.getTime());
        } else if ("month".equals(timeRange)) {
            calendar.add(Calendar.DAY_OF_MONTH, -29);
            calendar.set(Calendar.HOUR_OF_DAY, 0);
            calendar.set(Calendar.MINUTE, 0);
            calendar.set(Calendar.SECOND, 0);
            startTime = sdf.format(calendar.getTime());
            calendar.add(Calendar.DAY_OF_MONTH, 30);
            endTime = sdf.format(calendar.getTime());
        }

        List<Map<String, Object>> trendData = fengjiUnbalanceStatsService.queryTrendData(
            windTurbineCode, startTime, endTime
        );
        return Result.OK(trendData);
    } catch (Exception e) {
        log.error("查询趋势数据失败", e);
        return Result.error("查询失败：" + e.getMessage());
    }
}

2.1.4 查询最新不平衡等级

/**
 * 查询最新不平衡等级
 *
 * @param windTurbineCode 风机编号
 * @return
 */
@ApiOperation(value="查询最新不平衡等级", notes="查询最新不平衡等级")
@GetMapping(value = "/latestLevel")
public Result<Map<String, Object>> queryLatestLevel(
        @RequestParam(name="windTurbineCode", required=true) String windTurbineCode) {

    try {
        Integer level = fengjiUnbalanceStatsService.queryLatestLevel(windTurbineCode);

        Map<String, Object> result = new HashMap<>();
        result.put("level", level);

        if (level != null) {
            String levelDesc = "";
            switch (level) {
                case 1: levelDesc = "无影响"; break;
                case 2: levelDesc = "轻微影响"; break;
                case 3: levelDesc = "中度影响"; break;
                case 4: levelDesc = "严重影响"; break;
                case 5: levelDesc = "紧急处理"; break;
                default: levelDesc = "未知"; break;
            }
            result.put("levelDesc", levelDesc);
        }

        return Result.OK(result);
    } catch (Exception e) {
        log.error("查询最新等级失败", e);
        return Result.error("查询失败：" + e.getMessage());
    }
}

  methods: {
    // 加载风场列表
    loadWindFarms() {
      getAction("/info/fengdianDeviceInfo/list", { pageNo: 1, pageSize: 100 }).then(res => {
        if (res.success) {
          this.windFarmOptions = res.result.records
        }
      })
    },
    // 加载风机列表
    loadWindTurbines() {
      if (!this.queryParam.windFarmCode) {
        this.windTurbineOptions = []
        return
      }
      getAction("/info/patrolFanInfo/list", {
        pageNo: 1,
        pageSize: 100,
        region: this.queryParam.windFarmCode
      }).then(res => {
        if (res.success) {
          this.windTurbineOptions = res.result.records
        }
      })
    },
    // 加载等级配置
    loadThresholds() {
      getAction(this.url.thresholds).then(res => {
        if (res.success && res.result) {
          this.levelData = res.result.map(item => ({
            ...item,
            percent: item.level // 用于饼图显示
          }))
        }
      })
    },
    // 加载趋势数据
    loadTrendData() {
      if (!this.queryParam.windTurbineCode) {
        this.trendData = []
        return
      }
      getAction(this.url.trend, {
        windTurbineCode: this.queryParam.windTurbineCode,
        timeRange: this.queryParam.timeRange
      }).then(res => {
        if (res.success && res.result) {
          this.trendData = res.result
        }
      })
    },
    // 加载对比数据
    loadCompareData() {
      if (!this.queryParam.windFarmCode) {
        this.compareData = []
        return
      }
      getAction(this.url.compare, {
        windFarmCode: this.queryParam.windFarmCode,
        timeRange: this.queryParam.timeRange
      }).then(res => {
        if (res.success && res.result) {
          this.compareData = res.result
        }
      })
    },
    // 加载表格数据
    loadData() {
      this.loading = true
      getAction(this.url.list, {
        ...this.queryParam,
        pageNo: this.ipagination.current,
        pageSize: this.ipagination.pageSize
      }).then(res => {
        if (res.success) {
          this.dataSource = res.result.records
          this.ipagination.total = res.result.total
        }
        this.loading = false
      }).catch(() => {
        this.loading = false
      })
    },
    // 查询
    searchQuery() {
      this.ipagination.current = 1
      this.loadData()
      this.loadTrendData()
      this.loadCompareData()
    },
    // 重置
    searchReset() {
      this.queryParam = {
        windFarmCode: "",
        windTurbineCode: "",
        timeRange: "day"
      }
      this.ipagination.current = 1
      this.loadData()
      this.trendData = []
      this.compareData = []
    },
    // 表格分页变化
    handleTableChange(pagination) {
      this.ipagination.current = pagination.current
      this.ipagination.pageSize = pagination.pageSize
      this.loadData()
    },
    // 获取等级文本
    getLevelText(level) {
      const levelMap = {
        1: "无影响",
        2: "轻微影响",
        3: "中度影响",
        4: "严重影响",
        5: "紧急处理"
      }
      return levelMap[level] || "未知"
    },
    // 获取等级状态颜色
    getLevelStatus(level) {
      const statusMap = {
        1: "success",
        2: "processing",
        3: "warning",
        4: "error",
        5: "error"
      }
      return statusMap[level] || "default"
    }
  },

    this.loadWindFarms()
    this.loadThresholds()
    this.loadData()
  }

5.2.2 Mapper XML SQL 示例

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN" "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<mapper namespace="org.jeecg.modules.fengji.stats.mapper.FengjiUnbalanceStatsMapper">

    <!-- 查询单风机不平衡趋势数据 -->
    <select id="queryTrendData" resultType="java.util.Map">
        SELECT
            stats_time as statsTime,
            unbalance_value as unbalanceValue,
            unbalance_level as unbalanceLevel,
            wind_speed as windSpeed,
            rotation_speed as rotationSpeed
        FROM
            fengji_unbalance_stats
        WHERE
            wind_turbine_code = #{windTurbineCode}
            AND stats_time >= #{startTime}
            AND stats_time < #{endTime}
        ORDER BY
            stats_time ASC
    </select>

    <!-- 查询多风机不平衡对比数据 -->
    <select id="queryCompareData" resultType="java.util.Map">
        SELECT
            wind_turbine_code as windTurbineCode,
            AVG(unbalance_value) as unbalanceValue,
            AVG(wind_speed) as windSpeed,
            AVG(rotation_speed) as rotationSpeed,
            COUNT(*) as dataCount
        FROM
            fengji_unbalance_stats
        WHERE
            wind_farm_code = #{windFarmCode}
            AND stats_time >= #{startTime}
            AND stats_time < #{endTime}
        GROUP BY
            wind_turbine_code
        ORDER BY
            wind_turbine_code
    </select>

    <!-- 查询最新不平衡等级 -->
    <select id="queryLatestLevel" resultType="java.lang.Integer">
        SELECT
            unbalance_level
        FROM
            fengji_unbalance_stats
        WHERE
            wind_turbine_code = #{windTurbineCode}
        ORDER BY
            stats_time DESC
        LIMIT 1
    </select>

</mapper>

对于不常变化的等级配置数据，可以使用 Redis 缓存，减少数据库查询： 

@Autowired
private RedisTemplate<String, Object> redisTemplate;

public List<Map<String, Object>> getThresholds() {
    // 尝试从缓存读取
    String cacheKey = "unbalance:thresholds";
    Object cached = redisTemplate.opsForValue().get(cacheKey);
    if (cached != null) {
        return (List<Map<String, Object>>) cached;
    }

    // 从数据库查询
    List<Map<String, Object>> levels = ...;

    // 写入缓存（过期时间 1 小时）
    redisTemplate.opsForValue().set(cacheKey, levels, 1, TimeUnit.HOURS);

    return levels;
}
