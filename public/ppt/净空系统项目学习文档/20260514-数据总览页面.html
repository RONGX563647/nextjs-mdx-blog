2.3.1 概览统计实现

@Override
public HomeOverviewVO getOverview(String windFarmCode, String queryType) {
    HomeOverviewVO overview = new HomeOverviewVO();
    try {
        // 1. 获取阈值配置（从系统配置表）
        ClearanceThresholdVO thresholds = getThresholds();
        Double warningThreshold = thresholds.getWarning();

        // 2. 获取风场设备列表
        List<FengdianDeviceInfo> devices = getDevicesByWindFarm(windFarmCode);

        // 3. 计算时间范围
        Date endDate = new Date();
        Date startDate = calculateStartDate(queryType);

        // 4. 从日报表查询统计数据
        List<DeviceDailyStats> statsList = deviceDailyStatsService.getDailyStats(
            null, windFarmCode, startDate, endDate
        );

        // 5. 统计机组状态
        int totalMachines = devices.size();
        int onlineMachines = 0;
        int abnormalMachines = 0;

        for (FengdianDeviceInfo device : devices) {
            if ("1".equals(device.getRunStatus())) {
                onlineMachines++;
            } else if ("2".equals(device.getRunStatus())) {
                abnormalMachines++;
            }
        }

        overview.setTotalMachines(totalMachines);
        overview.setOnlineMachines(onlineMachines);
        overview.setAbnormalMachines(abnormalMachines);

        // 6. 处理空数据
        if (statsList.isEmpty()) {
            overview.setClearanceRate(0.0);
            overview.setUnbalancedMachines(0);
            return overview;
        }

        // 7. 计算净空达标率和不平衡机数
        long totalRecords = 0;
        long qualifiedDays = 0;
        Set<String> unbalancedDevices = new HashSet<>();

        for (DeviceDailyStats stats : statsList) {
            if (stats.getTotalRecords() != null && stats.getTotalRecords() > 0) {
                totalRecords += stats.getTotalRecords();

                // 净空达标判断
                if (stats.getAvgClearanceValue() != null &&
                    stats.getAvgClearanceValue().doubleValue() >= warningThreshold) {
                    qualifiedDays++;
                }

                // 不平衡判断
                if (stats.getMinClearanceValue() != null &&
                    stats.getMinClearanceValue().doubleValue() < warningThreshold) {
                    unbalancedDevices.add(stats.getDeviceNo());
                }
            }
        }

        overview.setUnbalancedMachines(unbalancedDevices.size());

        // 8. 计算达标率
        long totalDays = statsList.stream()
            .filter(s -> s.getTotalRecords() != null && s.getTotalRecords() > 0)
            .count();

        if (totalDays > 0) {
            overview.setClearanceRate(BigDecimal.valueOf(qualifiedDays * 100.0 / totalDays)
                .setScale(2, RoundingMode.HALF_UP).doubleValue());
        } else {
            overview.setClearanceRate(0.0);
        }

    } catch (Exception e) {
        log.error("获取首页概览统计失败", e);
        // 异常时返回默认值，防止前端渲染失败
        overview.setClearanceRate(0.0);
        overview.setUnbalancedMachines(0);
        overview.setTotalMachines(0);
        overview.setOnlineMachines(0);
        overview.setAbnormalMachines(0);
    }
    return overview;
}

2.3.2 净空趋势计算

@Override
public List<HomeTrendVO> getClearanceTrend(String windFarmCode, String timeRange) {
    try {
        Date endDate = new Date();
        Date startDate = calculateStartDate(timeRange);

        // 从日报表查询数据
        List<DeviceDailyStats> statsList = deviceDailyStatsService.getDailyStats(
            null, windFarmCode, startDate, endDate
        );

        if (statsList.isEmpty()) {
            return new ArrayList<>();
        }

        // 根据时间范围选择计算方式
        if ("day".equals(timeRange)) {
            // 当日小时趋势
            return calculateHourlyTrendFromDailyStats(statsList);
        } else {
            // 周/月趋势（按天）
            return calculateDailyTrend(statsList);
        }
    } catch (Exception e) {
        log.error("获取净空趋势数据失败", e);
        return new ArrayList<>();
    }
}

private List<HomeTrendVO> calculateHourlyTrendFromDailyStats(List<DeviceDailyStats> statsList) {
    HomeTrendVO vo = new HomeTrendVO();
    vo.setTime(new SimpleDateFormat("HH:00").format(new Date()));
    
    // 计算平均值
    double avg = statsList.stream()
        .filter(s -> s.getAvgClearanceValue() != null)
        .mapToDouble(s -> s.getAvgClearanceValue().doubleValue())
        .average().orElse(0.0);
    vo.setValue(Math.round(avg * 100.0) / 100.0);

    List<HomeTrendVO> result = new ArrayList<>();
    result.add(vo);
    return result;
}

private List<HomeTrendVO> calculateDailyTrend(List<DeviceDailyStats> statsList) {
    // 按日期分组
    Map<String, List<DeviceDailyStats>> groupedByDate = statsList.stream()
        .filter(s -> s.getStatsDate() != null)
        .collect(Collectors.groupingBy(
            s -> new SimpleDateFormat("MM-dd").format(s.getStatsDate())
        ));

    List<HomeTrendVO> result = new ArrayList<>();
    for (Map.Entry<String, List<DeviceDailyStats>> entry : groupedByDate.entrySet()) {
        HomeTrendVO vo = new HomeTrendVO();
        vo.setTime(entry.getKey());
        
        // 计算该日平均值
        double avg = entry.getValue().stream()
            .filter(s -> s.getAvgClearanceValue() != null)
            .mapToDouble(s -> s.getAvgClearanceValue().doubleValue())
            .average().orElse(0.0);
        vo.setValue(Math.round(avg * 100.0) / 100.0);
        result.add(vo);
    }

    // 按时间排序
    result.sort(Comparator.comparing(HomeTrendVO::getTime));
    return result;
}

2.3.3 不平衡等级计算（最新实现）

@Override
public List<HomeDistributionVO> getUnbalanceDistribution(String windFarmCode, String timeRange) {
    try {
        Date endDate = new Date();
        Date startDate = calculateStartDate(timeRange);

        // 从 device_daily_stats 表查询统计数据
        List<DeviceDailyStats> statsList = deviceDailyStatsService.getDailyStatsByWindFarm(
            windFarmCode, startDate, endDate
        );

        log.info("不平衡分布查询：windFarmCode={}, timeRange={}, statsCount={}",
            windFarmCode, timeRange, statsList.size());

        if (statsList.isEmpty()) {
            return getDefaultDistribution();
        }

        // 直接累加等级统计字段（不再重新计算）
        int level1Count = statsList.stream()
            .mapToInt(s -> s.getExcellentCount() != null ? s.getExcellentCount() : 0)
            .sum();
        
        int level2Count = statsList.stream()
            .mapToInt(s -> s.getGoodCount() != null ? s.getGoodCount() : 0)
            .sum();
        
        int level3Count = statsList.stream()
            .mapToInt(s -> s.getMediumCount() != null ? s.getMediumCount() : 0)
            .sum();
        
        int level4Count = statsList.stream()
            .mapToInt(s -> s.getWarningCount() != null ? s.getWarningCount() : 0)
            .sum();
        
        int level5Count = statsList.stream()
            .mapToInt(s -> s.getDangerCount() != null ? s.getDangerCount() : 0)
            .sum();

        // 构建返回结果（确保 1-5 级都有数据）
        List<HomeDistributionVO> result = new ArrayList<>();
        
        HomeDistributionVO vo1 = new HomeDistributionVO();
        vo1.setItem("1 级");
        vo1.setCount(level1Count);
        result.add(vo1);
        
        HomeDistributionVO vo2 = new HomeDistributionVO();
        vo2.setItem("2 级");
        vo2.setCount(level2Count);
        result.add(vo2);
        
        HomeDistributionVO vo3 = new HomeDistributionVO();
        vo3.setItem("3 级");
        vo3.setCount(level3Count);
        result.add(vo3);
        
        HomeDistributionVO vo4 = new HomeDistributionVO();
        vo4.setItem("4 级");
        vo4.setCount(level4Count);
        result.add(vo4);
        
        HomeDistributionVO vo5 = new HomeDistributionVO();
        vo5.setItem("5 级");
        vo5.setCount(level5Count);
        result.add(vo5);

        log.info("不平衡分布结果：1 级={}, 2 级={}, 3 级={}, 4 级={}, 5 级={}", 
            level1Count, level2Count, level3Count, level4Count, level5Count);
        return result;
    } catch (Exception e) {
        log.error("获取不平衡分布数据失败", e);
        return getDefaultDistribution();
    }
}
💡 关键设计点（2026 年 5 月更新）：
  • 性能优化：直接使用 device_daily_stats 表中预计算的等级统计字段 （excellentCount, goodCount, mediumCount, warningCount, dangerCount）
  • 不再重新计算：避免了重复的阈值判断和等级计算逻辑，提升查询性能
  • 数据补全：即使某等级数量为 0，也返回完整数据（1-5 级）
  • 日志完善：添加详细的日志记录，便于排查问题

2.4 数据访问层 HomeStatsMapper

Mapper 层使用 MyBatis @Select 注解直接编写 SQL，主要用于告警统计查询。 

@Mapper
public interface HomeStatsMapper {

    /**
     * 获取告警类型统计
     * 关联字典表获取告警类型名称
     */
    @Select("SELECT " +
            "COALESCE(d.item_text, w.alarm_type) as type, " +
            "COUNT(*) as count " +
            "FROM warn_info w " +
            "LEFT JOIN sys_dict_item d ON d.item_value = w.alarm_type " +
            "  AND d.dict_id = (SELECT id FROM sys_dict WHERE dict_code = 'alarm_type' LIMIT 1) " +
            "WHERE w.alarm_time >= #{startTime} AND w.alarm_time < #{endTime} " +
            "GROUP BY w.alarm_type, d.item_text " +
            "ORDER BY count DESC")
    List<HomeAlarmStatsVO> selectAlarmStats(
        @Param("startTime") Date startTime,
        @Param("endTime") Date endTime
    );
}
📌 SQL 设计要点：
  • LEFT JOIN：即使字典表中没有配置，也能返回告警类型（使用原始值）
  • COALESCE：确保 type 字段不为 NULL
  • 时间范围：使用 >= startTime AND < endTime 避免时间边界问题

4.1 API 请求封装

前端使用 Axios 封装了统一的 HTTP 请求方法，包含签名认证。

// manage.js - API 请求封装
import Vue from 'vue'
import { axios } from '@/utils/request'
import signMd5Utils from '@/utils/encryption/signMd5Utils'

/**
 * GET 请求
 * @param url 请求路径
 * @param parameter 请求参数
 */
export function getAction(url, parameter) {
  // 生成签名（防止请求被篡改）
  let sign = signMd5Utils.getSign(url, parameter);
  let signHeader = {
    "X-Sign": sign,
    "X-TIMESTAMP": signMd5Utils.getTimestamp()
  };
  
  return axios({
    url: url,
    method: 'get',
    params: parameter,
    headers: signHeader
  })
}

/**
 * POST 请求
 */
export function postAction(url, parameter) {
  let sign = signMd5Utils.getSign(url, parameter);
  let signHeader = {
    "X-Sign": sign,
    "X-TIMESTAMP": signMd5Utils.getTimestamp()
  };
  
  return axios({
    url: url,
    method: 'post',
    data: parameter,
    headers: signHeader
  })
}
📖 签名认证机制
  • X-Sign：使用 MD5 算法对 URL+ 参数 + 时间戳生成签名，防止请求被篡改
  • X-TIMESTAMP：请求时间戳，后端验证时间戳防止重放攻击
  • 统一拦截：Axios 拦截器统一添加 Token 到请求头

第五章：核心功能详解

5.1 净空达标率计算

5.1.1 计算公式

净空达标率 = (达标天数 / 总天数) × 100%

其中：
- 达标天数：avg_clearance_value >= warningThreshold 的天数
- 总天数：stats 记录总数（有数据的记录）

5.1.2 实现代码

// 服务层实现（HomeStatsServiceImpl）
long totalRecords = 0;
long qualifiedDays = 0;
Set<String> unbalancedDevices = new HashSet<>();

for (DeviceDailyStats stats : statsList) {
    if (stats.getTotalRecords() != null && stats.getTotalRecords() > 0) {
        totalRecords += stats.getTotalRecords();

        // 净空达标判断
        if (stats.getAvgClearanceValue() != null &&
            stats.getAvgClearanceValue().doubleValue() >= warningThreshold) {
            qualifiedDays++;
        }

        // 不平衡判断
        if (stats.getMinClearanceValue() != null &&
            stats.getMinClearanceValue().doubleValue() < warningThreshold) {
            unbalancedDevices.add(stats.getDeviceNo());
        }
    }
}

long totalDays = statsList.stream()
    .filter(s -> s.getTotalRecords() != null && s.getTotalRecords() > 0)
    .count();

if (totalDays > 0) {
    overview.setClearanceRate(BigDecimal.valueOf(qualifiedDays * 100.0 / totalDays)
        .setScale(2, RoundingMode.HALF_UP).doubleValue());
} else {
    overview.setClearanceRate(0.0);
}
📌 阈值配置：
  • 阈值从 sys_custom_config 表的 clearance_level_threshold 配置读取
  • 配置格式：JSON {"excellent":7.8, "good":6.5, "medium":5.5, "warning":4.2, "danger":0.0}
  • 配置不存在时使用默认值：warning = 4.2 米

5.2 不平衡等级计算

5.2.1 不平衡等级

等级 说明 净空值范围
1 级 优秀 >= excellent (7.8 米)
2 级 良好 >= good (6.5 米)
3 级 一般 >= medium (5.5 米)
4 级 警告 >= warning (4.2 米)
5 级 危险 < warning (4.2 米)

5.2.2 统计逻辑

private int calculateUnbalanceLevel(BigDecimal avgClearance, ClearanceThresholdVO thresholds) {
    double value = avgClearance.doubleValue();

    if (value >= thresholds.getExcellent()) {
        return 1;  // 优秀
    } else if (value >= thresholds.getGood()) {
        return 2;  // 良好
    } else if (value >= thresholds.getMedium()) {
        return 3;  // 一般
    } else if (value >= thresholds.getWarning()) {
        return 4;  // 警告
    } else {
        return 5;  // 危险
    }
}
💡 数据补全机制：
  • 后端确保返回 1-5 级的完整数据，即使某级数量为 0
  • 前端饼图能正确显示所有等级，避免缺失扇区
  • 使用 HashMap 初始化为所有等级设置 0 值

        Date startDate = calculateStartDate(timeRange);

        // 1. 定义完整的告警类型列表
        List<String> allTypes = Arrays.asList(
            "数据质量", "噪声过大", "通讯异常", "光照异常", "电机故障", "对焦失散", "轴承故障"
        );

        // 2. 初始化计数器
        Map<String, Integer> typeMap = new HashMap<>();
        for (String type : allTypes) {
            typeMap.put(type, 0);
        }

        // 3. 从 record_info 表统计"数据质量"告警
        List<RecordInfo> recordList = shardingRecordService.queryRecordInfoByWindFarm(
            null, null, startDate, endDate
        );

        for (RecordInfo record : recordList) {
            if (record.getIsAbnormal() != null && record.getIsAbnormal() == 1) {
                typeMap.merge("数据质量", 1, Integer::sum);
            }
        }

        // 4. 从 warn_info 表统计其他类型告警
        List<HomeAlarmStatsVO> warnResult = homeStatsMapper.selectAlarmStats(startDate, endDate);
        if (warnResult != null && !warnResult.isEmpty()) {
            for (HomeAlarmStatsVO warnVo : warnResult) {
                if (warnVo.getType() != null && !warnVo.getType().equals("数据质量")) {
                    typeMap.merge(warnVo.getType(), warnVo.getCount(), Integer::sum);
                }
            }
        }

        // 5. 补充缺失的类型（确保 7 种类型都有数据）
        List<HomeAlarmStatsVO> fullResult = new ArrayList<>();
        for (String type : allTypes) {
            HomeAlarmStatsVO vo = new HomeAlarmStatsVO();
            vo.setType(type);
            vo.setCount(typeMap.getOrDefault(type, 0));
            fullResult.add(vo);
        }

        log.info("告警统计完整结果：{}", fullResult);
        return fullResult;
    } catch (Exception e) {
        log.error("获取告警统计数据失败", e);

private Date calculateStartDate(String timeRange) {
    LocalDate today = LocalDate.now();
    LocalDate startLocalDate;

    switch (timeRange) {
        case "today":
        case "day":
            startLocalDate = today;  // 今天
            break;
        case "history":
            // history 默认查询最近 7 天（兼容前端逻辑）
            startLocalDate = today.minusDays(7);
            break;
        case "week":
            startLocalDate = today.minusDays(7);  // 7 天前
            break;
        case "month":
            startLocalDate = today.minusDays(30);  // 30 天前
            break;
        default:
            startLocalDate = today;
    }

    return Date.from(startLocalDate.atStartOfDay(ZoneId.systemDefault()).toInstant());
}
💡 时间处理技巧：
  • 使用 LocalDate：Java 8 新 API，更精确的时间处理
  • 时区处理：使用 ZoneId.systemDefault() 确保时区正确
  • 统一边界：查询范围是 startDate <= time < endDate（endDate 为当前时间）
  • history 参数：2026 年 5 月更新，默认查询 7 天数据
