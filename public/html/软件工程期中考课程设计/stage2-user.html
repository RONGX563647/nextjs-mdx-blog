阶段二：用户模块

一、用户模块概述

用户模块是整个系统的基础入口，负责用户注册、登录以及自动创建专属区块链节点。一个用户对应一个区块链节点，这是本系统的核心设计原则。

1.1 功能需求

  • 用户注册：支持用户账号注册，注册完成后自动创建专属区块链节点
  • 用户登录：用户通过账号密码登录系统，进入可视化操作主界面
  • 节点绑定：一个用户对应一个区块链节点，实现用户-节点绑定关系

1.2 核心流程

用户注册流程：
1. 用户填写注册信息（用户名、密码）
2. 后端验证用户信息
3. 创建用户账户
4. 自动生成区块链钱包（RSA密钥对）
5. 创建专属区块链节点
6. 返回用户信息和节点信息

用户登录流程：
1. 用户输入账号密码
2. 后端验证 credentials
3. 返回用户信息、节点信息、钱包信息
4. 前端跳转到主界面

二、用户注册实现

2.1 用户实体类

User.java - 用户数据模型：

@Data
@Entity
@Table(name = "users")
public class User {
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(unique = true, nullable = false)
    private String username;

    @Column(nullable = false)
    private String password;  // BCrypt 加密存储

    @Column(unique = true, length = 40)
    private String walletAddress;  // 区块链钱包地址

    @Column(unique = true)
    private String nodeId;  // 关联的节点 ID

    @Column(columnDefinition = "TEXT")
    private String publicKey;  // RSA 公钥

    @Column(columnDefinition = "TEXT")
    private String privateKey;  // RSA 私钥

    @Column
    private Double balance = 0.0;  // 钱包余额

    @Column
    private LocalDateTime createdAt;
}

说明：

  • 使用 Lombok 的 @Data 注解自动生成 getter/setter 方法
  • password 字段使用 BCrypt 加密存储
  • walletAddress 和 nodeId 设置为唯一索引
  • publicKey 和 privateKey 使用 TEXT 类型存储长字符串

2.2 注册 Controller

UserController.java - 用户注册接口：

@RestController
@RequestMapping("/api/user")
public class UserController {

    @Autowired
    private UserService userService;

    @PostMapping("/register")
    public ResponseEntity<Map<String, Object>> register(@RequestBody Map<String, String> request) {
        String username = request.get("username");
        String password = request.get("password");

        // 验证输入
        if (username == null || username.trim().isEmpty()) {
            return ResponseEntity.badRequest()
                .body(Map.of("status", "error", "message", "用户名不能为空"));
        }
        if (password == null || password.trim().isEmpty()) {
            return ResponseEntity.badRequest()
                .body(Map.of("status", "error", "message", "密码不能为空"));
        }

        // 检查用户名是否已存在
        if (userService.existsByUsername(username)) {
            return ResponseEntity.badRequest()
                .body(Map.of("status", "error", "message", "用户名已存在"));
        }

        // 执行注册
        User user = userService.register(username, password);

        Map<String, Object> response = new HashMap<>();
        response.put("status", "success");
        response.put("message", "注册成功");
        response.put("user", Map.of(
            "id", user.getId(),
            "username", user.getUsername(),
            "walletAddress", user.getWalletAddress(),
            "nodeId", user.getNodeId(),
            "balance", user.getBalance()
        ));
        return ResponseEntity.ok(response);
    }
}

2.3 注册 Service

UserService.java - 用户注册业务逻辑：

@Service
public class UserService {

    @Autowired
    private UserRepository userRepository;

    private PasswordEncoder passwordEncoder = new BCryptPasswordEncoder();

    public User register(String username, String password) {
        User user = new User();
        user.setUsername(username);
        user.setPassword(passwordEncoder.encode(password));
        user.setCreatedAt(LocalDateTime.now());

        // 生成区块链钱包
        Wallet wallet = new Wallet();
        user.setWalletAddress(wallet.getAddress());
        user.setPublicKey(wallet.getPublicKey());
        user.setPrivateKey(wallet.getPrivateKey());
        user.setBalance(0.0);

        // 生成节点 ID
        String nodeId = UUID.randomUUID().toString();
        user.setNodeId(nodeId);

        // 保存用户
        userRepository.save(user);

        return user;
    }

    public User login(String username, String password) {
        User user = userRepository.findByUsername(username)
            .orElse(null);

        if (user == null) {
            return null;
        }

        if (!passwordEncoder.matches(password, user.getPassword())) {
            return null;
        }

        return user;
    }

    public boolean existsByUsername(String username) {
        return userRepository.existsByUsername(username);
    }
}

2.4 钱包实现

Wallet.java - 区块链钱包生成：

public class Wallet {
    private String address;
    private String publicKey;
    private String privateKey;
    private KeyPair keyPair;

    public Wallet() {
        try {
            // 生成 RSA 密钥对
            KeyPairGenerator keyGen = KeyPairGenerator.getInstance("RSA");
            keyGen.initialize(2048);
            this.keyPair = keyGen.generateKeyPair();

            // 将公钥和私钥转换为 Base64 字符串
            this.publicKey = Base64.getEncoder()
                .encodeToString(keyPair.getPublic().getEncoded());
            this.privateKey = Base64.getEncoder()
                .encodeToString(keyPair.getPrivate().getEncoded());

            // 钱包地址由公钥的 SHA-256 哈希值生成
            this.address = HashUtil.sha256(this.publicKey).substring(0, 40);
        } catch (Exception e) {
            e.printStackTrace();
        }
    }

    public String signTransaction(String data) {
        try {
            Signature signature = Signature.getInstance("SHA256withRSA");
            signature.initSign(keyPair.getPrivate());
            signature.update(data.getBytes());
            return Base64.getEncoder().encodeToString(signature.sign());
        } catch (Exception e) {
            e.printStackTrace();
            return null;
        }
    }

    public boolean verifySignature(String data, String signatureStr, String publicKeyStr) {
        try {
            byte[] publicKeyBytes = Base64.getDecoder().decode(publicKeyStr);
            X509EncodedKeySpec keySpec = new X509EncodedKeySpec(publicKeyBytes);
            KeyFactory keyFactory = KeyFactory.getInstance("RSA");
            PublicKey publicKey = keyFactory.generatePublic(keySpec);

            Signature signature = Signature.getInstance("SHA256withRSA");
            signature.initVerify(publicKey);
            signature.update(data.getBytes());
            return signature.verify(Base64.getDecoder().decode(signatureStr));
        } catch (Exception e) {
            e.printStackTrace();
            return false;
        }
    }

    public String getAddress() { return address; }
    public String getPublicKey() { return publicKey; }
    public String getPrivateKey() { return privateKey; }
}

说明：

  • 钱包生成：使用 RSA 2048 位密钥对生成钱包地址和密钥
  • 交易签名：使用私钥对交易数据进行签名，保证交易真实性
  • 签名验证：使用公钥验证签名，确保交易未被篡改
  • 地址生成：对公钥进行 SHA-256 哈希，取前 40 位作为钱包地址

        "walletAddress", user.getWalletAddress(),
        "nodeId", user.getNodeId(),
        "balance", user.getBalance()
    ));
    return ResponseEntity.ok(response);
}

四、节点自动创建

4.1 节点创建流程

用户注册时，系统会自动为该用户创建一个专属区块链节点。节点创建流程如下：

  1. 生成唯一的节点 ID (UUID)
  2. 初始化该节点的区块链（包含创世区块）
  3. 将节点与用户钱包地址绑定
  4. 记录节点信息到数据库

4.2 节点服务实现

NodeService.java - 节点管理服务：

@Service
public class NodeService {

    public NodeInfo createNodeForUser(User user, Blockchain blockchain) {
        NodeInfo nodeInfo = new NodeInfo();
        nodeInfo.setNodeId(user.getNodeId());
        nodeInfo.setHost("localhost");
        nodeInfo.setPort(8080);
        nodeInfo.setWalletAddress(user.getWalletAddress());
        nodeInfo.setCreatedAt(LocalDateTime.now());
        nodeInfo.setChainLength(blockchain.getChain().size());
        nodeInfo.setPeers(new ArrayList<>());

        return nodeInfo;
    }

    public NodeInfo getNodeInfo(User user, Blockchain blockchain) {
        NodeInfo nodeInfo = new NodeInfo();
        nodeInfo.setNodeId(user.getNodeId());
        nodeInfo.setHost("localhost");
        nodeInfo.setPort(8080);
        nodeInfo.setWalletAddress(user.getWalletAddress());
        nodeInfo.setChainLength(blockchain.getChain().size());
        nodeInfo.setPeers(new ArrayList<>());
        
        return nodeInfo;
    }
}

说明：

  • createNodeForUser：用户注册时创建节点信息，初始化区块链
  • getNodeInfo：获取用户节点当前状态信息
  • 节点信息包含节点 ID、主机地址、端口、钱包地址、链长度等

<script setup>
import { reactive } from 'vue'
import { useRouter } from 'vue-router'
import { userApi } from '../api/user'
import { ElMessage } from 'element-plus'

const router = useRouter()

const registerForm = reactive({
  username: '',
  password: '',
  confirmPassword: ''
})

function goToLogin() {
  router.push('/login')
}

async function handleRegister() {
  if (!registerForm.username || !registerForm.password) {
    ElMessage.warning('请输入用户名和密码')
    return
  }
  
  if (registerForm.password !== registerForm.confirmPassword) {
    ElMessage.warning('两次密码输入不一致')
    return
  }
  
  try {
    const response = await userApi.register({
      username: registerForm.username,
      password: registerForm.password
    })
    if (response.data.status === 'success') {
      ElMessage.success('注册成功，请登录')
      router.push('/login')
    }
  } catch (error) {
    ElMessage.error(error.response?.data?.message || '注册失败')
  }
}
</script>

import { reactive } from 'vue'

  username: '',
  password: ''
})
