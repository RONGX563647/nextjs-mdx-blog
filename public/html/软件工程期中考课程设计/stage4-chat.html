阶段四：即时聊天功能

一、即时聊天功能概述

即时聊天交流平台基于 RabbitMQ 生产者-消费者模式实现，为区块链平台所有注册用户（节点成员）提供在线交流功能，用于商讨交易相关事宜。

1.1 功能需求

  • 通信主体：区块链平台所有注册用户（节点成员）
  • 通信内容：成员之间在线交流、聊天，用于商讨交易相关事宜
  • 通信技术：基于 RabbitMQ 生产者-消费者模式实现消息收发

1.2 架构设计

用户A (生产者)                    RabbitMQ                    用户B (消费者)
+-------------+              +-------------------+              +-------------+
| ChatProducer | ---消息---> |    Exchange       | ---消息---> | ChatConsumer |
|              |              |                   |              |              |
| 发送消息      |              |  消息路由/分发     |              | 接收消息      |
| 到队列        |              |                   |              | 推送给前端    |
+-------------+              +-------------------+              +-------------+
                                   |
                            +--------------+
                            |    Queue     |
                            |  消息存储     |
                            +--------------+

1.3 消息流转过程

  1. 用户在前端输入聊天消息并发送
  2. 前端通过 HTTP API 将消息发送到后端 ChatController
  3. ChatController 调用 ChatProducer 将消息发送到 RabbitMQ
  4. RabbitMQ 将消息存储到对应队列
  5. ChatConsumer 监听队列，接收到消息后处理
  6. 处理后的消息通过 WebSocket 或 SSE 推送给在线用户

2.3 RabbitMQConfig 配置类

@Configuration
public class RabbitMQConfig {
    // 交换机名称
    public static final String DIRECT_EXCHANGE = "chat-direct-exchange";
    public static final String TOPIC_EXCHANGE = "chat-topic-exchange";
    public static final String FANOUT_EXCHANGE = "chat-fanout-exchange";

    // 队列名称
    public static final String CHAT_QUEUE = "chat-queue";
    public static final String INFO_QUEUE = "info-queue";
    public static final String WARNING_QUEUE = "warning-queue";
    public static final String ERROR_QUEUE = "error-queue";
    public static final String ORDER_QUEUE = "order-queue";
    public static final String NOTIFICATION_QUEUE = "notification-queue";

    // 路由键
    public static final String INFO_ROUTING_KEY = "route.info";
    public static final String WARNING_ROUTING_KEY = "route.warning";
    public static final String ERROR_ROUTING_KEY = "route.error";
    public static final String ORDER_ROUTING_KEY = "route.order";
    public static final String NOTIFICATION_ROUTING_KEY = "route.notification";

    // 创建 Direct 交换机
    @Bean
    public DirectExchange directExchange() {
        return new DirectExchange(DIRECT_EXCHANGE, true, false);
    }

    // 创建 Topic 交换机
    @Bean
    public TopicExchange topicExchange() {
        return new TopicExchange(TOPIC_EXCHANGE, true, false);
    }

    // 创建 Fanout 交换机
    @Bean
    public FanoutExchange fanoutExchange() {
        return new FanoutExchange(FANOUT_EXCHANGE, true, false);
    }

    // 创建聊天队列
    @Bean
    public Queue chatQueue() {
        return new Queue(CHAT_QUEUE, true);
    }

    @Bean
    public Queue infoQueue() {
        return new Queue(INFO_QUEUE, true);
    }

    @Bean
    public Queue warningQueue() {
        return new Queue(WARNING_QUEUE, true);
    }

    @Bean
    public Queue errorQueue() {
        return new Queue(ERROR_QUEUE, true);
    }

    @Bean
    public Queue orderQueue() {
        return new Queue(ORDER_QUEUE, true);
    }

    @Bean
    public Queue notificationQueue() {
        return new Queue(NOTIFICATION_QUEUE, true);
    }

    // 绑定关系
    @Bean
    public Binding chatBinding(FanoutExchange fanoutExchange, Queue chatQueue) {
        return BindingBuilder.bind(chatQueue).to(fanoutExchange);
    }

    @Bean
    public Binding infoBinding(DirectExchange directExchange, Queue infoQueue) {
        return BindingBuilder.bind(infoQueue).to(directExchange).with(INFO_ROUTING_KEY);
    }

    @Bean
    public Binding warningBinding(DirectExchange directExchange, Queue warningQueue) {
        return BindingBuilder.bind(warningQueue).to(directExchange).with(WARNING_ROUTING_KEY);
    }

    @Bean
    public Binding errorBinding(DirectExchange directExchange, Queue errorQueue) {
        return BindingBuilder.bind(errorQueue).to(directExchange).with(ERROR_ROUTING_KEY);
    }

    @Bean
    public Binding orderBinding(TopicExchange topicExchange, Queue orderQueue) {
        return BindingBuilder.bind(orderQueue).to(topicExchange).with("order.*");
    }

    @Bean
    public Binding notificationBinding(FanoutExchange fanoutExchange, Queue notificationQueue) {
        return BindingBuilder.bind(notificationQueue).to(fanoutExchange);
    }

    // JSON 消息转换器
    @Bean
    public Jackson2JsonMessageConverter jsonMessageConverter() {
        ObjectMapper objectMapper = new ObjectMapper();
        objectMapper.registerModule(new JavaTimeModule());
        objectMapper.disable(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS);
        return new Jackson2JsonMessageConverter(objectMapper);
    }

    // RabbitTemplate
    @Bean
    public RabbitTemplate rabbitTemplate(ConnectionFactory connectionFactory) {
        RabbitTemplate rabbitTemplate = new RabbitTemplate(connectionFactory);
        rabbitTemplate.setMessageConverter(jsonMessageConverter());
        return rabbitTemplate;
    }

    // 消息监听容器工厂
    @Bean
    public SimpleRabbitListenerContainerFactory rabbitListenerContainerFactory(
            ConnectionFactory connectionFactory) {
        SimpleRabbitListenerContainerFactory factory = 
            new SimpleRabbitListenerContainerFactory();
        factory.setConnectionFactory(connectionFactory);
        factory.setMessageConverter(jsonMessageConverter());
        factory.setAcknowledgeMode(AcknowledgeMode.AUTO);
        return factory;
    }

    @Bean
    public RabbitAdmin rabbitAdmin(ConnectionFactory connectionFactory) {
        return new RabbitAdmin(connectionFactory);
    }
}

2.4 三种交换机类型对比

交换机类型 路由方式 使用场景
Direct Exchange 精确匹配路由键 INFO、WARNING、ERROR 消息按类型分发
Topic Exchange 通配符匹配路由键 ORDER 消息支持 order.create、order.update 等子类型
Fanout Exchange 广播到所有绑定队列 聊天消息、NOTIFICATION 消息全员接收

三、消息数据模型

3.1 MessageDTO

MessageDTO.java - 消息数据传输对象：

public class MessageDTO implements Serializable {

    private static final long serialVersionUID = 1L;

    private String messageId;
    private String content;
    private String sender;
    private MessageType type;
    private LocalDateTime timestamp;

    public MessageDTO() {
    }

    public MessageDTO(String messageId, String content, String sender, MessageType type) {
        this.messageId = messageId;
        this.content = content;
        this.sender = sender;
        this.type = type;
        this.timestamp = LocalDateTime.now();
    }

    // 静态工厂方法
    public static MessageDTO createChatMessage(String content, String sender) {
        return new MessageDTO(
            UUID.randomUUID().toString(),
            content,
            sender,
            MessageType.CHAT
        );
    }

    public static MessageDTO createInfoMessage(String content, String sender) {
        return new MessageDTO(
            UUID.randomUUID().toString(),
            content,
            sender,
            MessageType.INFO
        );
    }

    public static MessageDTO createWarningMessage(String content, String sender) {
        return new MessageDTO(
            UUID.randomUUID().toString(),
            content,
            sender,
            MessageType.WARNING
        );
    }

    public static MessageDTO createErrorMessage(String content, String sender) {
        return new MessageDTO(
            UUID.randomUUID().toString(),
            content,
            sender,
            MessageType.ERROR
        );
    }

    public static MessageDTO createOrderMessage(String content, String sender) {
        return new MessageDTO(
            UUID.randomUUID().toString(),
            content,
            sender,
            MessageType.ORDER
        );
    }

    public static MessageDTO createNotificationMessage(String content, String sender) {
        return new MessageDTO(
            UUID.randomUUID().toString(),
            content,
            sender,
            MessageType.NOTIFICATION
        );
    }

    // 消息类型枚举
    public enum MessageType {
        INFO,       // 信息消息
        WARNING,    // 警告消息
        ERROR,      // 错误消息
        ORDER,      // 订单消息
        NOTIFICATION, // 通知消息
        CHAT        // 聊天消息
    }

    // Getters and Setters
    public String getMessageId() { return messageId; }
    public void setMessageId(String messageId) { this.messageId = messageId; }
    public String getContent() { return content; }
    public void setContent(String content) { this.content = content; }
    public String getSender() { return sender; }
    public void setSender(String sender) { this.sender = sender; }
    public MessageType getType() { return type; }
    public void setType(MessageType type) { this.type = type; }
    public LocalDateTime getTimestamp() { return timestamp; }
    public void setTimestamp(LocalDateTime timestamp) { this.timestamp = timestamp; }

    @Override
    public String toString() {
        return "MessageDTO{" +
                "messageId='" + messageId + '\'' +
                ", content='" + content + '\'' +
                ", sender='" + sender + '\'' +
                ", type=" + type +
                ", timestamp=" + timestamp +
                '}';
    }
}

说明：

  • 实现 Serializable 接口支持网络传输
  • 提供 6 个静态工厂方法，方便创建不同类型的消息
  • MessageType 枚举定义了 6 种消息类型
  • 重写 toString() 方法方便日志输出

四、消息生产者

4.1 ChatProducer

@Component
public class ChatProducer {

    private static final Logger logger = LoggerFactory.getLogger(ChatProducer.class);

    @Autowired
    private RabbitTemplate rabbitTemplate;

    // 发送聊天消息（广播模式）
    public void sendChatMessage(String content, String sender) {
        MessageDTO message = new MessageDTO(
            UUID.randomUUID().toString(),
            content,
            sender,
            MessageDTO.MessageType.CHAT
        );
        rabbitTemplate.convertAndSend(
            RabbitMQConfig.FANOUT_EXCHANGE,
            "",
            message
        );
        logger.info("发送聊天消息: {} -> {}", sender, content);
    }

    // 发送 INFO 消息
    public void sendInfoMessage(String content, String sender) {
        MessageDTO message = new MessageDTO(
            UUID.randomUUID().toString(),
            content,
            sender,
            MessageDTO.MessageType.INFO
        );
        rabbitTemplate.convertAndSend(
            RabbitMQConfig.DIRECT_EXCHANGE,
            RabbitMQConfig.INFO_ROUTING_KEY,
            message
        );
        logger.info("发送INFO消息: {}", message);
    }

    // 发送 WARNING 消息
    public void sendWarningMessage(String content, String sender) {
        MessageDTO message = new MessageDTO(
            UUID.randomUUID().toString(),
            content,
            sender,
            MessageDTO.MessageType.WARNING
        );
        rabbitTemplate.convertAndSend(
            RabbitMQConfig.DIRECT_EXCHANGE,
            RabbitMQConfig.WARNING_ROUTING_KEY,
            message
        );
        logger.info("发送WARNING消息: {}", message);
    }

    // 发送 ERROR 消息
    public void sendErrorMessage(String content, String sender) {
        MessageDTO message = new MessageDTO(
            UUID.randomUUID().toString(),
            content,
            sender,
            MessageDTO.MessageType.ERROR
        );
        rabbitTemplate.convertAndSend(
            RabbitMQConfig.DIRECT_EXCHANGE,
            RabbitMQConfig.ERROR_ROUTING_KEY,
            message
        );
        logger.info("发送ERROR消息: {}", message);
    }

    // 发送 ORDER 消息
    public void sendOrderMessage(String content, String sender) {
        MessageDTO message = new MessageDTO(
            UUID.randomUUID().toString(),
            content,
            sender,
            MessageDTO.MessageType.ORDER
        );
        rabbitTemplate.convertAndSend(
            RabbitMQConfig.TOPIC_EXCHANGE,
            "order.create",
            message
        );
        logger.info("发送ORDER消息: {}", message);
    }

    // 发送 NOTIFICATION 消息
    public void sendNotificationMessage(String content, String sender) {
        MessageDTO message = new MessageDTO(
            UUID.randomUUID().toString(),
            content,
            sender,
            MessageDTO.MessageType.NOTIFICATION
        );
        rabbitTemplate.convertAndSend(
            RabbitMQConfig.FANOUT_EXCHANGE,
            "",
            message
        );
        logger.info("发送NOTIFICATION消息: {}", message);
    }

    // 通用发送方法
    public void sendMessage(MessageDTO.MessageType type, String content, String sender) {
        switch (type) {
            case INFO:
                sendInfoMessage(content, sender);
                break;
            case WARNING:
                sendWarningMessage(content, sender);
                break;
            case ERROR:
                sendErrorMessage(content, sender);
                break;
            case ORDER:
                sendOrderMessage(content, sender);
                break;
            case NOTIFICATION:
                sendNotificationMessage(content, sender);
                break;
            case CHAT:
                sendChatMessage(content, sender);
                break;
            default:
                logger.warn("未知消息类型: {}", type);
        }
    }
}

五、消息消费者

5.1 ChatConsumer

@Component
public class ChatConsumer {

    private static final Logger logger = LoggerFactory.getLogger(ChatConsumer.class);

    // 监听聊天队列（Fanout 广播模式）
    @RabbitListener(queues = RabbitMQConfig.CHAT_QUEUE, 
                    containerFactory = "rabbitListenerContainerFactory")
    public void receiveChatMessage(MessageDTO message) {
        processMessage("CHAT", message);
    }

    // 监听 INFO 队列
    @RabbitListener(queues = RabbitMQConfig.INFO_QUEUE, 
                    containerFactory = "rabbitListenerContainerFactory")
    public void receiveInfoMessage(MessageDTO message) {
        processMessage("INFO", message);
    }

    // 监听 WARNING 队列
    @RabbitListener(queues = RabbitMQConfig.WARNING_QUEUE, 
                    containerFactory = "rabbitListenerContainerFactory")
    public void receiveWarningMessage(MessageDTO message) {
        processMessage("WARNING", message);
    }

    // 监听 ERROR 队列
    @RabbitListener(queues = RabbitMQConfig.ERROR_QUEUE, 
                    containerFactory = "rabbitListenerContainerFactory")
    public void receiveErrorMessage(MessageDTO message) {
        processMessage("ERROR", message);
    }

    // 监听 ORDER 队列
    @RabbitListener(queues = RabbitMQConfig.ORDER_QUEUE, 
                    containerFactory = "rabbitListenerContainerFactory")
    public void receiveOrderMessage(MessageDTO message) {
        processMessage("ORDER", message);
    }

    // 监听 NOTIFICATION 队列
    @RabbitListener(queues = RabbitMQConfig.NOTIFICATION_QUEUE, 
                    containerFactory = "rabbitListenerContainerFactory")
    public void receiveNotificationMessage(MessageDTO message) {
        processMessage("NOTIFICATION", message);
    }

    // 统一消息处理方法
    private void processMessage(String type, MessageDTO message) {
        logger.info("========== 接收{}消息 ==========", type);
        logger.info("消息ID: {}", message.getMessageId());
        logger.info("发送者: {}", message.getSender());
        logger.info("内容: {}", message.getContent());
        logger.info("时间: {}", message.getTimestamp());
        executeProcess(type, message);
        logger.info("==================================");
    }

    // 消息处理分发
    private void executeProcess(String type, MessageDTO message) {
        switch (type) {
            case "CHAT":
                logger.info("处理CHAT消息: 推送给在线用户");
                // TODO: 通过 WebSocket/SSE 推送给前端
                break;
            case "INFO":
                logger.info("处理INFO消息: 记录系统日志");
                break;
            case "WARNING":
                logger.info("处理WARNING消息: 发送告警通知");
                break;
            case "ERROR":
                logger.info("处理ERROR消息: 发送邮件通知管理员");
                break;
            case "ORDER":
                logger.info("处理ORDER消息: 更新订单状态");
                break;
            case "NOTIFICATION":
                logger.info("处理NOTIFICATION消息: 推送通知给用户");
                break;
        }
    }
}

5.2 @RabbitListener 注解说明

属性 说明
queues 指定要监听的队列名称
containerFactory 指定消息监听容器工厂，包含消息转换器配置

工作原理：

  • Spring 启动时自动创建消费者监听器
  • 消费者持续监听指定队列
  • 当队列中有消息时，自动从 JSON 反序列化为 MessageDTO 对象
  • 调用对应的方法处理消息
  • 处理完成后自动发送 ACK 确认（AUTO 模式）

六、聊天控制器

6.1 ChatController

@RestController
@RequestMapping("/api/chat")
public class ChatController {

    @Autowired
    private ChatService chatService;

    // 发送聊天消息
    @PostMapping("/send")
    public ResponseEntity<Map<String, String>> sendChatMessage(
            @RequestBody Map<String, String> request) {
        String content = request.get("content");
        String sender = request.getOrDefault("sender", "Anonymous");

        if (content == null || content.trim().isEmpty()) {
            return ResponseEntity.badRequest()
                .body(Map.of("status", "error", "message", "消息内容不能为空"));
        }

        chatService.sendChatMessage(content, sender);
        return ResponseEntity.ok(Map.of(
            "status", "success", 
            "message", "消息已发送"
        ));
    }

    // 获取聊天消息类型列表
    @GetMapping("/types")
    public ResponseEntity<MessageDTO.MessageType[]> getMessageTypes() {
        return ResponseEntity.ok(MessageDTO.MessageType.values());
    }

    // 模拟场景：用户登录通知
    @PostMapping("/simulate/login")
    public ResponseEntity<Map<String, String>> simulateLogin(
            @RequestParam String username) {
        chatService.simulateUserLogin(username);
        return ResponseEntity.ok(Map.of(
            "status", "success", 
            "message", "模拟登录已触发"
        ));
    }

    // 模拟场景：交易讨论
    @PostMapping("/simulate/trade-discuss")
    public ResponseEntity<Map<String, String>> simulateTradeDiscuss(
            @RequestParam String sender,
            @RequestParam String tradeDetail) {
        chatService.simulateTradeDiscuss(sender, tradeDetail);
        return ResponseEntity.ok(Map.of(
            "status", "success", 
            "message", "交易讨论已触发"
        ));
    }
}

6.2 ChatService

@Service
public class ChatService {

    private static final Logger logger = LoggerFactory.getLogger(ChatService.class);

    @Autowired
    private ChatProducer chatProducer;

    // 发送聊天消息
    public void sendChatMessage(String content, String sender) {
        chatProducer.sendChatMessage(content, sender);
        logger.info("聊天消息已发送: {} -> {}", sender, content);
    }

    // 模拟用户登录
    public void simulateUserLogin(String username) {
        chatProducer.sendInfoMessage(
            "用户 [" + username + "] 登录区块链平台", 
            "AuthService"
        );
        chatProducer.sendNotificationMessage(
            "欢迎 " + username + " 加入区块链平台！", 
            "NotificationService"
        );
    }

    // 模拟交易讨论
    public void simulateTradeDiscuss(String sender, String tradeDetail) {
        chatProducer.sendChatMessage(
            "关于交易: " + tradeDetail, 
            sender
        );
        chatProducer.sendOrderMessage(
            "交易记录: " + tradeDetail, 
            "TradeService"
        );
    }

    // 通用消息发送
    public void sendCustomMessage(MessageDTO.MessageType type, 
                                   String content, 
                                   String sender) {
        chatProducer.sendMessage(type, content, sender);
    }
}

8.1 聊天组件

<template>
  <div class="chat-container">

        @keyup.enter="sendMessage"
      />
      <el-button type="primary" @click="sendMessage">发送</el-button>
    </div>
  </div>
</template>

<script setup>
import { ref, onMounted } from 'vue'
import axios from 'axios'

const messages = ref([])
const newMessage = ref('')
const currentUser = ref(localStorage.getItem('username') || 'guest')

// 发送消息
async function sendMessage() {
  if (!newMessage.value.trim()) return
  
  try {
    await axios.post('/api/chat/send', {
      content: newMessage.value,
      sender: currentUser.value
    })
    
    // 添加到本地消息列表
    messages.value.push({
      messageId: Date.now().toString(),
      content: newMessage.value,
      sender: currentUser.value,
      type: 'CHAT',
      timestamp: new Date().toISOString()
    })
    
    newMessage.value = ''
  } catch (error) {
    alert('消息发送失败')
  }
}

// 格式化时间
function formatTime(timestamp) {
  if (!timestamp) return ''
  const date = new Date(timestamp)
  return date.toLocaleTimeString()
}
