```javascript
import { createRouter, createWebHistory } from 'vue-router'
import Home from '../views/Home.vue'
import Article from '../views/Article.vue'
import Category from '../views/Category.vue'
import About from '../views/About.vue'

const routes = [
  {
    path: '/',
    name: 'Home',
    component: Home
  },
  {
    path: '/article/:id',
    name: 'Article',
    component: Article
  },
  {
    path: '/category/:name',
    name: 'Category',
    component: Category
  },
  {
    path: '/about',
    name: 'About',
    component: About
  }
]

const router = createRouter({
  history: createWebHistory(),
  routes
})

export default router
```

javascript

### **五、API 请求封装**

#### 1. 创建 API 模块`src/api/article.js`

javascript











```javascript
import axios from 'axios'

const service = axios.create({
  baseURL: 'https://api.example.com',  // 替换为实际API地址
  timeout: 5000
})

// 请求拦截器
service.interceptors.request.use(
  config => {
    // 添加token等操作
    return config
  },
  error => {
    console.error(error)
    return Promise.reject(error)
  }
)

// 响应拦截器
service.interceptors.response.use(
  response => {
    const res = response.data
    if (res.code !== 200) {
      return Promise.reject(new Error(res.message || 'Error'))
    } else {
      return res
    }
  },
  error => {
    console.error('请求错误:', error)
    return Promise.reject(error)
  }
)

// 获取文章列表
export const getArticleList = (params = {}) => {
  return service.get('/articles', { params })
}

// 获取文章详情
export const getArticleDetail = (id) => {
  return service.get(`/articles/${id}`)
}

// 获取分类列表
export const getCategories = () => {
  return service.get('/categories')
}
```

#### 2. 创建 store`src/store/article.js`

javascript











```javascript
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { getArticleList } from '@/api/article'

export const useArticleStore = defineStore('article', () => {
  const articles = ref([])
  const loading = ref(false)
  const error = ref(null)

  // 获取文章列表
  const fetchArticles = async (params) => {
    loading.value = true
    error.value = null
    try {
      const res = await getArticleList(params)
      articles.value = res.data
    } catch (err) {
      error.value = err.message
    } finally {
      loading.value = false
    }
  }

  // 计算属性：获取置顶文章
  const topArticles = computed(() => {
    return articles.value.filter(article => article.isTop)
  })

  return {
    articles,
    loading,
    error,
    fetchArticles,
    topArticles
  }
})
```

#### 2. 创建 Markdown 组件`src/components/MarkdownRenderer.vue`

```vue
<template>
  <div class="markdown-content" v-html="parsedMarkdown"></div>
</template>

<script setup>
import { ref, onMounted, watch } from 'vue'
import marked from 'marked'
import hljs from 'highlight.js'
import 'highlight.js/styles/github.css'

const props = defineProps({
  content: {
    type: String,
    required: true
  }
})

const parsedMarkdown = ref('')

// 配置marked
marked.setOptions({
  highlight: function(code, lang) {
    if (lang && hljs.getLanguage(lang)) {
      return hljs.highlight(code, { language: lang }).value
    }
    return hljs.highlightAuto(code).value
  },
  breaks: true
})

const renderMarkdown = () => {
  parsedMarkdown.value = marked.parse(props.content || '')
}

onMounted(() => {
  renderMarkdown()
})

watch(() => props.content, () => {
  renderMarkdown()
})
</script>
```

### **八、项目构建与部署**

#### 1. 配置生产环境

在`vue.config.js`中配置：

javascript











```javascript
const { defineConfig } = require('@vue/cli-service')
module.exports = defineConfig({
  transpileDependencies: true,
  publicPath: process.env.NODE_ENV === 'production' ? '/blog/' : '/',
  productionSourceMap: false,
  chainWebpack: config => {
    // 压缩图片
    config.module
      .rule('images')
      .use('image-webpack-loader')
      .loader('image-webpack-loader')
      .options({
        mozjpeg: { progressive: true, quality: 65 },
        optipng: { enabled: false },
        pngquant: { quality: [0.65, 0.90], speed: 4 },
        gifsicle: { interlaced: false }
      })
      .end()
  }
})
```

### **九、常见问题解决方案**

#### 1. 路由导航问题

| 问题描述           | 解决方案                            |
| ------------------ | ----------------------------------- |
| 刷新页面 404       | 配置服务器将所有请求指向 index.html |
| 路由跳转不更新组件 | 使用 watch 监听 $route 变化         |

#### 2. 样式问题

| 问题描述                | 解决方案                       |
| ----------------------- | ------------------------------ |
| Element Plus 样式不生效 | 确保在 main.js 中正确引入      |
| 全局样式冲突            | 使用 scoped 属性或 CSS Modules |
