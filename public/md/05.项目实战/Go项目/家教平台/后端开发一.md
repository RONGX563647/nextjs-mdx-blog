#### 2. 登录与注册功能

**AuthService.java**

java











```java
package com.jiajiao.service.impl;

import com.jiajiao.entity.User;
import com.jiajiao.repository.UserRepository;
import com.jiajiao.service.AuthService;
import com.jiajiao.util.JwtUtil;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.security.crypto.password.PasswordEncoder;
import org.springframework.stereotype.Service;

import java.util.Date;
import java.util.HashMap;
import java.util.Map;
import java.util.Optional;

@Service
public class AuthService implements AuthService {

    @Autowired
    private UserRepository userRepository;

    @Autowired
    private PasswordEncoder passwordEncoder;

    @Override
    public Map<String, Object> register(String phone, String password, Integer role) {
        Map<String, Object> result = new HashMap<>();

        // 检查手机号是否已注册
        Optional<User> userOptional = userRepository.findByPhone(phone);
        if (userOptional.isPresent()) {
            result.put("success", false);
            result.put("message", "手机号已注册");
            return result;
        }

        // 创建新用户
        User user = new User();
        user.setPhone(phone);
        user.setPassword(passwordEncoder.encode(password));
        user.setRole(role);
        user.setStatus(1); // 默认可用
        user.setCreateTime(new Date().getTime());

        userRepository.save(user);

        result.put("success", true);
        result.put("message", "注册成功");
        return result;
    }

    @Override
    public Map<String, Object> login(String phone, String password) {
        Map<String, Object> result = new HashMap<>();

        // 检查用户是否存在
        Optional<User> userOptional = userRepository.findByPhone(phone);
        if (!userOptional.isPresent()) {
            result.put("success", false);
            result.put("message", "用户不存在");
            return result;
        }

        User user = userOptional.get();

        // 验证密码
        if (!passwordEncoder.matches(password, user.getPassword())) {
            result.put("success", false);
            result.put("message", "密码错误");
            return result;
        }

        // 生成JWT Token
        String token = JwtUtil.generateToken(user.getId(), user.getPhone(), user.getRole());

        result.put("success", true);
        result.put("message", "登录成功");
        result.put("token", token);
        result.put("user", user);
        return result;
    }
}
```

#### 3. JWT 工具类

**JwtUtil.java**

java











```java
package com.jiajiao.util;

import io.jsonwebtoken.Claims;
import io.jsonwebtoken.Jwts;
import io.jsonwebtoken.SignatureAlgorithm;

import java.util.Date;
import java.util.HashMap;
import java.util.Map;

public class JwtUtil {

    // 密钥
    private static final String SECRET_KEY = "jiajiao_platform_secret_key_2023";

    // 过期时间（24小时）
    private static final long EXPIRATION_TIME = 24 * 60 * 60 * 1000;

    /**
     * 生成JWT Token
     */
    public static String generateToken(Long userId, String phone, Integer role) {
        Map<String, Object> claims = new HashMap<>();
        claims.put("userId", userId);
        claims.put("phone", phone);
        claims.put("role", role);

        return Jwts.builder()
                .setClaims(claims)
                .setSubject(phone)
                .setIssuedAt(new Date(System.currentTimeMillis()))
                .setExpiration(new Date(System.currentTimeMillis() + EXPIRATION_TIME))
                .signWith(SignatureAlgorithm.HS512, SECRET_KEY)
                .compact();
    }

    /**
     * 解析JWT Token
     */
    public static Claims parseToken(String token) {
        return Jwts.parser()
                .setSigningKey(SECRET_KEY)
                .parseClaimsJws(token)
                .getBody();
    }

    /**
     * 验证Token是否过期
     */
    public static boolean isTokenExpired(String token) {
        Date expiration = parseToken(token).getExpiration();
        return expiration.before(new Date());
    }
}
```

```java
package com.jiajiao.service;

import com.jiajiao.entity.TeacherInfo;
import com.jiajiao.repository.TeacherInfoRepository;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.util.Date;
import java.util.List;
import java.util.Optional;

@Service
public class TeacherService {

    @Autowired
    private TeacherInfoRepository teacherInfoRepository;

    /**
     * 获取教师列表
     */
    public List<TeacherInfo> getTeacherList(Integer status, String keyword) {
        if (keyword == null || keyword.isEmpty()) {
            return teacherInfoRepository.findByStatus(status);
        } else {
            return teacherInfoRepository.searchByKeyword(status, keyword);
        }
    }

    /**
     * 根据ID获取教师详情
     */
    public Optional<TeacherInfo> getTeacherById(Long id) {
        return teacherInfoRepository.findById(id);
    }

    /**
     * 添加教师信息
     */
    public TeacherInfo addTeacher(TeacherInfo teacherInfo) {
        teacherInfo.setCreateTime(new Date().getTime());
        teacherInfo.setStatus(0); // 默认待审核
        return teacherInfoRepository.save(teacherInfo);
    }

    /**
     * 更新教师信息
     */
    public TeacherInfo updateTeacher(TeacherInfo teacherInfo) {
        return teacherInfoRepository.save(teacherInfo);
    }

    /**
     * 审核教师信息
     */
    public void approveTeacher(Long id, Integer status) {
        teacherInfoRepository.findById(id).ifPresent(teacher -> {
            teacher.setStatus(status);
            teacherInfoRepository.save(teacher);
        });
    }
}
```

#### 2. Controller 实现

**TeacherController.java**

```java
package com.jiajiao.controller;

import com.jiajiao.common.Result;
import com.jiajiao.entity.TeacherInfo;
import com.jiajiao.service.TeacherService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;

import java.util.List;
import java.util.Optional;

@RestController
@RequestMapping("/api/teacher")
public class TeacherController {

    @Autowired
    private TeacherService teacherService;

    /**
     * 获取教师列表
     */
    @GetMapping("/list")
    public Result<List<TeacherInfo>> getTeacherList(
            @RequestParam(defaultValue = "1") Integer status,
            @RequestParam(required = false) String keyword) {
        List<TeacherInfo> teacherList = teacherService.getTeacherList(status, keyword);
        return Result.success(teacherList);
    }

    /**
     * 获取教师详情
     */
    @GetMapping("/{id}")
    public Result<TeacherInfo> getTeacherById(@PathVariable Long id) {
        Optional<TeacherInfo> teacherOptional = teacherService.getTeacherById(id);
        return teacherOptional.map(Result::success)
                .orElseGet(() -> Result.error("教师不存在"));
    }

    /**
     * 添加教师信息
     */
    @PostMapping("/add")
    public Result<TeacherInfo> addTeacher(@RequestBody TeacherInfo teacherInfo) {
        TeacherInfo savedTeacher = teacherService.addTeacher(teacherInfo);
        return Result.success(savedTeacher);
    }

    /**
     * 更新教师信息
     */
    @PostMapping("/update")
    public Result<TeacherInfo> updateTeacher(@RequestBody TeacherInfo teacherInfo) {
        TeacherInfo updatedTeacher = teacherService.updateTeacher(teacherInfo);
        return Result.success(updatedTeacher);
    }

    /**
     * 审核教师信息
     */
    @PostMapping("/approve/{id}")
    public Result<?> approveTeacher(@PathVariable Long id, @RequestParam Integer status) {
        teacherService.approveTeacher(id, status);
        return Result.success();
    }
}
```

```java
package com.jiajiao.common;

import lombok.Data;

@Data
public class Result<T> {
    private Integer code;
    private String message;
    private T data;

    private Result(Integer code, String message, T data) {
        this.code = code;
        this.message = message;
        this.data = data;
    }

    public static <T> Result<T> success(T data) {
        return new Result<>(200, "操作成功", data);
    }

    public static <T> Result<T> success() {
        return new Result<>(200, "操作成功", null);
    }

    public static <T> Result<T> error(String message) {
        return new Result<>(500, message, null);
    }

    public static <T> Result<T> error(Integer code, String message) {
        return new Result<>(code, message, null);
    }
}
```

### **七、常见问题解决方案**

#### 1. 数据库相关问题

| 问题描述       | 可能原因               | 解决方案                       |
| -------------- | ---------------------- | ------------------------------ |
| 无法连接数据库 | 数据库服务未启动       | 检查 MySQL 服务状态，启动服务  |
| 表不存在       | JPA 自动建表失败       | 检查数据库连接配置，手动创建表 |
| 中文乱码       | 数据库字符集配置不正确 | 修改数据库字符集为 utf8mb4     |

#### 2. 接口开发问题

| 问题描述                  | 可能原因               | 解决方案                   |
| ------------------------- | ---------------------- | -------------------------- |
| 404 Not Found             | 路径配置错误           | 检查 Controller 注解和路径 |
| 500 Internal Server Error | 空指针异常或数据库错误 | 查看日志，调试代码         |
| 跨域问题                  | 未配置 CORS            | 添加 CorsConfig 类         |

#### 3. 安全与权限问题

| 问题描述       | 可能原因             | 解决方案                     |
| -------------- | -------------------- | ---------------------------- |
| Token 验证失败 | Token 过期或签名错误 | 重新登录获取新 Token         |
| 权限不足       | 角色权限配置不正确   | 检查 Security 配置和角色定义 |
