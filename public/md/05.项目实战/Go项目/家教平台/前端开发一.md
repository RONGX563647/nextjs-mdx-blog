### **一、前端环境搭建（超详细步骤）**

#### 1. HBuilderX 安装与配置

**步骤 1：下载与安装**

1. 访问官网：https://www.dcloud.io/hbuilderx.html
2. 点击「App 开发版」下载（Windows 版为.zip 文件）
3. 解压下载的文件到任意目录（如 D:\Program Files\HBuilderX）
4. 双击`HBuilderX.exe`启动程序

**步骤 2：配置开发者工具**

1. 首次启动会提示安装插件，选择「默认安装」
2. 点击「工具」→「插件安装」，确保以下插件已安装：
   - uView UI（UI 组件库）
   - Easy WXSS（样式辅助工具）
   - 代码格式化插件（如 Prettier）

**步骤 3：创建项目**

1. 点击「文件」→「新建」→「项目」
2. 填写信息：
   - 项目名称：jiajiao-front
   - 项目类型：uni-app
   - 模板：默认模板
   - 路径：选择本地目录（如 D:\projects\jiajiao-front）
3. 点击「创建」，等待项目初始化完成

```vue
<template>  
  <view class="container">  
    <!-- 顶部导航 -->  
    <u-navbar title="家教平台" background-color="#367CF0"></u-navbar>  

    <!-- 登录表单 -->  
    <view class="login-form">  
      <!-- Logo区域 -->  
      <view class="logo-area">  
        <u-image src="/static/logo.png" mode="aspectFit" width="120rpx" height="120rpx"></u-image>  
        <text class="logo-text">智慧家教</text>  
      </view>  

      <!-- 输入框组 -->  
      <view class="input-group">  
        <u-input  
          v-model="phone"  
          placeholder="请输入手机号"  
          prefix-icon="phone"  
          clearable  
          @blur="validatePhone"  
        ></u-input>  
        
        <u-input  
          v-model="password"  
          placeholder="请输入密码"  
          prefix-icon="lock"  
          type="password"  
          clearable  
          @blur="validatePassword"  
        ></u-input>  
        
        <!-- 错误提示 -->  
        <view class="error-tip" v-if="phoneError">{{ phoneError }}</view>  
        <view class="error-tip" v-if="passwordError">{{ passwordError }}</view>  
      </view>  

      <!-- 操作按钮 -->  
      <view class="btn-group">  
        <u-button  
          :loading="loading"  
          @click="handleLogin"  
          text="登录"  
          type="primary"  
          shape="circle"  
          disabled="false"  
        ></u-button>  
        
        <view class="link-group">  
          <text @click="toRegister" class="link-text">注册账号</text>  
          <text @click="toForgot" class="link-text">忘记密码?</text>  
        </view>  
      </view>  
    </view>  
  </view>  
</template>  

<script>  
export default {  
  data() {  
    return {  
      phone: '',  
      password: '',  
      phoneError: '',  
      passwordError: '',  
      loading: false  
    }  
  },  
  methods: {  
    // 手机号验证  
    validatePhone() {  
      const phoneReg = /^1[3-9]\d{9}$/;  
      if (!this.phone) {  
        this.phoneError = '请输入手机号';  
      } else if (!phoneReg.test(this.phone)) {  
        this.phoneError = '手机号格式不正确';  
      } else {  
        this.phoneError = '';  
      }  
    },  

    // 密码验证  
    validatePassword() {  
      if (!this.password) {  
        this.passwordError = '请输入密码';  
      } else if (this.password.length < 6) {  
        this.passwordError = '密码长度不能少于6位';  
      } else {  
        this.passwordError = '';  
      }  
    },  

    // 登录处理  
    async handleLogin() {  
      // 先验证表单  
      this.validatePhone();  
      this.validatePassword();  
      
      if (this.phoneError || this.passwordError) {  
        return;  
      }  

      // 显示加载状态  
      this.loading = true;  

      try {  
        // 调用登录接口（后续实现）  
        await new Promise(resolve => setTimeout(resolve, 1000));  
        
        // 模拟登录成功  
        uni.showToast({  
          title: '登录成功',  
          icon: 'success',  
          duration: 2000  
        });  

        // 跳转到首页  
        setTimeout(() => {  
          uni.switchTab({  
            url: '/pages/index/index'  
          });  
        }, 2000);  
      } catch (error) {  
        uni.showToast({  
          title: '登录失败，请重试',  
          icon: 'none',  
          duration: 2000  
        });  
      } finally {  
        // 隐藏加载状态  
        this.loading = false;  
      }  
    },  

    // 跳转到注册页  
    toRegister() {  
      uni.navigateTo({  
        url: '/pages/login/register'  
      });  
    },  

    // 跳转到忘记密码页  
    toForgot() {  
      uni.navigateTo({  
        url: '/pages/login/forgot'  
      });  
    }  
  }  
}  
</script>
```

```vue
<template>

      </view>  
      
      <!-- 密码 -->  
      <u-input  
        v-model="password"  
        placeholder="请设置密码"  
        prefix-icon="lock"  
        type="password"  
        clearable  
        @blur="validatePassword"  
      ></u-input>  
      
      <!-- 确认密码 -->  
      <u-input  
        v-model="confirmPassword"  
        placeholder="请再次输入密码"  
        prefix-icon="lock"  
        type="password"  
        clearable  
        @blur="validateConfirmPassword"  
      ></u-input>

      <view class="error-tip" v-if="passwordError">{{ passwordError }}</view>  
      <view class="error-tip" v-if="confirmPasswordError">{{ confirmPasswordError }}</view>  
      
      <!-- 注册按钮 -->  
      <u-button  
        :loading="loading"  
        @click="handleRegister"  
        text="注册"  
        type="primary"  
        shape="circle"  
        margin="60rpx 0"  
      ></u-button>  
      
      <!-- 已有账号 -->  
      <view class="link-group">  
        <text @click="toLogin" class="link-text">已有账号?立即登录</text>  
      </view>  
    </view>  
  </view>  
</template>  

<script>  
export default {  
  data() {  
    return {  
      phone: '',  
      code: '',  
      password: '',  
      confirmPassword: '',  
      phoneError: '',  
      codeError: '',  
      passwordError: '',  
      confirmPasswordError: '',  
      loading: false,  
      countdown: 0  
    }  
  },  
  methods: {  
    // 发送验证码  
    async sendCode() {  
      // 验证手机号  
      this.validatePhone();  
      if (this.phoneError) {  
        return;  
      }  

      // 发送验证码（模拟）  
      this.countdown = 60;  
      const timer = setInterval(() => {  
        this.countdown--;  
        if (this.countdown <= 0) {  
          clearInterval(timer);  
        }  
      }, 1000);  

      try {  
        // 调用发送验证码接口  
        uni.showToast({  
          title: '验证码已发送',  
          icon: 'none'  
        });  
      } catch (error) {  
        uni.showToast({  
          title: '发送失败，请重试',  
          icon: 'none'  
        });  
        this.countdown = 0;  
      }  
    },  

    // 注册处理  
    async handleRegister() {  
      // 验证表单  
      this.validatePhone();  
      this.validateCode();  
      this.validatePassword();  
      this.validateConfirmPassword();  
      
      if (  
        this.phoneError ||  
        this.codeError ||  
        this.passwordError ||  
        this.confirmPasswordError  
      ) {  
        return;  
      }  

      // 显示加载状态  
      this.loading = true;  

      try {  
        // 调用注册接口（模拟）  
        await new Promise(resolve => setTimeout(resolve, 1000));  
        
        uni.showToast({  
          title: '注册成功',  
          icon: 'success'  
        });  

        // 注册成功后跳转到登录页  
        setTimeout(() => {  
          uni.navigateBack();  
        }, 2000);  
      } catch (error) {  
        uni.showToast({  
          title: '注册失败，请重试',  
          icon: 'none'  
        });  
      } finally {  
        this.loading = false;  
      }  
    },  

    // 跳转到登录页  
    toLogin() {  
      uni.navigateBack();  
    }  
  }  
}  
</script>
```

#### 2. 教师列表页实现

**teacher.vue 完整代码：**

vue











```vue
<template>  
  <view class="container">  
    <u-navbar title="教师列表" background-color="#367CF0"></u-navbar>  

    <!-- 搜索框 -->  
    <view class="search-box">  
      <u-search  
        v-model="keyword"  
        placeholder="搜索教师姓名/科目"  
        @confirm="searchTeacher"  
        @clear="getTeacherList"  
      ></u-search>  
    </view>  

    <!-- 筛选条件 -->  
    <view class="filter-bar">  
      <u-tabs  
        :list="subjectList"  
        v-model="activeSubject"  
        @change="onSubjectChange"  
        line-width="30"  
        height="80rpx"  
        font-size="28rpx"  
      ></u-tabs>  
    </view>  

    <!-- 教师列表 -->  
    <view class="teacher-list">  
      <view class="teacher-item" v-for="(item, index)" :key="index" @click="toTeacherDetail(item.id)">  
        <u-image  
          :src="item.avatar || '/static/default_avatar.png'"  
          mode="aspectFill"  
          width="140rpx"  
          height="140rpx"  
          class="teacher-avatar"  
        ></u-image>  
        
        <view class="teacher-info">  
          <view class="teacher-name">  
            <text>{{ item.realName }}</text>  
            <text class="teacher-title">{{ item.title }}</text>  
          </view>  
          
          <view class="teacher-subject">  
            <text class="tag">{{ item.subject }}</text>  
            <text class="experience">{{ item.teachingYears }}年教龄</text>  
          </view>  
          
          <view class="teacher-intro">  
            {{ item.introduction.substring(0, 30) }}...  
          </view>  
          
          <view class="teacher-footer">  
            <view class="rating">  
              <u-rate v-model="item.rating" size="24rpx" color="#FFD700" readonly></u-rate>  
              <text class="rating-text">{{ item.rating }}</text>  
            </view>  
            <u-button text="预约" type="primary" size="mini" @click.stop="toAppointment(item.id)"></u-button>  
          </view>  
        </view>  
      </view>  

      <!-- 加载提示 -->  
      <view class="loading-tip" v-if="loading">  
        <u-loadmore :status="loadStatus"></u-loadmore>  
      </view>  

      <!-- 空状态 -->  
      <view class="empty-state" v-if="!loading && teacherList.length === 0">  
        <u-empty mode="no-data" description="暂无数据"></u-empty>  
      </view>  
    </view>  
  </view>

<script>  
export default {  
  data() {  
    return {  
      keyword: '',  
      activeSubject: 0,  
      subjectList: [  
        { id: 0, name: '全部' },  
        { id: 1, name: '数学' },  
        { id: 2, name: '语文' },  
        { id: 3, name: '英语' },  
        { id: 4, name: '物理' },  
        { id: 5, name: '化学' }  
      ],  
      teacherList: [],  
      page: 1,  
      pageSize: 10,  
      hasMore: true,  
      loading: false,  
      loadStatus: 'loading'  
    }  
  },  
  onLoad() {  
    this.getTeacherList();  
  },  
  methods: {  
    // 获取教师列表  
    async getTeacherList() {  
      if (this.loading || !this.hasMore) {  
        return;  
      }  

      this.loading = true;  
      this.loadStatus = 'loading';  

      try {  
        // 模拟接口请求  
        await new Promise(resolve => setTimeout(resolve, 800));  

        // 生成模拟数据  
        const mockData = this.generateMockTeachers(this.page);  
        
        if (this.page === 1) {  
          this.teacherList = mockData;  
        } else {  
          this.teacherList = [...this.teacherList, ...mockData];  
        }  

        // 判断是否还有更多数据  
        this.hasMore = mockData.length >= this.pageSize;  
        this.loadStatus = this.hasMore ? 'loading' : 'nomore';  

        this.page++;  
      } catch (error) {  
        this.loadStatus = 'error';  
        uni.showToast({  
          title: '加载失败，请重试',  
          icon: 'none'  
        });  
      } finally {  
        this.loading = false;  
      }  
    },  

    // 生成模拟教师数据  
    generateMockTeachers(page) {  
      const teachers = [];  
      const startIndex = (page - 1) * this.pageSize + 1;  
      const endIndex = Math.min(startIndex + this.pageSize - 1, 25);  

      for (let i = startIndex; i <= endIndex; i++) {  
        const subjects = ['数学', '语文', '英语', '物理', '化学', '生物'];  
        const titles = ['高级教师', '特级教师', '资深教师', '优秀教师'];  
        const subjectIndex = this.activeSubject === 0 ? Math.floor(Math.random() * 6) : this.activeSubject - 1;  

        teachers.push({  
          id: i,  
          realName: `张老师${i}`,  
          avatar: `/static/teacher_avatar/${i % 10}.jpg`,  
          title: titles[Math.floor(Math.random() * titles.length)],  
          subject: subjects[subjectIndex],  
          teachingYears: Math.floor(Math.random() * 15) + 3,  
          introduction: `拥有丰富的教学经验，擅长${subjects[subjectIndex]}教学，教学方法独特，深受学生喜爱。`,  
          rating: (Math.random() * 1.5 + 3.5).toFixed(1)  
        });  
      }  

      return teachers;  
    },  

    // 科目筛选  
    onSubjectChange(index) {  
      this.activeSubject = index;  
      this.page = 1;  
      this.hasMore = true;  
      this.getTeacherList();  
    },  

    // 搜索教师  
    searchTeacher() {  
      this.page = 1;  
      this.hasMore = true;  
      this.getTeacherList();  
    },  

    // 跳转到教师详情页  
    toTeacherDetail(id) {  
      uni.navigateTo({  
        url: `/pages/teacher/detail?id=${id}`  
      });  
    },  

    // 跳转到预约页  
    toAppointment(id) {  
      uni.navigateTo({  
        url: `/pages/appointment/appointment?teacherId=${id}`  
      });  
    },  

    // 下拉刷新  
    onPullDownRefresh() {  
      this.page = 1;  
      this.hasMore = true;  
      this.getTeacherList().then(() => {  
        uni.stopPullDownRefresh();  
      });  
    },  

    // 上拉加载  
    onReachBottom() {  
      this.getTeacherList();  
    }  
  }  
}  
</script>
```

### **四、接口请求与状态管理**

#### 1. 封装 API 请求

在`utils`目录下创建`request.js`：

javascript











```javascript
// 请求基地址（开发环境）  
const BASE_URL = 'http://localhost:8080/api';  

// 封装请求方法  
export function request(url, data = {}, method = 'GET') {  
  return new Promise((resolve, reject) => {  
    // 获取本地存储的token  
    const token = uni.getStorageSync('token');  

    uni.request({  
      url: BASE_URL + url,  
      data,  
      method,  
      header: {  
        'Content-Type': 'application/json',  
        'Authorization': token ? `Bearer ${token}` : ''  
      },  
      success: (res) => {  
        if (res.statusCode === 200) {  
          // 处理业务错误  
          if (res.data.code !== 200) {  
            uni.showToast({  
              title: res.data.message || '请求失败',  
              icon: 'none'  
            });  
            reject(res.data);  
            return;  
          }  
          
          resolve(res.data.data);  
        } else {  
          uni.showToast({  
            title: `请求失败: ${res.errMsg}`,  
            icon: 'none'  
          });  
          reject(res);  
        }  
      },  
      fail: (err) => {  
        uni.showToast({  
          title: `网络错误: ${err.errMsg}`,  
          icon: 'none'  
        });  
        reject(err);  
      }  
    });  
  });  
}  

// 登录接口  
export function login(phone, password) {  
  return request('/auth/login', { phone, password }, 'POST');  
}  

// 注册接口  
export function register(phone, password, role = 1) {  
  return request('/auth/register', { phone, password, role }, 'POST');  
}  

// 获取教师列表接口  
export function getTeacherList(params) {  
  return request('/teacher/list', params, 'GET');  
}  
```

#### 2. 状态管理（全局变量）

在`store`目录下创建`index.js`：

```javascript
import Vue from 'vue';  

const store = new Vue({  
  data() {  
    return {  
      userInfo: uni.getStorageSync('userInfo') || null,  
      token: uni.getStorageSync('token') || ''  
    }  
  },  
  methods: {  
    // 登录成功后保存用户信息  
    saveUserInfo(userInfo, token) {  
      this.userInfo = userInfo;  
      this.token = token;  
      
      // 保存到本地存储  
      uni.setStorageSync('userInfo', userInfo);  
      uni.setStorageSync('token', token);  
    },  

    // 退出登录  
    logout() {  
      this.userInfo = null;  
      this.token = '';  
      
      // 清除本地存储  
      uni.removeStorageSync('userInfo');  
      uni.removeStorageSync('token');  
      
      // 跳转到登录页  
      uni.reLaunch({  
        url: '/pages/login/login'  
      });  
    }  
  }  
});  

export default store;
```

### **五、常见问题解决方案**

#### 1. 样式兼容问题

| 问题描述               | 解决方案                                   |
| ---------------------- | ------------------------------------------ |
| 不同手机字体大小不一致 | 使用 rpx 单位，避免使用固定 px             |

#### 2. 性能优化建议

1. 图片懒加载：使用 u-lazy-load 组件
2. 长列表优化：使用虚拟列表组件（如 uni-app 官方的 virtual-list）
3. 减少全局变量：避免在 App.vue 中定义过多全局数据
4. 组件化开发：将常用 UI 封装成组件复用
