​    // 转发工具：将分片/帧转发到下一跳（路由器或端系统）

​    public static void deliver(Object data, String nextHop) {

​        if (routers.containsKey(nextHop)) {

​            // 下一跳是路由器：转发分片（分组交换）或帧（多协议）

​            Router router = routers.get(nextHop);

​            if (data instanceof Packet.Fragment) {

​                router.receiveFragment((Packet.Fragment) data);

​            } else if (data instanceof Frame) {

​                router.receiveFrame((Frame) data);

​            }

​        } else if (endSystems.containsKey(nextHop)) {

​            // 下一跳是端系统：交付分片或帧

​            EndSystem es = endSystems.get(nextHop);

​            if (data instanceof Packet.Fragment) {

​                es.receiveFragment((Packet.Fragment) data);

​            } else if (data instanceof Frame) {

​                es.receiveFrame((Frame) data);

​            }

​        } else {

​            System.out.printf("下一跳%s不存在，数据丢失%n", nextHop);

​        }

​    }

​    /**

     * 数据包类：支持MTU分片与重组（对应分组交换、MTU知识点）

          * 核心：超过MTU时拆分为分片，接收端重组为完整包

​     */

​    public static class Packet {

​        // 分片内部类：携带重组必需的元数据（分片ID、索引等）

​        public static class Fragment {

​            public final String src;          // 源端系统ID

​            public final String dest;         // 目的端系统ID

​            public final String data;         // 分片数据

​            public final int seq;             // 所属数据包序号

​            public final int fragId;          // 分片唯一标识（同包分片ID相同）

​            public final int totalFrags;      // 总分片数

​            public final int fragIndex;       // 当前分片索引（0开始）

​            public final List<String> route = new ArrayList<>(); // 路由轨迹


​            public Fragment(String src, String dest, String data, int seq,

​                           int fragId, int totalFrags, int fragIndex) {

​                this.src = src;

​                this.dest = dest;

​                this.data = data;

​                this.seq = seq;

​                this.fragId = fragId;

​                this.totalFrags = totalFrags;

​                this.fragIndex = fragIndex;

​            }


​            @Override

​            public String toString() {

​                return String.format("Frag(seq=%d, id=%d, idx=%d/%d, data=%s)",

​                        seq, fragId, fragIndex, totalFrags,

​                        data.length() > 15 ? data.substring(0, 15) + "..." : data);

​            }

​        }


​        public final String src;       // 源端系统

​        public final String dest;      // 目的端系统

​        public final String data;      // 完整数据

​        public final int seq;          // 数据包序号

​        public final List<String> route = new ArrayList<>(); // 路由轨迹


​        public Packet(String src, String dest, String data, int seq) {

​            this.src = src;

​            this.dest = dest;

​            this.data = data;

​            this.seq = seq;

​        }


​        /**

         * 拆分数据包为分片（解决MTU限制）

                  * @param mtu 最大传输单元（字节），头部占20字节，数据区=MTU-20

​         */

​        public List<Fragment> split(int mtu) {

​            List<Fragment> frags = new ArrayList<>();

​            byte[] bytes = data.getBytes();

​            int maxData = mtu - 20; // 数据区最大长度（扣除头部）

​            int totalFrags = (int) Math.ceil((double) bytes.length / maxData);

​            int fragId = new Random().nextInt(10000); // 生成唯一分片ID


​            for (int i = 0; i < totalFrags; i++) {

​                int start = i * maxData;

​                int end = Math.min(start + maxData, bytes.length);

​                String fragData = new String(bytes, start, end - start);

​                frags.add(new Fragment(src, dest, fragData, seq, fragId, totalFrags, i));

​            }

​            System.out.printf("[%s] 数据包(seq=%d)拆分为%d个分片（MTU=%d）%n", src, seq, totalFrags, mtu);

​            return frags;

​        }


​        /**

         * 从分片重组为完整数据包（处理乱序分片）

​         */

​        public static Packet reassemble(List<Fragment> frags) {

​            if (frags.isEmpty()) return null;

​            Fragment first = frags.get(0);

​            int total = first.totalFrags;

​            // 检查分片是否完整

​            if (frags.size() != total) {

​                System.out.printf("[%s] 分片不完整（需%d个，当前%d个）%n", first.dest, total, frags.size());

​                return null;

​            }


​            // 按索引排序分片（解决乱序问题）

​            frags.sort(Comparator.comparingInt(f -> f.fragIndex));


​            // 拼接数据

​            StringBuilder fullData = new StringBuilder();

​            for (Fragment f : frags) fullData.append(f.data);


​            // 合并路由轨迹

​            Packet packet = new Packet(first.src, first.dest, fullData.toString(), first.seq);

​            frags.forEach(f -> packet.route.addAll(f.route));

​            System.out.printf("[%s] 重组数据包(seq=%d)：%s%n", first.dest, first.seq, fullData);

​            return packet;

​        }


​        @Override

​        public String toString() {

​            return String.format("Packet(seq=%d, %s→%s, data=%s, 路由=%s)",

​                    seq, src, dest, data.length() > 20 ? data.substring(0, 20) + "..." : data, route);

​        }

​    }



​    /**

     * 帧结构类：用于UDP/TCP/停止等待协议（对应数据链路层、错误检测知识点）

          * 包含序列号、CRC校验码，支持有效性验证

​     */

​    public static class Frame {

​        public final byte seq;       // 序列号（用于排序和重传）

​        public final long crc;       // CRC校验码（错误检测）

​        public final byte[] data;    // 帧数据

​        public final String src;     // 源端系统

​        public final String dest;    // 目的端系统


​        public Frame(byte seq, byte[] data, String src, String dest) {

​            this.seq = seq;

​            this.data = data;

​            this.src = src;

​            this.dest = dest;

​            this.crc = calcCRC(seq, data); // 计算CRC（含序列号+数据，防止篡改）

​        }


​        // 从字节数组解析帧

​        public static Frame fromBytes(byte[] bytes, String src, String dest) {

​            if (bytes.length < 5) return null; // 1字节seq + 4字节crc + 数据

​            byte seq = bytes[0];

​            long crc = ((bytes[1] & 0xFFL) << 24) | ((bytes[2] & 0xFFL) << 16)

​                    | ((bytes[3] & 0xFFL) << 8) | (bytes[4] & 0xFFL);

​            byte[] data = new byte[bytes.length - 5];

​            System.arraycopy(bytes, 5, data, 0, data.length);

​            return new Frame(seq, data, src, dest, crc);

​        }


​        // 带CRC的构造函数（解析用）

​        private Frame(byte seq, byte[] data, String src, String dest, long crc) {

​            this.seq = seq;

​            this.data = data;

​            this.src = src;

​            this.dest = dest;

​            this.crc = crc;

​        }


​        // 转为字节数组（网络传输用）

​        public byte[] toBytes() {

​            byte[] frame = new byte[5 + data.length]; // 1+4+数据

​            frame[0] = seq;

​            frame[1] = (byte) (crc >> 24);

​            frame[2] = (byte) (crc >> 16);

​            frame[3] = (byte) (crc >> 8);

​            frame[4] = (byte) crc;

​            System.arraycopy(data, 0, frame, 5, data.length);

​            return frame;

​        }


​        // 验证帧有效性（CRC匹配则有效）

​        public boolean isValid() {

​            return calcCRC(seq, data) == crc;

​        }


​        // 计算CRC校验码（基于CRC32，对应错误检测知识点）

​        private long calcCRC(byte seq, byte[] data) {

​            CRC32 crc32 = new CRC32();

​            crc32.update(seq); // 包含序列号，防止序列号被篡改

​            crc32.update(data);

​            return crc32.getValue();

​        }


​        @Override

​        public String toString() {

​            return String.format("Frame(seq=%d, %s→%s, data=%s, 有效=%s)",

​                    seq, src, dest, new String(data).length() > 15 ? new String(data).substring(0, 15) + "..." : new String(data),

​                    isValid() ? "是" : "否");

​        }

​    }



​    /**

     * 路由器类：实现存储-转发、路由表（LPM）、队列管理（对应分组交换、路由知识点）

          * 核心：接收数据后缓存，按路由表转发到下一跳

​     */

​    public static class Router {

​        public final String id; // 路由器ID（如Router1）

​        private final Map<String, String> routeTable; // 路由表：目的端系统→下一跳

​        private final Queue<Object> queue; // 混合队列：存储Fragment或Frame

​        private final int maxQueueSize; // 最大队列容量（防止溢出）

​        private final Lock lock = new ReentrantLock(); // 线程安全锁


​        public Router(String id, Map<String, String> routeTable, int maxQueueSize) {

​            this.id = id;

​            this.routeTable = routeTable;

​            this.queue = new ConcurrentLinkedQueue<>();

​            this.maxQueueSize = maxQueueSize;

​            startForwardThread(); // 启动转发线程（存储-转发的转发环节）

​        }


​        // 接收分片（分组交换）

​        public void receiveFragment(Packet.Fragment frag) {

​            lock.lock();

​            try {

​                if (queue.size() >= maxQueueSize) {

​                    System.out.printf("[%s] 队列满，丢弃分片：%s%n", id, frag);

​                    return;

​                }

​                frag.route.add(id); // 记录路由轨迹

​                queue.add(frag);

​                System.out.printf("[%s] 接收分片，队列大小=%d：%s%n", id, queue.size(), frag);

​            } finally {

​                lock.unlock();

​            }

​        }


​        // 接收帧（多协议传输）

​        public void receiveFrame(Frame frame) {

​            lock.lock();

​            try {

​                if (queue.size() >= maxQueueSize) {

​                    System.out.printf("[%s] 队列满，丢弃帧：%s%n", id, frame);

​                    return;

​                }

​                queue.add(frame);

​                System.out.printf("[%s] 接收帧，队列大小=%d：%s%n", id, queue.size(), frame);

​            } finally {

​                lock.unlock();

​            }

​        }


​        // 转发线程：按顺序处理队列中的数据（存储-转发的核心）

​        private void startForwardThread() {

​            new Thread(() -> {

​                while (!Thread.interrupted()) {

​                    try {

​                        Thread.sleep(100); // 模拟路由器处理延迟

​                        Object data = queue.poll();

​                        if (data != null) {

​                            forward(data);

​                        }

​                    } catch (InterruptedException e) {

​                        Thread.currentThread().interrupt();

​                    }

​                }

​            }, "Forward-" + id).start();

​        }


​        // 转发数据到下一跳（基于路由表LPM匹配）

​        private void forward(Object data) {

​            String dest;

​            if (data instanceof Packet.Fragment) {

​                dest = ((Packet.Fragment) data).dest;

​            } else if (data instanceof Frame) {

​                dest = ((Frame) data).dest;

​            } else {

​                System.out.printf("[%s] 未知数据类型，丢弃%n", id);

​                return;

​            }


​            // 最长前缀匹配（LPM）获取下一跳

​            String nextHop = routeTable.get(dest);

​            if (nextHop == null) {

​                System.out.printf("[%s] 无路由到%s，丢弃数据%n", id, dest);

​                return;

​            }


​            System.out.printf("[%s] 转发到%s：%s%n", id, nextHop, data);

​            NetworkProtocolSimulator.deliver(data, nextHop);

​        }


​        // 路由命令处理（add/match/list）

​        public void handleRouteCommand(String[] args) {

​            if (args.length == 0) {

​                System.out.println("路由子命令：add <cidr> <nexthop> | match <ip> | list");

​                return;

​            }

​            switch (args[0]) {

​                case "add": // 添加路由（CIDR格式）

​                    if (args.length != 3) {

​                        System.out.println("用法：route add <routerId> <cidr> <nexthop>");

​                        return;

​                    }

​                    routeTable.put(args[1], args[2]); // 简化实现（实际需解析CIDR前缀）

​                    System.out.printf("[%s] 添加路由：%s→%s%n", id, args[1], args[2]);

​                    break;

​                case "match": // 匹配路由（最长前缀）

​                    if (args.length != 2) {

​                        System.out.println("用法：route match <routerId> <ip>");

​                        return;

​                    }

​                    String nextHop = routeTable.get(args[1]); // 简化LPM（实际需按前缀长度匹配）

​                    System.out.printf("[%s] %s的匹配路由下一跳：%s%n", id, args[1], nextHop == null ? "无" : nextHop);

​                    break;

​                case "list": // 列出路由表

​                    System.out.printf("[%s] 路由表：%n", id);

​                    routeTable.forEach((dest, hop) -> System.out.printf("  %s → %s%n", dest, hop));

​                    break;

​                default:

​                    System.out.println("未知路由子命令");

​            }

​        }

​    }



​    /**

     * 端系统类：模拟主机，负责生成数据、协议传输、ARP缓存、分片重组（对应端系统知识点）

​     */

​    public static class EndSystem {

​        public final String id; // 端系统ID（如HostA）

​        private final String gateway; // 网关路由器ID

​        private final int mtu; // 最大传输单元

​        private int nextSeq = 0; // 数据包/帧序列号计数器


​        // ARP缓存：IP→[MAC, 过期时间]（地址解析协议）

​        private final Map<String, Object[]> arpCache = new HashMap<>();

​        private static final long ARP_EXPIRE = 300 * 1000; // ARP过期时间300秒

​        private final Lock arpLock = new ReentrantLock(); // ARP操作锁


​        // 分片缓存：fragId→分片列表（用于重组）

​        private final Map<Integer, List<Packet.Fragment>> fragCache = new HashMap<>();

​        // 帧缓存：用于停止等待协议的ACK确认

​        private final Map<Byte, Frame> frameCache = new HashMap<>();


​        public EndSystem(String id, String gateway, int mtu) {

​            this.id = id;

​            this.gateway = gateway;

​            this.mtu = mtu;

​        }


​        /**

         * 发送数据：支持三种协议

                  * @param dest 目的端系统ID

                  * @param data 发送的数据

                           * @param protocol 协议：tcp/udp/stopwait

​         */

​        public void sendData(String dest, String data, String protocol) {

​            switch (protocol.toLowerCase()) {

​                case "tcp":

​                    sendTCP(dest, data);

​                    break;

​                case "udp":

​                    sendUDP(dest, data);

​                    break;

​                case "stopwait":

​                    sendStopWait(dest, data);

​                    break;

​                default:

​                    System.out.println("未知协议，支持tcp/udp/stopwait");

​            }

​        }


​        // TCP发送（可靠传输，基于Java Socket，底层实现三次握手/重传）

​        private void sendTCP(String dest, String data) {

​            try {

​                // 实际中需通过ARP获取IP，此处简化为直接连接

​                Router gw = NetworkProtocolSimulator.getRouter(gateway);

​                if (gw == null) {

​                    System.out.println("网关不存在，发送失败");

​                    return;

​                }

# 网络协议综合模拟器核心知识点理论详解

该项目涵盖了计算机网络从物理层到应用层的核心理论，尤其聚焦于数据传输、路由选择和协议交互的底层逻辑。以下从理论角度全面解析项目涉及的核心知识点：

## 一、分组交换（Packet Switching）

### 1. 基本概念

分组交换是现代计算机网络的核心传输方式，将完整数据分割为**分组（Packet）** 独立传输，每个分组包含数据和控制信息（源 / 目的地址、序号等）。与传统电路交换（如电话网络）相比，分组交换无需预先建立专用链路，可动态共享网络资源。

### 2. 核心优势

- **资源利用率高**：多个通信可共享链路带宽，避免电路交换的资源浪费
- **灵活性强**：适应突发流量，单条链路故障不影响整体通信（可通过其他路径转发）
- **经济性好**：无需为每个连接预留资源，降低网络建设成本

### 3. 项目中的实现

- **Packet 类**：抽象数据包结构，包含源 / 目的地址、完整数据和序列号
- **分片机制**：通过`split()`方法将大数据包拆分为小分片（应对 MTU 限制）
- **重组机制**：通过`reassemble()`方法将分片还原为完整数据包

## 二、MTU 与分片重组（MTU and Fragmentation）

### 1. MTU 的定义

MTU（Maximum Transmission Unit，最大传输单元）指数据链路层帧所能承载的最大数据量（单位：字节），由物理链路特性决定（如以太网 MTU 通常为 1500 字节）。MTU 限制包括帧头（如以太网帧头 28 字节），因此实际数据载荷能力为`MTU - 帧头长度`。

### 2. 为什么需要分片？

当数据包大小超过链路 MTU 时，必须拆分为**分片（Fragment）** 传输：

- 避免帧过大导致物理层传输效率下降
- 减少单帧传输失败的重传成本（小分片重传代价更低）
- 适配不同链路的 MTU 差异（网络中不同链路可能有不同 MTU）

### 3. 分片重组的关键机制

- 分片元数据

  ：每个分片需携带：

  - 所属数据包标识（`fragId`）
  - 总分片数（`totalFrags`）
  - 分片索引（`fragIndex`）：用于重组时排序

- **重组条件**：接收端需收集所有分片（数量等于`totalFrags`）才能重组

- **超时丢弃**：若分片丢失，超过一定时间未集齐则丢弃已接收分片

### 4. 项目中的实现

- `Packet.split(int mtu)`：计算分片数量，拆分数据并生成分片元数据
- `Packet.reassemble(List<Fragment>)`：校验分片完整性，按索引排序后拼接数据

java

```java
// 分片逻辑核心：根据MTU计算最大数据载荷，拆分数据
int maxData = mtu - 20; // 扣除20字节头部
int totalFrags = (int) Math.ceil((double) bytes.length / maxData);
```

## 三、存储 - 转发机制（Store-and-Forward）

### 1. 基本概念

存储 - 转发是路由器 / 交换机的核心工作模式：设备先完整接收分组（存储），验证无误后再转发到下一跳（转发）。与之相对的是 “直通转发（Cut-through）”，后者接收部分数据就开始转发，虽延迟低但无法检测错误。

### 2. 核心作用

- **错误检测**：完整接收后可校验数据完整性（如 CRC 验证），避免错误数据继续传输
- **流量控制**：通过队列缓存调节数据转发速率，避免下游设备过载
- **路由选择**：有充足时间基于最新路由表选择最佳路径

### 3. 项目中的实现

- **Router 类**：通过`ConcurrentLinkedQueue`缓存接收的分片 / 帧（存储）
- **转发线程**：`startForwardThread()`启动独立线程，按顺序处理队列数据（转发）
- **队列管理**：限制最大队列大小，满时丢弃数据（模拟拥塞控制）

java

```java
// 存储-转发核心：先入队（存储），再由转发线程处理（转发）
queue.add(frag); // 存储
// 转发线程中：

if (data != null) forward(data); // 转发
```

## 四、ARP 协议（Address Resolution Protocol）

### 1. 基本作用

ARP 解决**IP 地址到 MAC 地址的映射问题**：网络层使用 IP 地址寻址，而数据链路层需要 MAC 地址进行帧传输，ARP 在两者之间建立转换关系。

### 2. 工作流程

1. **查询**：主机需要发送数据给某 IP 时，先查本地 ARP 缓存。若未找到，发送**ARP 请求广播**（询问 “谁有 IP X？请回复 MAC Y”）
2. **响应**：目标 IP 对应的主机收到广播后，发送**ARP 应答单播**（告知 “我有 IP X，我的 MAC 是 Y”）
3. **缓存**：请求方收到应答后，将 IP-MAC 映射存入 ARP 缓存（设置过期时间，通常 300 秒）

### 3. 项目中的实现

- **ARP 缓存**：`EndSystem`中用`Map<String, Object[]>`存储 IP→[MAC, 过期时间]
- **缓存管理**：通过`arpLock`保证多线程安全，查询时检查过期时间
- **命令支持**：`handleARPCommand()`实现 ARP 条目添加 / 查询 / 清空

java

```java
// ARP缓存核心：存储IP-MAC映射并处理过期

    // 未找到或已过期
}
```

## 五、路由选择与最长前缀匹配（LPM）

### 1. 路由基本概念

路由是指数据从源到目的的路径选择过程，由路由器通过**路由表**实现。路由表条目通常包含：

- 目的网络前缀（如 192.168.1.0/24）
- 下一跳地址（数据应转发到的设备）
- 出接口（转发数据的物理接口）

### 2. 最长前缀匹配（LPM）

LPM 是路由器选择最佳路径的核心算法：

- 每个 IP 地址可视为 32 位二进制数，前缀长度表示网络位位数（如 / 24 表示前 24 位为网络位）
- 匹配时，选择与目的 IP 前缀重合最长的路由条目
- 例：目的 IP 为 192.168.1.5 时，192.168.1.0/24 比 192.168.0.0/16 匹配更长，被优先选择

### 3. 项目中的实现

- **路由表存储**：`Router`中用`Map<String, String>`存储目的→下一跳
- **LPM 匹配**：`handleRouteCommand()`中的`match`子命令实现前缀匹配逻辑
- **动态路由**：支持通过命令添加路由条目，模拟路由表更新

## 六、传输层协议：TCP、UDP 与停止等待

### 1. UDP（User Datagram Protocol）

- 核心特点

  ：无连接、不可靠、无拥塞控制的简单协议

  - 无连接：发送前无需建立连接，直接发送数据报
  - 不可靠：不保证数据到达、不保证顺序、不重传丢失数据
  - 高效：头部仅 8 字节，适合实时性要求高的场景（如视频通话、DNS）

- **项目实现**：`sendUDP()`直接发送帧，不等待确认，可能丢包

```java

}
```

### 2. TCP（Transmission Control Protocol）

- 核心特点

  ：面向连接、可靠传输、拥塞控制的复杂协议

  - 面向连接：通过 “三次握手” 建立连接，“四次挥手” 关闭连接
  - 可靠传输：通过序列号、确认 (ACK)、超时重传保证数据完整有序
  - 拥塞控制：通过慢启动、拥塞避免等算法适应网络负载

- **项目实现**：`sendTCP()`模拟三次握手和可靠传输，依赖底层确认机制

```java

// 发送数据（假设底层处理重传和确认）
```

### 3. 停止等待协议（Stop-and-Wait）

- 核心思想

  ：一种简单的可靠传输协议，发送方每发一帧就停止等待，直到收到确认 (ACK) 才发下一帧

  - 解决丢包：超时未收到 ACK 则重传当前帧
  - 解决乱序：通过序列号确保接收顺序
  - 缺点：效率低（信道利用率低），适合低带宽场景

- **项目实现**：`sendStopWait()`和`sendAndWaitAck()`实现帧发送→等待 ACK→超时重传逻辑

java

```java
// 停止等待核心：每帧发送后等待ACK，超时重传
while (true) {
    socket.send(frame); // 发送帧
    try {
        socket.receive(ack); // 等待ACK
        if (ack有效) break;
    } catch (TimeoutException) {
        continue; // 超时重传
    }
}
```

## 七、错误检测：CRC 校验（Cyclic Redundancy Check）

### 1. 基本原理

CRC 是一种高效的错误检测算法，通过多项式除法计算数据的校验和（CRC 码），接收方重新计算并比对，判断数据是否被篡改或传输错误。

- **计算过程**：
  1. 发送方：将数据视为二进制数，除以约定多项式（如 CRC32 使用 0x04C11DB7），余数即为 CRC 码
  2. 接收方：对收到的数据（含 CRC 码）重新计算，若余数为 0 则数据有效
- **优势**：
  - 检测能力强：可检测所有单比特错误、双比特错误、奇数个错误和大部分突发错误
  - 计算高效：硬件和软件均可快速实现

### 2. 项目中的实现

- **Frame 类**：`calcCRC()`方法基于`CRC32`计算校验码
- **有效性验证**：`isValid()`方法比对计算的 CRC 与帧中存储的 CRC

java

```java
// CRC核心：发送方计算，接收方验证
private long calcCRC(byte seq, byte[] data) {
    CRC32 crc32 = new CRC32();
    crc32.update(seq); // 包含序列号，防止序号篡改
    crc32.update(data);
    return crc32.getValue();
}

public boolean isValid() {
    return calcCRC(seq, data) == crc; // 验证一致性
}
```
