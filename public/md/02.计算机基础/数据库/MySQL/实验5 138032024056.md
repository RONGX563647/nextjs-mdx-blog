## 四、数据结构和算法设计

### （1）存储结构

采用**二叉链表**存储二叉树，节点结构定义：

c

```c
typedef struct BiTNode {
    // 节点数据，假设为字符型，可按需修改
    char data; 
    // 左子树指针
    struct BiTNode *lchild; 
    // 右子树指针
    struct BiTNode *rchild; 
} BiTNode, *BiTree; 
```



辅助结构：



- **队列**（用于层次遍历、完全二叉树判定 ）：可基于顺序表或链表实现，此处用链表队列，节点存储二叉树节点指针。
- **栈**（用于中序非递归遍历 ）：基于顺序栈或链栈，存储二叉树节点指针，辅助模拟递归过程。

### （2）算法整体思路

1. **二叉树创建（先序）**： 递归读取先序序列字符，若为 `#` 则创建空树（返回 `NULL` ）；否则创建新节点，递归构建左子树，再递归构建右子树，返回当前节点作为根。

2. 递归遍历

   ：

   - 先序：`根 -> 左 -> 右`，递归访问根节点后，依次递归左、右子树。
   - 中序：`左 -> 根 -> 右`，先递归左子树，访问根节点，再递归右子树。
   - 后序：`左 -> 右 -> 根`，先递归左、右子树，最后访问根节点。

3. **层次遍历**： 初始化队列，根节点入队；循环取队首节点，访问该节点，若有左 / 右子节点则入队；直到队空，实现按层访问。

4. **二叉树相似判别**： 递归比较两棵树：若都为空则相似；若一为空一非空则不相似；否则递归比较左子树和右子树是否相似。

5. **完全二叉树判定**： 层次遍历二叉树，维护标记 `isLeafStage`（是否进入 “叶子层” ）。遇到空节点后，若后续还有非空节点则不是完全二叉树；或根据节点编号规则（完全二叉树按层编号，若实际编号与理论编号不符则不满足 ）判断。

6. **中序非递归遍历**： 初始化栈，当前节点指向根；循环：当前节点非空则入栈并遍历左子树；当前节点为空则出栈访问，然后遍历右子树；直到栈空且当前节点为空。

## 五、程序代码和运行实例

### （1）完整程序代码

c



运行



```c
#include <stdio.h>
#include <stdlib.h>

// 二叉树节点定义
typedef struct BiTNode {
    char data;
    struct BiTNode *lchild;
    struct BiTNode *rchild;
} BiTNode, *BiTree;

// 队列节点定义（用于层次遍历、完全二叉树判定）
typedef struct QueueNode {
    BiTree data;
    struct QueueNode *next;
} QueueNode;
typedef struct {
    QueueNode *front;
    QueueNode *rear;
} LinkQueue;

// 栈节点定义（用于中序非递归遍历）
typedef struct StackNode {
    BiTree data;
    struct StackNode *next;
} StackNode, *LinkStack;

// 队列初始化
void initQueue(LinkQueue *q) {
    q->front = q->rear = (QueueNode *)malloc(sizeof(QueueNode));
    q->front->next = NULL;
}

// 队列入队
void enQueue(LinkQueue *q, BiTree data) {
    QueueNode *newNode = (QueueNode *)malloc(sizeof(QueueNode));
    newNode->data = data;
    newNode->next = NULL;
    q->rear->next = newNode;
    q->rear = newNode;
}

// 队列出队
int deQueue(LinkQueue *q, BiTree *data) {
    if (q->front == q->rear) return 0; // 队空
    QueueNode *p = q->front->next;
    *data = p->data;
    q->front->next = p->next;
    if (q->rear == p) q->rear = q->front;
    free(p);
    return 1;
}

// 栈初始化
void initStack(LinkStack *s) {
    *s = NULL;
}

// 栈入栈
int pushStack(LinkStack *s, BiTree data) {
    StackNode *newNode = (StackNode *)malloc(sizeof(StackNode));
    if (!newNode) return 0;
    newNode->data = data;
    newNode->next = *s;
    *s = newNode;
    return 1;
}

// 栈出栈
int popStack(LinkStack *s, BiTree *data) {
    if (!*s) return 0;
    StackNode *p = *s;
    *data = p->data;
    *s = p->next;
    free(p);
    return 1;
}

// 1. 创建二叉链表（先序序列，#表示空节点）
void createBiTree(BiTree *T) {
    char ch;
    // 注意：需保证输入流正确，建议从控制台按先序序列输入（如 "AB#D##C##" ）
    scanf(" %c", &ch); 
    if (ch == '#') {
        *T = NULL;
    } else {
        *T = (BiTree)malloc(sizeof(BiTNode));
        (*T)->data = ch;
        createBiTree(&(*T)->lchild);
        createBiTree(&(*T)->rchild);
    }
}

// 2. 先序遍历（递归）
void preOrderTraverse(BiTree T) {
    if (T) {
        printf("%c ", T->data);
        preOrderTraverse(T->lchild);
        preOrderTraverse(T->rchild);
    }
}

// 3. 中序遍历（递归）
void inOrderTraverse(BiTree T) {
    if (T) {
        inOrderTraverse(T->lchild);
        printf("%c ", T->data);
        inOrderTraverse(T->rchild);
    }
}

// 4. 后序遍历（递归）
void postOrderTraverse(BiTree T) {
    if (T) {
        postOrderTraverse(T->lchild);
        postOrderTraverse(T->rchild);
        printf("%c ", T->data);
    }
}

// 5. 层次遍历
void levelOrderTraverse(BiTree T) {
    LinkQueue q;
    initQueue(&q);
    if (T) enQueue(&q, T);
    BiTree p;
    while (deQueue(&q, &p)) {
        printf("%c ", p->data);
        if (p->lchild) enQueue(&q, p->lchild);
        if (p->rchild) enQueue(&q, p->rchild);
    }
    printf("\n");
}

// 选做1：判别两棵二叉树是否相似
int isSimilar(BiTree T1, BiTree T2) {
    if (!T1 && !T2) return 1; // 都空，相似
    if ((!T1 && T2) || (T1 && !T2)) return 0; // 一为空一非空，不相似
    // 递归比较左右子树
    return isSimilar(T1->lchild, T2->lchild) && isSimilar(T1->rchild, T2->rchild); 
}

// 选做2：判别是否为完全二叉树（基于层次遍历+编号规则）
int isCompleteBiTree(BiTree T) {
    if (!T) return 1; // 空树视为完全二叉树
    LinkQueue q;
    initQueue(&q);
    enQueue(&q, T);
    int isLeafStage = 0; // 是否进入“叶子层”标记
    BiTree p;
    while (deQueue(&q, &p)) {
        if (isLeafStage) {
            // 进入叶子层后，节点不能有子节点
            if (p->lchild || p->rchild) return 0; 
        } else {
            if (p->lchild && p->rchild) { 
                enQueue(&q, p->lchild);
                enQueue(&q, p->rchild);
            } else if (!p->lchild && p->rchild) { 
                return 0;
            } else { 
                isLeafStage = 1;
                if (p->lchild) enQueue(&q, p->lchild); 
            }
        }
    }
    return 1;
}

// 选做3：中序遍历的非递归算法
void inOrderTraverseNonRecursive(BiTree T) {
    LinkStack s;
    initStack(&s);
    BiTree p = T;
    while (p || !isEmptyStack(s)) {
        if (p) { 
            pushStack(&s, p);
            p = p->lchild;
        } else { 
            popStack(&s, &p);
            printf("%c ", p->data);
            p = p->rchild;
        }
    }
    printf("\n");
}

// 辅助函数：判断栈是否为空
int isEmptyStack(LinkStack s) {
    return s == NULL;
}

// 测试函数
int main() {
    BiTree T, T2;
    printf("请输入二叉树 T 的先序序列（#表示空节点，如 AB#D##C## ）：\n");
    createBiTree(&T);
    
    printf("\n先序遍历（递归）结果：");
    preOrderTraverse(T);
    printf("\n");
    
    printf("中序遍历（递归）结果：");
    inOrderTraverse(T);
    printf("\n");
    
    printf("后序遍历（递归）结果：");
    postOrderTraverse(T);
    printf("\n");
    
    printf("层次遍历结果：");
    levelOrderTraverse(T);
    
    // 测试相似性（需另外创建 T2 ，这里简单示例，可手动输入 T2 序列对比）
    printf("\n请输入二叉树 T2 的先序序列（用于相似性测试）：\n");
    createBiTree(&T2);
    if (isSimilar(T, T2)) {
        printf("二叉树 T 和 T2 相似！\n");
    } else {
        printf("二叉树 T 和 T2 不相似！\n");
    }
    
    // 测试完全二叉树
    if (isCompleteBiTree(T)) {
        printf("二叉树 T 是完全二叉树！\n");
    } else {
        printf("二叉树 T 不是完全二叉树！\n");
    }
    
    // 测试中序非递归遍历
    printf("中序遍历（非递归）结果：");
    inOrderTraverseNonRecursive(T);
    
    // 注意：实际使用需手动管理内存释放，此处简化省略
    return 0;
}
```

## 六、程序的调试与分析

1. **调试要点**：
   - 创建二叉树时，需严格按先序序列输入（含 `#` 标记空节点 ），否则递归逻辑会出错。可通过打印中间节点验证构建是否正确。
   - 递归遍历易因树结构过深导致栈溢出（小数据量测试无影响 ），实际复杂场景需关注。
   - 层次遍历、非递归中序遍历依赖队列 / 栈的正确操作，需调试队列 / 栈的入队、出队、入栈、出栈逻辑，避免内存泄漏或访问空指针。
   - 完全二叉树判定中，“叶子层” 标记逻辑易出错，需反复测试边界情况（如单节点树、满二叉树、缺右侧节点的树 ）。
2. **算法复杂度分析**：
   - 二叉树创建、递归遍历：时间复杂度均为 \(O(n)\)（n 为节点数，每个节点访问一次 ）；空间复杂度 \(O(h)\)（h 为树高，递归栈 / 辅助栈深度 ）。
   - 层次遍历：时间复杂度 \(O(n)\)（每个节点入队、出队一次 ）；空间复杂度 \(O(n)\)（队列最多存储一层节点，最坏为 \(n/2\) ，记为 \(O(n)\) ）。
   - 相似判别、完全二叉树判定：时间复杂度 \(O(n)\)（需遍历两棵树 / 一棵树所有节点 ）；空间复杂度 \(O(h)\)（递归栈或队列深度 ）。
