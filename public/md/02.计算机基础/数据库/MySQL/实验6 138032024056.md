### （2）算法思路

1. **哈夫曼树构建**：
   - 统计报文中各字符频率，创建对应叶子节点
   - 使用优先队列（最小堆）维护节点，每次取出权值最小的两个节点合并
   - 合并后的新节点权值为两节点之和，插入队列，重复直至队列只剩一个节点
2. **哈夫曼编码生成**：
   - 从根节点开始递归遍历，左路径标记为 0，右路径标记为 1
   - 到达叶子节点时记录完整编码路径，存入编码表
3. **报文编码**：
   - 遍历原始报文，查表替换每个字符为对应哈夫曼编码
4. **报文解码**：
   - 从哈夫曼树根节点开始，按二进制串 0/1 选择左右子树
   - 到达叶子节点时输出对应字符，重置到根节点继续解码
5. **孩子 - 兄弟链表构建**：
   - 使用哈希表记录节点指针，按层次序处理二元组
   - 若父节点不存在则创建，添加子节点为其首个孩子或兄弟
6. **树的度计算**：
   - 遍历所有节点，统计每个节点的孩子数
   - 维护全局最大值作为树的度

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 哈夫曼树节点定义
typedef struct HuffmanNode {
    char data;
    int weight;
    struct HuffmanNode *left, *right;
} HuffmanNode;

// 优先队列节点定义
typedef struct PQNode {
    HuffmanNode *node;
    struct PQNode *next;
} PQNode;

// 哈夫曼编码表项定义
typedef struct {
    char data;
    char code[100];
} HuffmanCode;

// 优先队列操作
PQNode* createPQNode(HuffmanNode *hNode) {
    PQNode *pqNode = (PQNode*)malloc(sizeof(PQNode));
    pqNode->node = hNode;
    pqNode->next = NULL;
    return pqNode;
}

// 插入节点（按权值升序）
void insert(PQNode **head, HuffmanNode *hNode) {
    PQNode *pqNode = createPQNode(hNode);
    if (!(*head) || hNode->weight < (*head)->node->weight) {
        pqNode->next = *head;
        *head = pqNode;
        return;
    }
    PQNode *current = *head;
    while (current->next && current->next->node->weight <= hNode->weight) {
        current = current->next;
    }
    pqNode->next = current->next;
    current->next = pqNode;
}

// 取出最小权值节点
HuffmanNode* extractMin(PQNode **head) {
    if (!*head) return NULL;
    PQNode *temp = *head;
    HuffmanNode *minNode = temp->node;
    *head = (*head)->next;
    free(temp);
    return minNode;
}

// 创建哈夫曼节点
HuffmanNode* createHuffmanNode(char data, int weight) {
    HuffmanNode *node = (HuffmanNode*)malloc(sizeof(HuffmanNode));
    node->data = data;
    node->weight = weight;
    node->left = node->right = NULL;
    return node;
}

// 构建哈夫曼树
HuffmanNode* buildHuffmanTree(char data[], int weights[], int size) {
    PQNode *head = NULL;
    for (int i = 0; i < size; i++) {
        insert(&head, createHuffmanNode(data[i], weights[i]));
    }
    while (head->next) {
        HuffmanNode *left = extractMin(&head);
        HuffmanNode *right = extractMin(&head);
        HuffmanNode *parent = createHuffmanNode('$', left->weight + right->weight);
        parent->left = left;
        parent->right = right;
        insert(&head, parent);
    }
    return extractMin(&head);
}

// 生成哈夫曼编码表
void generateCodes(HuffmanNode *root, int arr[], int top, HuffmanCode codes[], int *codeCount) {
    if (root->left) {
        arr[top] = 0;
        generateCodes(root->left, arr, top + 1, codes, codeCount);
    }
    if (root->right) {
        arr[top] = 1;
        generateCodes(root->right, arr, top + 1, codes, codeCount);
    }
    if (!root->left && !root->right) {
        codes[*codeCount].data = root->data;
        codes[*codeCount].code[0] = '\0';
        for (int i = 0; i < top; i++) {
            char temp[2] = {arr[i] + '0', '\0'};
            strcat(codes[*codeCount].code, temp);
        }
        (*codeCount)++;
    }
}

// 编码报文
void encode(char *message, HuffmanCode codes[], int codeCount) {
    printf("编码结果: ");
    for (int i = 0; message[i] != '\0'; i++) {
        for (int j = 0; j < codeCount; j++) {
            if (codes[j].data == message[i]) {
                printf("%s", codes[j].code);
                break;
            }
        }
    }
    printf("\n");
}

// 解码报文
void decode(HuffmanNode *root, char *encoded) {
    printf("解码结果: ");
    HuffmanNode *current = root;
    for (int i = 0; encoded[i] != '\0'; i++) {
        if (encoded[i] == '0') current = current->left;
        else current = current->right;
        
        if (!current->left && !current->right) {
            printf("%c", current->data);
            current = root;
        }
    }
    printf("\n");
}

// 释放哈夫曼树内存
void freeHuffmanTree(HuffmanNode *root) {
    if (!root) return;
    freeHuffmanTree(root->left);
    freeHuffmanTree(root->right);
    free(root);
}

// 选做部分：孩子-兄弟链表结构
typedef struct TreeNode {
    char data;
    struct TreeNode *firstChild;
    struct TreeNode *nextSibling;
} TreeNode;

// 创建树节点
TreeNode* createTreeNode(char data) {
    TreeNode *node = (TreeNode*)malloc(sizeof(TreeNode));
    node->data = data;
    node->firstChild = node->nextSibling = NULL;
    return node;
}

// 构建孩子-兄弟链表
TreeNode* buildTree() {
    TreeNode *root = NULL;
    TreeNode *nodes[256] = {NULL}; // 简单哈希表存储节点
    char parent, child;
    
    while (1) {
        scanf(" %c%c", &parent, &child);
        if (parent == '^' && child == '^') break;
        
        if (parent == '^') { // 根节点
            root = createTreeNode(child);
            nodes[child] = root;
            continue;
        }
        
        if (!nodes[parent]) {
            nodes[parent] = createTreeNode(parent);
        }
        
        TreeNode *newChild = createTreeNode(child);
        nodes[child] = newChild;
        
        // 添加为第一个孩子或兄弟
        if (!nodes[parent]->firstChild) {
            nodes[parent]->firstChild = newChild;
        } else {
            TreeNode *sibling = nodes[parent]->firstChild;
            while (sibling->nextSibling) sibling = sibling->nextSibling;
            sibling->nextSibling = newChild;
        }
    }
    return root;
}

// 计算树的度
int treeDegree(TreeNode *root) {
    if (!root) return 0;
    
    int maxDegree = 0;
    int currentDegree = 0;
    
    TreeNode *child = root->firstChild;
    while (child) {
        currentDegree++;
        int childDegree = treeDegree(child);
        if (childDegree > maxDegree) maxDegree = childDegree;
        child = child->nextSibling;
    }
    
    if (currentDegree > maxDegree) maxDegree = currentDegree;
    return maxDegree;
}

// 释放树内存
void freeTree(TreeNode *root) {
    if (!root) return;
    freeTree(root->firstChild);
    freeTree(root->nextSibling);
    free(root);
}

int main() {
    // 哈夫曼编码部分
    printf("===== 哈夫曼编码系统 =====\n");
    char data[] = {'A', 'B', 'C', 'D', 'E'};
    int weights[] = {5, 9, 12, 13, 16};
    int size = sizeof(data) / sizeof(data[0]);
    
    HuffmanNode *root = buildHuffmanTree(data, weights, size);
    HuffmanCode codes[100];
    int codeCount = 0;
    int arr[100], top = 0;
    
    generateCodes(root, arr, top, codes, &codeCount);
    
    printf("哈夫曼编码表:\n");
    for (int i = 0; i < codeCount; i++) {
        printf("%c: %s\n", codes[i].data, codes[i].code);
    }
    
    char message[] = "ABCDE";
    encode(message, codes, codeCount);
    
    char encoded[] = "0000110111111"; // ABCDE的编码
    decode(root, encoded);
    
    freeHuffmanTree(root);
    
    // 选做部分：树的孩子-兄弟链表
    printf("\n===== 孩子-兄弟链表构建 =====\n");
    printf("请输入树的边（格式：FC，^A表示根节点，^^结束）:\n");
    TreeNode *treeRoot = buildTree();
    
    int degree = treeDegree(treeRoot);
    printf("树的度为: %d\n", degree);
    
    freeTree(treeRoot);
    
    return 0;
}
```

## 六、程序调试与分析

1. **调试要点**：
   - 优先队列插入逻辑需确保节点按权值升序排列
   - 编码生成时需注意路径记录数组的初始化与回溯处理
   - 解码过程中需正确重置当前节点到根节点
   - 孩子 - 兄弟链表构建需处理好节点的首次插入与兄弟连接
2. **算法复杂度分析**：
   - 哈夫曼树构建：O (nlogn)（优先队列操作主导）
   - 编码生成：O (n)（遍历每个节点一次）
   - 报文编码 / 解码：O (m)（m 为报文长度）
   - 树的度计算：O (n)（遍历每个节点一次）
