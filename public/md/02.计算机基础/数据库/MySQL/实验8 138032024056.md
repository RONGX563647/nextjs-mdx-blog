### （2）算法思路

1. **插入操作**：
   递归比较关键字，小于根节点则插入左子树，大于则插入右子树，相等则忽略
2. **中序遍历**：
   递归执行左子树→根节点→右子树的访问顺序，输出有序序列
3. **删除操作**：
   - 若节点为叶子节点：直接删除
   - 若节点只有一个子树：用子树替代当前节点
   - 若节点有两个子树：用右子树的最小节点（或左子树的最大节点）替换，再删除该替代节点
4. **合法性验证**：
   递归检查每个节点的关键字是否满足：
   左子树所有节点关键字 < 当前节点关键字 < 右子树所有节点关键字
   （优化：使用中序遍历的有序性验证）

#include <stdio.h>
#include <stdlib.h>

// 二叉排序树节点定义
typedef struct BSTNode {
    int key;
    struct BSTNode *left;
    struct BSTNode *right;
} BSTNode, *BSTree;

// 插入节点
BSTree insertNode(BSTree root, int key) {
    if (root == NULL) {
        BSTNode *newNode = (BSTNode*)malloc(sizeof(BSTNode));
        newNode->key = key;
        newNode->left = newNode->right = NULL;
        return newNode;
    }
    
    if (key < root->key)
        root->left = insertNode(root->left, key);
    else if (key > root->key)
        root->right = insertNode(root->right, key);
    
    return root; // 相等时不插入，保持唯一性
}

// 中序遍历
void inorderTraversal(BSTree root) {
    if (root != NULL) {
        inorderTraversal(root->left);
        printf("%d ", root->key);
        inorderTraversal(root->right);
    }
}

// 查找最小节点
BSTree findMin(BSTree root) {
    while (root->left != NULL)
        root = root->left;
    return root;
}

// 删除节点
BSTree deleteNode(BSTree root, int key) {
    if (root == NULL) return root;
    
    if (key < root->key)
        root->left = deleteNode(root->left, key);
    else if (key > root->key)
        root->right = deleteNode(root->right, key);
    else {
        // 找到待删除节点
        if (root->left == NULL) {
            BSTNode *temp = root->right;
            free(root);
            return temp;
        } else if (root->right == NULL) {
            BSTNode *temp = root->left;
            free(root);
            return temp;
        }
        
        // 有两个子节点的情况
        BSTNode *temp = findMin(root->right);
        root->key = temp->key;
        root->right = deleteNode(root->right, temp->key);
    }
    return root;
}

// 销毁二叉树
void destroyTree(BSTree root) {
    if (root != NULL) {
        destroyTree(root->left);
        destroyTree(root->right);
        free(root);
    }
}

// 选做：验证是否为二叉排序树
int isValidBST(BSTree root, int *prev) {
    if (root == NULL) return 1;
    
    // 递归检查左子树
    if (!isValidBST(root->left, prev)) return 0;
    
    // 检查当前节点
    if (*prev != -1 && root->key <= *prev) return 0;
    *prev = root->key;
    
    // 递归检查右子树
    return isValidBST(root->right, prev);
}

int main() {
    int keys[] = {19, 38, 12, 40, 41, 39, 54, 76, 35, 47, 80, 14, 9, 44, 60};
    int n = sizeof(keys) / sizeof(keys[0]);
    
    BSTree root = NULL;
    
    // ① 构建二叉排序树
    for (int i = 0; i < n; i++) {
        root = insertNode(root, keys[i]);
    }
    
    // ② 第一次中序遍历
    printf("中序遍历结果: ");
    inorderTraversal(root);
    printf("\n");
    
    // ③ 删除节点54
    root = deleteNode(root, 54);
    printf("删除节点54后的中序遍历结果: ");
    inorderTraversal(root);
    printf("\n");
    
    // ⑤ 销毁树
    destroyTree(root);
    root = NULL;
    
    // 选做：验证二叉排序树
    // 重建树用于测试
    for (int i = 0; i < n; i++) {
        root = insertNode(root, keys[i]);
    }
    
    int prev = -1;
    if (isValidBST(root, &prev)) {
        printf("该树是二叉排序树\n");
    } else {
        printf("该树不是二叉排序树\n");
    }
    
    destroyTree(root);
    
    return 0;
}
