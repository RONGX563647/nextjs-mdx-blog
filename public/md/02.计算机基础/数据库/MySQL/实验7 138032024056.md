### （2）算法思路

1. **邻接表构建**：
   - 初始化顶点数组，为每个顶点创建空邻接表
   - 读入边信息，为每条边创建邻接表节点并插入对应顶点链表
2. **深度优先遍历（DFS）**：
   - 递归访问当前顶点的所有未访问邻接顶点
   - 使用辅助数组标记已访问顶点，避免重复访问
3. **广度优先遍历（BFS）**：
   - 使用队列存储待访问顶点
   - 出队访问当前顶点，将其所有未访问邻接顶点入队
4. **路径计数算法**：
   - 递归枚举所有长度为 k 的路径，禁止访问已在路径中的顶点
   - 使用邻接矩阵快速判断边的存在性
5. **Dijkstra 算法**：
   - 维护距离数组记录源点到各顶点的最短距离
   - 使用优先队列（最小堆）选择当前距离最小的顶点进行松弛操作
6. **Prim 算法**：
   - 维护顶点集合 U（已加入生成树的顶点）
   - 每次选择 U 到非 U 集合中权值最小的边，将对应顶点加入 U

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 邻接表节点定义
typedef struct AdjListNode {
    int dest;
    struct AdjListNode* next;
} AdjListNode;

typedef struct AdjList {
    AdjListNode *head;
} AdjList;

typedef struct Graph {
    int V;
    AdjList* array;
} Graph;

// 创建邻接表节点
AdjListNode* newAdjListNode(int dest) {
    AdjListNode* newNode = (AdjListNode*)malloc(sizeof(AdjListNode));
    newNode->dest = dest;
    newNode->next = NULL;
    return newNode;
}

// 创建图
Graph* createGraph(int V) {
    Graph* graph = (Graph*)malloc(sizeof(Graph));
    graph->V = V;
    graph->array = (AdjList*)malloc(V * sizeof(AdjList));
    for (int i = 0; i < V; ++i)
        graph->array[i].head = NULL;
    return graph;
}

// 添加边（无向图）
void addEdge(Graph* graph, int src, int dest) {
    AdjListNode* newNode = newAdjListNode(dest);
    newNode->next = graph->array[src].head;
    graph->array[src].head = newNode;

    newNode = newAdjListNode(src);
    newNode->next = graph->array[dest].head;
    graph->array[dest].head = newNode;
}

// 深度优先遍历辅助函数
void DFSUtil(int v, int visited[], Graph* graph) {
    visited[v] = 1;
    printf("%d ", v);
    AdjListNode* pCrawl = graph->array[v].head;
    while (pCrawl) {
        int adjVertex = pCrawl->dest;
        if (!visited[adjVertex])
            DFSUtil(adjVertex, visited, graph);
        pCrawl = pCrawl->next;
    }
}

// 深度优先遍历
void DFS(Graph* graph, int start) {
    int* visited = (int*)calloc(graph->V, sizeof(int));
    printf("DFS starting from vertex %d: ", start);
    DFSUtil(start, visited, graph);
    printf("\n");
    free(visited);
}

// 队列结构（用于BFS）
typedef struct Queue {
    int front, rear, size;
    unsigned capacity;
    int* array;
} Queue;

Queue* createQueue(unsigned capacity) {
    Queue* queue = (Queue*)malloc(sizeof(Queue));
    queue->capacity = capacity;
    queue->front = queue->size = 0;
    queue->rear = capacity - 1;
    queue->array = (int*)malloc(queue->capacity * sizeof(int));
    return queue;
}

int isEmpty(Queue* queue) {
    return (queue->size == 0);
}

void enqueue(Queue* queue, int item) {
    queue->rear = (queue->rear + 1) % queue->capacity;
    queue->array[queue->rear] = item;
    queue->size = queue->size + 1;
}

int dequeue(Queue* queue) {
    int item = queue->array[queue->front];
    queue->front = (queue->front + 1) % queue->capacity;
    queue->size = queue->size - 1;
    return item;
}

// 广度优先遍历
void BFS(Graph* graph, int start) {
    int* visited = (int*)calloc(graph->V, sizeof(int));
    Queue* queue = createQueue(graph->V);
    
    visited[start] = 1;
    enqueue(queue, start);
    
    printf("BFS starting from vertex %d: ", start);
    
    while (!isEmpty(queue)) {
        int s = dequeue(queue);
        printf("%d ", s);
        
        AdjListNode* pCrawl = graph->array[s].head;
        while (pCrawl) {
            int adjVertex = pCrawl->dest;
            if (!visited[adjVertex]) {
                visited[adjVertex] = 1;
                enqueue(queue, adjVertex);
            }
            pCrawl = pCrawl->next;
        }
    }
    printf("\n");
    
    free(visited);
    free(queue->array);
    free(queue);
}

// 释放图内存
void freeGraph(Graph* graph) {
    if (!graph) return;
    for (int i = 0; i < graph->V; i++) {
        AdjListNode* current = graph->array[i].head;
        while (current) {
            AdjListNode* temp = current;
            current = current->next;
            free(temp);
        }
    }
    free(graph->array);
    free(graph);
}

// 添加有向边（邻接矩阵）
void addDirectedEdge(MGraph* mgraph, int src, int dest) {

}

// 路径计数辅助函数
int countPathsUtil(MGraph* mgraph, int src, int dest, int k, int path[], int pos) {
    if (k == 0 && src == dest) return 1;
    if (k == 0) return 0;
    
    int count = 0;
    
    // 检查路径中是否已包含当前顶点（避免回路）
    for (int i = 0; i < pos; i++) {
        if (path[i] == src) return 0;
    }
    
    path[pos] = src;
    
    for (int i = 0; i < mgraph->V; i++) {
        if (mgraph->matrix[src][i]) {
            count += countPathsUtil(mgraph, i, dest, k-1, path, pos+1);
        }
    }
    
    return count;
}

// 路径计数
int countPaths(MGraph* mgraph, int src, int dest, int k) {
    int* path = (int*)malloc(mgraph->V * sizeof(int));
    int count = countPathsUtil(mgraph, src, dest, k, path, 0);
    free(path);
    return count;
}

// 释放邻接矩阵内存
void freeMGraph(MGraph* mgraph) {

    for (int i = 0; i < mgraph->V; i++) {
        free(mgraph->matrix[i]);
    }
    free(mgraph->matrix);
    free(mgraph);
}

// Dijkstra算法（选做）
#define INF 999999

void dijkstra(int **graph, int V, int src) {
    int* dist = (int*)malloc(V * sizeof(int));
    int* sptSet = (int*)calloc(V, sizeof(int));
    
    for (int i = 0; i < V; i++)
        dist[i] = INF, sptSet[i] = 0;
    
    dist[src] = 0;
    
    for (int count = 0; count < V-1; count++) {
        int min_dist = INF, min_index;
        
        for (int v = 0; v < V; v++)
            if (!sptSet[v] && dist[v] <= min_dist)
                min_dist = dist[v], min_index = v;
        
        int u = min_index;
        sptSet[u] = 1;
        
        for (int v = 0; v < V; v++)
            if (!sptSet[v] && graph[u][v] && dist[u] != INF
                && dist[u]+graph[u][v] < dist[v])
                dist[v] = dist[u] + graph[u][v];
    }
    
    printf("Dijkstra's Shortest Paths from vertex %d:\n", src);
    for (int i = 0; i < V; i++)
        printf("To vertex %d: %d\n", i, dist[i]);
    
    free(dist);
    free(sptSet);
}

// Prim算法（选做）
void primMST(int **graph, int V) {
    int* parent = (int*)malloc(V * sizeof(int));
    int* key = (int*)malloc(V * sizeof(int));
    int* mstSet = (int*)calloc(V, sizeof(int));
    
    for (int i = 0; i < V; i++)
        key[i] = INF, mstSet[i] = 0;
    
    key[0] = 0;
    parent[0] = -1;
    
    for (int count = 0; count < V-1; count++) {
        int min_key = INF, min_index;
        
        for (int v = 0; v < V; v++)
            if (!mstSet[v] && key[v] < min_key)
                min_key = key[v], min_index = v;
        
        int u = min_index;
        mstSet[u] = 1;
        
        for (int v = 0; v < V; v++)
            if (graph[u][v] && !mstSet[v] && graph[u][v] < key[v])
                parent[v] = u, key[v] = graph[u][v];
    }
    
    printf("Prim's Minimum Spanning Tree:\n");
    for (int i = 1; i < V; i++)
        printf("Edge: %d-%d Weight: %d\n", parent[i], i, graph[i][parent[i]]);
    
    free(parent);
    free(key);
    free(mstSet);
}

## 六、程序调试与分析

1. **调试要点**：
   - 邻接表构建时需注意无向图的双向边添加
   - DFS 递归调用需正确维护访问标记数组
   - BFS 队列操作需处理好队满与队空条件
   - 路径计数时需严格检查路径中是否存在回路
   - Dijkstra 算法需正确处理 INF 值的比较与更新
2. **算法复杂度分析**：
   - DFS/BFS 遍历：O (V+E)（邻接表）
   - 路径计数：O (V^k)（枚举所有可能路径）
   - Dijkstra 算法：O (V²)（朴素实现），可优化至 O ((V+E) logV)（优先队列）
   - Prim 算法：O (V²)（适合稠密图）
