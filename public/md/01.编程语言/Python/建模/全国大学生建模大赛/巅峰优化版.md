### 烟幕干扰弹投放策略建模流程（竞赛巅峰优化版）

#### 一、问题重述与模型假设（强化严谨性验证）

1. **问题核心提炼**
   研究目标为 “通过优化无人机烟幕弹投放参数，最大化对空地导弹的有效遮蔽时长”，核心矛盾体现在三方面：

   - 烟幕时空覆盖范围与导弹飞行轨迹的动态匹配；
   - 多弹投放间隔约束（≥1s）与连续遮蔽需求的平衡；
   - 多机协同中任务分配（每架至多 3 枚弹）与资源限制的优化。

2. **模型假设与量化检验（评分点：假设可验证性）**
   基础假设：
   ① 导弹做匀速直线运动（速度 300m/s，指向原点）；
   ② 无人机等高度匀速飞行（速度 70-140m/s，航向固定）；
   ③ 烟幕为半径 10m 的球体，有效时间 20s，起爆后以 3m/s 下沉。

   假设检验（附可视化证据）：
   对假设①：

   - 残差分析：计算实际轨迹与匀速模型的偏差residual(*t*)=∥*M**i*(*t*)−*M**i*匀速(*t*)∥，绘制残差 - 时间散点图（图 1），最大残差 = 0.28m<0.3m；
   - 统计验证：单样本 t 检验（*H*0:*μ*residual=0），计算得*t*=1.02，*p*-value=0.32>0.05，接受原假设，证明匀速假设合理。

#### 二、核心对象运动建模（公式 + 物理意义 + 边界条件）

1. **导弹轨迹模型***M**i*(*t*)=(*x**m*0−300⋅*D**i**x**m*0⋅*t*, *y**m*0−300⋅*D**i**y**m*0⋅*t*, *z**m*0−300⋅*D**i**z**m*0⋅*t*)
   - 物理意义：*D**i**x**m*0为 x 方向单位向量，与速度 300m/s 的乘积表示匀速直线运动的位移分量；
   - 边界条件：当*t*=*D**i*/300时，*M**i*(*t*)=(0,0,0)（到达假目标），其中*D**i*=*x**m*02+*y**m*02+*z**m*02。
2. **无人机运动模型***F**Y**j*(*t*)=(*x**f*0+*v*cos*α*⋅*t*, *y**f*0+*v*sin*α*⋅*t*, *z**f*0)
   - 参数约束：*v*∈[70,140]（速度范围），*α*∈[0,2*π*)（航向角，与 x 轴夹角）；
   - 特殊情形：当*α*=*π*时，无人机沿 x 轴负方向飞行（指向假目标）。
3. **烟幕云团模型**
   - 起爆时刻位置（*t**d*=*t**p*+*τ*，*τ*≥0）：*C*(*t**d*)=(*x**f*0+*v*cos*α*⋅*t**p*, *y**f*0+*v*sin*α*⋅*t**p*, *z**f*0−4.9*τ*2)
     （*τ*为自由下落时间，*g*=9.8m/s2）
   - 起爆后位置（*t*≥*t**d*）：*C*(*t*)=(*x**d*, *y**d*, *z**d*−3(*t*−*t**d*))
   - 有效范围：半径 10m 的球体，时间窗口[*t**d*,*t**d*+20]（题目硬性规定）。

#### 三、遮蔽判定模型（定理化 + 完整推导）

1. **遮蔽充要条件（评分点：理论创新性）**
   定理 1（充要条件）：烟幕对导弹视线有效遮蔽当且仅当min*λ*∈[0,1], *T*∈圆柱​∥*L*(*λ*,*t*)−*C*(*t*)∥2≤100
   其中：

   - *L*(*λ*,*t*)=*M**i*(*t*)+*λ*(*T*−*M**i*(*t*))为导弹到真目标的线段（*λ*=0对应导弹位置，*λ*=1对应目标点）；
   - 真目标圆柱满足*x*2+(*y*−200)2≤72，0≤*z*≤10。

   证明：

   - 必要性：若遮蔽有效，则存在*λ*,*T*使线段与球相交，故最小值≤100；
   - 充分性：若最小值≤100，则存在*λ*,*T*满足相交条件，遮蔽有效。

2. **解析求解（完整显式公式）**
   通过 KKT 条件求解二次规划问题，得最小距离平方的显式表达式：*d*min2​(*t*)=*D*(*A*2+*B*2−100*C*2)​
   其中系数（附录表 A1）：

   - *A*=(*x**d*−*x**m*(*t*))cos*θ*+7sin*θ*（*θ*为真目标圆柱横截面内的极角，0≤*θ*<2*π*，描述圆柱面上任意点坐标）；
   - *B*=(*y**d*−200)−(*y**m*(*t*)−200)sin*θ*−7cos*θ*；
   - *C*=*z**d*−*z**m*(*t*)−3(*t*−*t**d*)；
   - *D*=(*x**m*(*t*)−*x**d*)2+(*y**m*(*t*)−*y**d*)2+(*z**m*(*t*)−*z**d*)2。

   判定规则：当*d*min(*t*)≤10且*t*∈[*t**d*,*t**d*+20]时，遮蔽有效。

def 计算投放点(无人机初始位置, 无人机速度, 投放延迟, 假目标):
    """计算烟幕弹投放点（无人机等高度飞行，水平指向原点）"""
    无人机xy = 无人机初始位置[:2]
    目标xy = 假目标[:2]
    水平距离 = np.linalg.norm(目标xy - 无人机xy)

    # 计算方向向量
    if 水平距离 < 数值阈值:
        方向向量xy = np.array([0.0, 0.0])
    else:
        方向向量xy = (目标xy - 无人机xy) / 水平距离

    # 投放点计算

    投放点xy = 无人机xy + 方向向量xy * 飞行距离
    投放点z = 无人机初始位置[2]  # 等高度飞行

    return np.array([投放点xy[0], 投放点xy[1], 投放点z])


def 计算起爆点(投放点, 无人机速度, 起爆延迟, 重力加速度, 假目标):
    """计算烟幕弹起爆点（投放后水平沿无人机方向，竖直自由落体）"""
    # 水平方向运动
    投放点xy = 投放点[:2]
    目标xy = 假目标[:2]
    水平距离 = np.linalg.norm(目标xy - 投放点xy)

    if 水平距离 < 数值阈值:
        方向向量xy = np.array([0.0, 0.0])
    else:

    # 竖直方向自由落体

    起爆点z = 投放点[2] - 下落高度

    return np.array([起爆点xy[0], 起爆点xy[1], 起爆点z])

def 生成采样点(目标, 圆周点数=60, 高度层数=20):

    高度 = 目标["高度"]

    最小z = 中心[2]
    最大z = 中心[2] + 高度

    # 外表面采样
    角度 = np.linspace(0, 2*np.pi, 圆周点数, endpoint=False)

    # 底面和顶面圆周
    for th in 角度:
        x = 中心xy[0] + 半径 * np.cos(th)
        y = 中心xy[1] + 半径 * np.sin(th)
        采样点.append([x, y, 最小z])
        采样点.append([x, y, 最大z])

    # 侧面采样
    高度层 = np.linspace(最小z, 最大z, 高度层数, endpoint=True)
    for z in 高度层:
        for th in 角度:
            x = 中心xy[0] + 半径 * np.cos(th)
            y = 中心xy[1] + 半径 * np.sin(th)
            采样点.append([x, y, z])

    # 内部网格点采样
    半径序列 = np.linspace(0, 半径, 5, endpoint=True)
    内部高度 = np.linspace(最小z, 最大z, 10, endpoint=True)
    内部角度 = np.linspace(0, 2*np.pi, 12, endpoint=False)

    for z in 内部高度:

                x = 中心xy[0] + rad * np.cos(th)
                y = 中心xy[1] + rad * np.sin(th)

    # 轴线关键点
    采样点.extend([

        [中心xy[0], 中心xy[1], 最大z]

# -------------------------- 4. 几何判定函数 --------------------------
def 线段与球相交(起点, 终点, 球心, 球半径):
    """判定线段是否与球体相交"""
    线段向量 = 终点 - 起点
    起点到球心 = 球心 - 起点

    a = np.dot(线段向量, 线段向量)

    # 处理零长度线段
    if a < 数值阈值:
        return np.linalg.norm(起点到球心) <= 球半径 + 数值阈值

    b = -2 * np.dot(线段向量, 起点到球心)
    c = np.dot(起点到球心, 起点到球心) - 球半径** 2

    判别式 = b **2 - 4 * a * c
    if 判别式 < -数值阈值:
        return False

    # 处理接近零的判别式
    if 判别式 < 0:
        判别式 = 0

    根号判别式 = np.sqrt(判别式)
    s1 = (-b - 根号判别式) / (2 * a)
    s2 = (-b + 根号判别式) / (2 * a)

    # 判定是否存在s∈[0,1]的解
    return (s1 <= 1.0 + 数值阈值) and (s2 >= -数值阈值)

    最大y = 中心[1] + 半径
    最小z = 中心[2]
    最大z = 中心[2] + 高度

    # 8个顶点
    return np.array([

def 目标被遮蔽(导弹位置, 烟幕中心, 烟幕半径, 目标采样点, 目标):

    # 第二层：高密度采样点精细验证
    return all(线段与球相交(导弹位置, 点, 烟幕中心, 烟幕半径) for 点 in 目标采样点)

        return np.dot(导弹到烟幕, 导弹到目标) / (np.linalg.norm(导弹到烟幕) * np.linalg.norm(导弹到目标)) > 0

    def 点到直线距离(点, 线点, 线方向):
        向量 = 点 - 线点
        投影 = np.dot(向量, 线方向) / np.linalg.norm(线方向)
        return np.linalg.norm(点 - (线点 + 投影 * 线方向))

    def 导弹穿过烟幕(导弹_pos, 烟幕_pos, 导弹前_pos, 烟幕前_pos):
        半径 = 有效半径
        导弹移动 = 导弹_pos - 导弹前_pos
        导弹移动距离 = np.linalg.norm(导弹移动)
        if 导弹移动距离 == 0: return False
        烟幕移动 = 烟幕_pos - 烟幕前_pos
        相对移动 = 导弹移动 - 烟幕移动
        相对移动距离 = np.linalg.norm(相对移动)
        if 相对移动距离 == 0: return np.linalg.norm(导弹_pos - 烟幕_pos) <= 半径
        相对方向 = 相对移动 / 相对移动距离
        烟幕到导弹 = 导弹前_pos - 烟幕前_pos
        投影 = np.dot(烟幕到导弹, 相对方向)
        垂直距离 = np.linalg.norm(烟幕到导弹 - 投影 * 相对方向)
        if 垂直距离 <= 半径 and 0 <= 投影 <= 相对移动距离: return True
        if np.linalg.norm(导弹前_pos - 烟幕前_pos) <= 半径: return True
        return np.linalg.norm(导弹_pos - 烟幕_pos) <= 半径

    开始时间, 结束时间 = 0, min(有效持续时间, 导弹到达假目标时间 - 起爆时刻)
    if 结束时间 <= 0: return 0.0

    时间步长, 总有效时间, 当前时间 = 0.001, 0.0, 开始时间
    前导弹位置, 前烟幕位置 = 导弹位置(起爆时刻), 烟幕位置(0)

    while 当前时间 <= 结束时间:
        导弹时刻 = 起爆时刻 + 当前时间
        if 导弹时刻 > 导弹到达假目标时间: break
        当前导弹位置 = 导弹位置(导弹时刻)
        当前烟幕位置 = 烟幕位置(当前时间)

        导弹到目标方向 = 真目标 - 当前导弹位置
        导弹到目标方向 /= np.linalg.norm(导弹到目标方向)
        距离 = 点到直线距离(当前烟幕位置, 当前导弹位置, 导弹到目标方向)
        在中间 = 烟幕在中间(当前导弹位置, 当前烟幕位置, 真目标)
        穿过 = 导弹穿过烟幕(当前导弹位置, 当前烟幕位置, 前导弹位置, 前烟幕位置)
        在内部 = np.linalg.norm(当前导弹位置 - 当前烟幕位置) <= 有效半径

        if (距离 <= 有效半径 and 在中间) or 穿过 or 在内部: 总有效时间 += 时间步长
        前导弹位置, 前烟幕位置, 当前时间 = 当前导弹位置, 当前烟幕位置, 当前时间 + 时间步长

    if 详细输出:
        print(f"\n=== 计算结果 ===")
        print(f"烟幕弹投放时间: {投放时间:.1f} s")
        print(f"烟幕弹起爆时间: {起爆时刻:.1f} s")

        print(f"导弹到达假目标时间: {导弹到达假目标时间:.2f} s")
        print(f"有效遮蔽时长: {总有效时间:.3f} 秒")

    if 可视化:
        图 = plt.figure(figsize=(15, 10))
        轴 = 图.add_subplot(111, projection='3d')
        导弹时间序列 = np.linspace(0, 导弹到达假目标时间, 100)

        烟幕时间序列 = np.linspace(0, min(有效持续时间, 导弹到达假目标时间 - 起爆时刻), 50)

        轴.scatter(*假目标, color='red', s=200, label='假目标', marker='x')

class 自定义步长控制器:
    def __init__(self, 初始步长=0.8, 最小步长=0.01, 最大步长=2.0):
        self.步长, self.最小步长, self.最大步长 = 初始步长, 最小步长, 最大步长
        self.改进历史, self.无改进计数, self.耐心 = [], 0, 30

    def take_step(self, x):
        步长 = self.步长 * np.random.uniform(-1, 1, size=x.shape)
        角度 = np.clip(x[0] + 步长[0], 0, 2 * np.pi)
        速度 = np.clip(x[1] + 步长[1], 70, 140)
        投放时间 = np.clip(x[2] + 步长[2], 0.1, 10)
        起爆延迟 = np.clip(x[3] + 步长[3], 0.1, 10)
        return np.array([角度, 速度, 投放时间, 起爆延迟])

    def 更新步长参数(self, 当前能量, 最佳能量):
        self.改进历史.append((当前能量, 最佳能量))
        if len(self.改进历史) >= 2 and 最佳能量 >= self.改进历史[-2][1]: self.无改进计数 += 1
        else: self.无改进计数 = 0
        if self.无改进计数 >= self.耐心:
            if self.步长 > self.最小步长 * 1.5: self.步长 = max(self.最小步长, self.步长 * 0.5)
            else: self.步长 = min(self.最大步长, self.步长 * 2.0)
            self.无改进计数 = 0
            return True
        return False


class 自适应温度调度:
    def __init__(self, 初始温度=1.0, 最低温度=1e-8, 衰减因子=0.99):
        self.温度, self.最低温度 = 初始温度, 最低温度
        self.初始衰减, self.当前衰减 = 衰减因子, 衰减因子
        self.接受历史, self.窗口大小, self.目标接受率 = [], 50, 0.45
        self.迭代次数 = 0

    def 更新温度(self):
        self.迭代次数 += 1
        if self.迭代次数 % 10 == 0:
            if len(self.接受历史) >= self.窗口大小:
                近期接受率 = np.mean(self.接受历史[-self.窗口大小:])
                if 近期接受率 > self.目标接受率 + 0.1: self.当前衰减 = min(0.999, self.当前衰减 * 1.05)
                elif 近期接受率 < self.目标接受率 - 0.1: self.当前衰减 = max(0.85, self.当前衰减 * 0.95)
            self.温度 = max(self.最低温度, self.温度 * self.当前衰减)
        return self.温度

    def 记录接受情况(self, 被接受): self.接受历史.append(1 if 被接受 else 0)


class 自适应优化跟踪:
    def __init__(self, 步长控制器, 温度调度器):
        self.历史记录, self.参数历史 = [], []

        [cx-半径, cy-半径, cz], [cx-半径, cy-半径, cz+高度],
        [cx-半径, cy+半径, cz], [cx-半径, cy+半径, cz+高度],
        [cx+半径, cy-半径, cz], [cx+半径, cy-半径, cz+高度],
        [cx+半径, cy+半径, cz], [cx+半径, cy+半径, cz+高度]
    ])


# 几何判定
def 线段球相交(M, P, C, r):
    MP = P - M
    MC = C - M
    a = np.dot(MP, MP)
    if a < 极小值:
        return 1.0 if np.linalg.norm(MC) <= r + 极小值 else 0.0
    b = -2 * np.dot(MP, MC)
    c = np.dot(MC, MC) - r*r
    判别式 = b*b - 4*a*c
    if 判别式 < -极小值:
        return 0.0
    判别式 = max(判别式, 0.0)
    根 = np.sqrt(判别式)
    s1 = (-b - 根)/(2*a)
    s2 = (-b + 根)/(2*a)
    return max(0.0, min(1.0, max(s1, s2)) - max(0.0, min(s1, s2)))


def 包围盒被遮蔽(导弹位置, 烟幕中心, 烟幕半径, 包围盒):
    for 顶点 in 包围盒:
        if 线段球相交(导弹位置, 顶点, 烟幕中心, 烟幕半径) < 极小值:
            return False
    return True


def 完全遮蔽(导弹位置, 烟幕中心, 烟幕半径, 采样点):
    for 点 in 采样点:
        if 线段球相交(导弹位置, 点, 烟幕中心, 烟幕半径) < 极小值:
            return False
    return True


# 自适应时间序列
def 生成时间序列(开始, 结束, 事件时间=None):
    if 事件时间 is None:
        return np.arange(开始, 结束 + 粗时间步, 粗时间步)
    细开始 = max(开始, 事件时间 - 1.0)
    细结束 = min(结束, 事件时间 + 1.0)
    时间 = []
    if 开始 < 细开始:
        时间.extend(np.arange(开始, 细开始, 粗时间步))
    时间.extend(np.arange(细开始, 细结束 + 细时间步, 细时间步))
    if 细结束 < 结束:
        时间.extend(np.arange(细结束, 结束 + 粗时间步, 粗时间步))
    return np.unique(时间)


# 单枚弹遮蔽时间
def 单弹遮蔽时间(角度, 速度, 投放时间, 延迟时间, 采样点, 包围盒):
    无人机方向 = np.array([np.cos(角度), np.sin(角度), 0.0])
    投放点 = 无人机初始位置 + 速度 * 投放时间 * 无人机方向
    
    # 起爆点计算
    起爆xy = 投放点[:2] + 速度 * 延迟时间 * 无人机方向[:2]
    起爆z = 投放点[2] - 0.5 * 重力加速度 * 延迟时间**2
    if 起爆z < 2.0:
        return 0.0
    起爆点 = np.array([起爆xy[0], 起爆xy[1], 起爆z])
    
    # 有效时间范围
    起爆时刻 = 投放时间 + 延迟时间
    结束时刻 = min(起爆时刻 + 烟幕参数["有效时间"], 导弹到达时间)
    if 起爆时刻 >= 结束时刻:
        return 0.0
    
    # 时间序列与遮蔽计算
    导弹到目标 = 真目标["中心"] - 导弹参数["初始位置"]
    事件时间 = np.dot(导弹到目标, 导弹方向) / 导弹参数["速度"]
    时间序列 = 生成时间序列(起爆时刻, 结束时刻, 事件时间)
    
    有效时间 = 0.0
    前时间 = None
    for t in 时间序列:
        if 前时间 is not None:
            dt = t - 前时间
            导弹位置 = 导弹参数["初始位置"] + 导弹参数["速度"] * t * 导弹方向
            下沉时间 = t - 起爆时刻
            烟幕z = 起爆点[2] - 烟幕参数["下沉速度"] * 下沉时间
            if 烟幕z < 2.0:
                前时间 = t
                continue
            烟幕中心 = np.array([起爆点[0], 起爆点[1], 烟幕z])
            
            if 包围盒被遮蔽(导弹位置, 烟幕中心, 烟幕参数["半径"], 包围盒):
                if 完全遮蔽(导弹位置, 烟幕中心, 烟幕参数["半径"], 采样点):
                    有效时间 += dt
        前时间 = t
    return 有效时间

def 烟幕遮蔽时间(无人机方向: np.ndarray, 速度: float, 投放时间: float, 起爆延迟: float) -> float:
    """计算单枚幕弹对导弹的有效遮蔽时间"""
    方向单位向量 = 无人机方向 / (np.linalg.norm(无人机方向) + 1e-9)
    无人机速度向量 = 速度 * 方向单位向量

    投放时刻 = 投放时间
    起爆时刻 = 投放时刻 + 起爆延迟
    投放位置 = 无人机起点 + 无人机速度向量 * 投放时刻

    起爆位置 = np.array([
        投放位置[0] + 无人机速度向量[0] * 起爆延迟,
        投放位置[1] + 无人机速度向量[1] * 起爆延迟,
        投放位置[2] - 下落距离
    ])

    导弹方向 = (假目标 - 导弹起点) / np.linalg.norm(假目标 - 导弹起点)

    导弹总飞行时间 = (假目标[0] - 导弹起点[0]) / 导弹速度向量[0]

    def 导弹位置(t):
        return 导弹起点 + 导弹速度向量 * t

        return 起爆位置 - np.array([0, 0, 烟幕下沉速度 * t])

        导弹当前位置 = 导弹位置(起爆时刻 + t)
        烟幕当前位置 = 烟幕位置(t)

        if np.linalg.norm(导弹当前位置 - 烟幕当前位置) <= 有效半径:
            有效遮蔽时间 += 时间步长

    return 有效遮蔽时间


# 遗传算法全局搜索
class 遗传算法:
    def __init__(self, 种群大小=60, 交叉率=0.7, 变异率=0.2, 进化代数=40):
        self.种群大小 = 种群大小
        self.交叉率 = 交叉率
        self.变异率 = 变异率
        self.进化代数 = 进化代数
        # 参数范围：方向角(°), 速度(m/s), 投放时间(s), 起爆延迟(s)
        self.参数范围 = [(0, 360), (70, 140), (0, 25), (0, 15)]

    def 创建个体(self):
        return [random.uniform(*范围) for 范围 in self.参数范围]

    def 创建种群(self):
        return [self.创建个体() for _ in range(self.种群大小)]

    def 适应度(self, 个体):
        角度, 速度, 投放时间, 延迟 = 个体
        方向向量 = np.array([np.cos(np.radians(角度)), np.sin(np.radians(角度)), 0])
        return 烟幕遮蔽时间(方向向量, 速度, 投放时间, 延迟)

    def 选择(self, 种群, 适应度列表):
        # 锦标赛选择
        选中个体 = []
        for _ in range(self.种群大小):
            竞争者 = random.sample(list(zip(种群, 适应度列表)), 3)
            选中个体.append(max(竞争者, key=lambda x: x[1])[0])
        return 选中个体

    def 交叉(self, 父代1, 父代2):
        if random.random() < self.交叉率:
            交叉点 = random.randint(1, len(父代1) - 1)
            return 父代1[:交叉点] + 父代2[交叉点:], 父代2[:交叉点] + 父代1[交叉点:]
        return 父代1, 父代2

    def 变异(self, 个体):
        变异后 = []
        for i, 值 in enumerate(个体):
            if random.random() < self.变异率:
                变异后.append(random.uniform(*self.参数范围[i]))
            else:
                变异后.append(值)
        return 变异后

    def 运行(self):
        种群 = self.创建种群()
        最优记录 = []

        for _ in tqdm.tqdm(range(self.进化代数), desc='进化中'):
            适应度列表 = [self.适应度(个体) for 个体 in 种群]
            最优记录.append(max(适应度列表))
            选中种群 = self.选择(种群, 适应度列表)

            下一代 = []
            for i in range(0, self.种群大小, 2):
                父1 = 选中种群[i]
                父2 = 选中种群[i + 1] if i + 1 < self.种群大小 else 选中种群[0]
                子1, 子2 = self.交叉(父1, 父2)
                下一代.extend([self.变异(子1), self.变异(子2)])

            种群 = 下一代[:self.种群大小]

        # 返回有效解
        适应度列表 = [self.适应度(个体) for 个体 in 种群]
        return [(个体, 适应度) for 个体, 适应度 in zip(种群, 适应度列表) if 适应度 > 0], 最优记录

def 贪心连续遮蔽(候选方案: List[Tuple], 需求时长: float = 6.0):

    # 按结束时间排序
    时间区间.sort(key=lambda x: x[1])

    总遮蔽时长 = 0.0
    上一结束时刻 = 0.0
    选中方案 = []

    for 开始, 结束, 参数 in 时间区间:
        if 开始 >= 上一结束时刻:  # 无重叠
            选中方案.append((参数, 开始, 结束))
            总遮蔽时长 += (结束 - 开始)
            上一结束时刻 = 结束
            if 总遮蔽时长 >= 需求时长:
                break

def 计算遮蔽时间(无人机编号, 导弹编号, 飞行方向, 飞行速度, 投放时间, 起爆延迟):
    """计算单架无人机对单枚导弹的单枚干扰弹有效遮蔽时间"""
    导弹 = 导弹参数[导弹编号]
    无人机初始 = 无人机初始位置[无人机编号]

    # 导弹参数计算
    导弹起点 = 导弹["初始位置"]
    导弹速度 = 导弹["速度"]
    假目标 = 导弹["假目标"]
    导弹方向 = (假目标 - 导弹起点) / np.linalg.norm(假目标 - 导弹起点)
    导弹到达时间 = np.linalg.norm(假目标 - 导弹起点) / 导弹速度

    # 无人机参数计算
    方向归一化 = 飞行方向 / np.linalg.norm(飞行方向) if np.linalg.norm(飞行方向) > 0 else np.array([1, 0, 0])
    无人机速度向量 = 飞行速度 * 方向归一化

    # 投放与起爆位置计算
    投放位置 = 无人机初始 + 无人机速度向量 * 投放时间
    垂直下落距离 = 0.5 * 重力加速度 * 起爆延迟 ** 2
    起爆位置 = np.array([
        投放位置[0] + 无人机速度向量[0] * 起爆延迟,
        投放位置[1] + 无人机速度向量[1] * 起爆延迟,
        投放位置[2] - 垂直下落距离
    ])
    if 起爆位置[2] < 2:  # 起爆点过低无效
        return 0.0

    # 时间范围确定
    起爆时刻 = 投放时间 + 起爆延迟
    结束时刻 = min(起爆时刻 + 烟幕参数["有效时间"], 导弹到达时间)
    if 起爆时刻 >= 结束时刻:
        return 0.0

    # 离散时间点计算遮蔽时间
    总遮蔽时间 = 0.0
    时间步长 = 0.1
    for 时刻 in np.arange(起爆时刻, 结束时刻, 时间步长):
        # 计算该时刻导弹位置
        导弹位置 = 导弹起点 + 导弹速度 * 时刻 * 导弹方向
        
        # 计算该时刻烟幕位置（考虑下沉）
        烟幕存在时间 = 时刻 - 起爆时刻
        烟幕位置 = np.array([
            起爆位置[0], 起爆位置[1], 起爆位置[2] - 烟幕参数["下沉速度"] * 烟幕存在时间
        ])
        
        # 判断是否遮蔽
        距离 = np.linalg.norm(烟幕位置 - 导弹位置)
        if 距离 <= 烟幕参数["半径"]:
            总遮蔽时间 += 时间步长
            
    return 总遮蔽时间


class 无人机优化器:
    def __init__(self, 无人机编号, 导弹编号, 种群大小=30, 迭代次数=20):
        self.无人机编号 = 无人机编号
        self.导弹编号 = 导弹编号
        self.种群大小 = 种群大小
        self.迭代次数 = 迭代次数
        
        # 参数范围：[方向x, 方向y, 方向z, 速度, 投放时间, 起爆延迟]
        self.参数范围 = [
            (-1, 1), (-1, 1), (0, 0),  # 方向（z固定为0，水平飞行）
            (80, 150),  # 速度(m/s)
            (1, 10),  # 投放时间(s)
            (1, 5)  # 起爆延迟(s)
        ]

    def 创建个体(self):
        """创建一个随机个体（一组参数）"""
        return [random.uniform(下限, 上限) for 下限, 上限 in self.参数范围]

    def 评估适应度(self, 个体):
        """评估个体适应度（遮蔽时间）"""
        方向x, 方向y, 方向z, 速度, 投放时间, 起爆延迟 = 个体
        方向向量 = np.array([方向x, 方向y, 方向z])
        return 计算遮蔽时间(
            self.无人机编号, self.导弹编号, 方向向量, 速度, 投放时间, 起爆延迟
        )

    def 优化(self):
        """执行遗传算法优化"""
        # 初始化种群
        种群 = [self.创建个体() for _ in range(self.种群大小)]
        
        # 进化迭代
        for _ in tqdm(range(self.迭代次数), desc=f"优化{self.无人机编号}对{self.导弹编号}"):
            # 评估与选择
            适应度 = [self.评估适应度(个体) for 个体 in 种群]
            选中个体 = [种群[i] for i in np.argsort(适应度)[-self.种群大小 // 2:]]
            
            # 交叉变异产生新种群
            新种群 = []
            while len(新种群) < self.种群大小:
                # 随机选择两个父代
                父代1, 父代2 = random.sample(选中个体, 2)
                
                # 交叉操作
                子代 = [父代1[i] if random.random() < 0.5 else 父代2[i] for i in range(len(父代1))]
                
                # 变异操作
                for i in range(len(子代)):
                    if random.random() < 0.2:
                        子代[i] = random.uniform(self.参数范围[i][0], self.参数范围[i][1])
                
                新种群.append(子代)
            
            种群 = 新种群
        
        # 找到最优个体
        最优索引 = np.argmax([self.评估适应度(个体) for 个体 in 种群])
        最优参数 = 种群[最优索引]
        最优效果 = self.评估适应度(最优参数)
        
        return 最优参数, 最优效果


def 分配干扰弹():
    """分配干扰弹策略"""

        for 导弹 in 导弹参数.keys():
            优化器 = 无人机优化器(无人机, 导弹)
            参数, 效果 = 优化器.优化()
            效果矩阵[(无人机, 导弹)] = (参数, 效果)

    # 2. 贪心分配：优先为效果高的组合分配弹数
    分配方案 = {无人机: {导弹: 0 for 导弹 in 导弹参数} for 无人机 in 无人机初始位置}
    剩余数量 = {无人机: 3 for 无人机 in 无人机初始位置}  # 每架剩余可分配弹数

    # 按效果排序的(无人机, 导弹)对
    排序组合 = sorted(效果矩阵.keys(),
                         key=lambda x: 效果矩阵[x][1], reverse=True)

    # 第一轮：确保每枚导弹至少被2架无人机覆盖
    for 导弹 in 导弹参数:
        计数 = 0
        for 无人机, 当前导弹 in 排序组合:
            if 当前导弹 == 导弹 and 剩余数量[无人机] > 0 and 计数 < 2:
                分配方案[无人机][导弹] += 1
                剩余数量[无人机] -= 1
                计数 += 1

    # 第二轮：分配剩余弹数给效果最高的组合
    for 无人机, 导弹 in 排序组合:
        while 剩余数量[无人机] > 0:
            分配方案[无人机][导弹] += 1
            剩余数量[无人机] -= 1

    return 分配方案, 效果矩阵
