```python
from numba import njit
import numpy as np

@njit(cache=True)
def get_interval(v, alpha, t_p, tau, x_f0, y_f0, z_f0, m0):
    t_d = t_p + tau
    x_d = x_f0 + v * np.cos(alpha) * t_p
    y_d = y_f0 + v * np.sin(alpha) * t_p
    z_d = z_f0 - 4.9 * tau**2
    # 定理1 显式系数（对应正文公式）
    A, B, C, D = compute_coeff(x_d, y_d, z_d, t_d, m0)
    t1, t2 = solve_quadratic(A, B, C)   # 解析根
    return max(t1, t_d), min(t2, t_d + 20)
```
