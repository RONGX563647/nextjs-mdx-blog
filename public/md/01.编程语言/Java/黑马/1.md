# 完整版简易MyBatis框架实现

下面是一个功能完整的简易MyBatis框架实现，包含了XML/注解解析、完整的缓存机制、动态SQL、事务管理和连接池集成：

package com.simple.mybatis;

        try (Reader reader = Resources.getResourceAsReader(configPath)) {

        } catch (Exception e) {

        this.sqlSessionFactory = new DefaultSqlSessionFactory(configuration);

        return sqlSessionFactory.openSession(autoCommit);
    }
    
    public SqlSessionFactory getSqlSessionFactory() {
        return sqlSessionFactory;
    }
}

package com.simple.mybatis;

import java.io.*;

/**
 * 资源工具类
 */
public class Resources {
    
    public static Reader getResourceAsReader(String resource) throws IOException {
        return new InputStreamReader(getResourceAsStream(resource));
    }
    
    public static InputStream getResourceAsStream(String resource) throws IOException {
        ClassLoader[] classLoaders = getClassLoaders();
        for (ClassLoader classLoader : classLoaders) {
            InputStream inputStream = classLoader.getResourceAsStream(resource);
            if (inputStream != null) {
                return inputStream;
            }
        }
        throw new IOException("Could not find resource: " + resource);
    }
    
    private static ClassLoader[] getClassLoaders() {
        return new ClassLoader[]{
            Thread.currentThread().getContextClassLoader(),
            Resources.class.getClassLoader()
        };
    }
}

// XML配置构建器 - XmlConfigBuilder.java
package com.simple.mybatis.builder;

import com.simple.mybatis.config.Configuration;
import com.simple.mybatis.config.Environment;
import com.simple.mybatis.io.Resources;
import org.w3c.dom.Document;
import org.w3c.dom.Element;
import org.w3c.dom.Node;
import org.w3c.dom.NodeList;

import javax.sql.DataSource;
import javax.xml.parsers.DocumentBuilder;
import javax.xml.parsers.DocumentBuilderFactory;
import java.io.Reader;
import java.util.Properties;

/**
 * XML配置构建器
 */
public class XmlConfigBuilder {
    
    private final Reader reader;
    private final Configuration configuration;
    private Properties props;
    
    public XmlConfigBuilder(Reader reader) {
        this(reader, new Configuration(), null);
    }
    
    public XmlConfigBuilder(Reader reader, Configuration configuration, Properties props) {
        this.reader = reader;
        this.configuration = configuration;
        this.props = props;
    }
    
    public Configuration parse() {
        try {
            DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
            DocumentBuilder builder = factory.newDocumentBuilder();
            Document document = builder.parse(new org.xml.sax.InputSource(reader));
            Element root = document.getDocumentElement();
            parseConfiguration(root);
            return configuration;
        } catch (Exception e) {
            throw new RuntimeException("Error parsing SQL Mapper Configuration. Cause: " + e, e);
        }
    }
    
    private void parseConfiguration(Element root) {
        // 解析environments
        NodeList environments = root.getElementsByTagName("environments");
        if (environments.getLength() > 0) {
            parseEnvironments((Element) environments.item(0));
        }
        
        // 解析mappers
        NodeList mappers = root.getElementsByTagName("mappers");
        if (mappers.getLength() > 0) {
            parseMappers((Element) mappers.item(0));
        }
        
        // 解析settings
        NodeList settings = root.getElementsByTagName("settings");
        if (settings.getLength() > 0) {
            parseSettings((Element) settings.item(0));
        }
    }
    
    private void parseEnvironments(Element environments) {
        String defaultEnv = environments.getAttribute("default");
        NodeList environmentList = environments.getElementsByTagName("environment");
        
        for (int i = 0; i < environmentList.getLength(); i++) {
            Element environment = (Element) environmentList.item(i);
            String id = environment.getAttribute("id");
            
            if (id.equals(defaultEnv)) {
                parseEnvironment(environment);
                break;
            }
        }
    }
    
    private void parseEnvironment(Element environment) {
        Element dataSourceElement = (Element) environment.getElementsByTagName("dataSource").item(0);
        DataSource dataSource = getDataSource(dataSourceElement);
        
        Environment env = new Environment.Builder(environment.getAttribute("id"))
            .dataSource(dataSource)
            .build();
        
        configuration.setEnvironment(env);
    }
    
    private DataSource getDataSource(Element dataSource) {
        String type = dataSource.getAttribute("type");
        Properties props = new Properties();
        
        NodeList properties = dataSource.getElementsByTagName("property");
        for (int i = 0; i < properties.getLength(); i++) {
            Element property = (Element) properties.item(i);
            String name = property.getAttribute("name");
            String value = property.getAttribute("value");
            props.setProperty(name, value);
        }
        
        // 这里简化实现，实际应该根据type创建不同的数据源
        try {
            Class<?> dataSourceClass = Resources.classForName(type);
            DataSource ds = (DataSource) dataSourceClass.newInstance();
            
            // 使用反射设置属性
            for (String key : props.stringPropertyNames()) {
                String methodName = "set" + key.substring(0, 1).toUpperCase() + key.substring(1);
                java.lang.reflect.Method method = dataSourceClass.getMethod(methodName, String.class);
                method.invoke(ds, props.getProperty(key));
            }
            
            return ds;
        } catch (Exception e) {
            throw new RuntimeException("Error creating datasource. Cause: " + e, e);
        }
    }
    
    private void parseMappers(Element mappers) {
        NodeList mapperList = mappers.getElementsByTagName("mapper");
        for (int i = 0; i < mapperList.getLength(); i++) {
            Element mapper = (Element) mapperList.item(i);
            String resource = mapper.getAttribute("resource");
            parseMapper(resource);
        }
    }
    
    private void parseMapper(String resource) {
        try (Reader reader = Resources.getResourceAsReader(resource)) {
            XmlMapperBuilder mapperParser = new XmlMapperBuilder(reader, configuration, resource);
            mapperParser.parse();
        } catch (Exception e) {
            throw new RuntimeException("Error parsing mapper resource: " + resource, e);
        }
    }
    
    private void parseSettings(Element settings) {
        NodeList settingList = settings.getElementsByTagName("setting");
        for (int i = 0; i < settingList.getLength(); i++) {
            Element setting = (Element) settingList.item(i);
            String name = setting.getAttribute("name");
            String value = setting.getAttribute("value");
            
            if ("cacheEnabled".equals(name)) {
                configuration.setCacheEnabled(Boolean.parseBoolean(value));
            } else if ("useGeneratedKeys".equals(name)) {
                configuration.setUseGeneratedKeys(Boolean.parseBoolean(value));
            } else if ("defaultStatementTimeout".equals(name)) {
                configuration.setDefaultStatementTimeout(Integer.parseInt(value));
            } else if ("defaultFetchSize".equals(name)) {
                configuration.setDefaultFetchSize(Integer.parseInt(value));
            }
        }
    }
}

// XML Mapper构建器 - XmlMapperBuilder.java
package com.simple.mybatis.builder;

import com.simple.mybatis.config.Configuration;
import com.simple.mybatis.mapping.MappedStatement;
import com.simple.mybatis.mapping.SqlCommandType;
import com.simple.mybatis.mapping.SqlSource;
import com.simple.mybatis.scripting.LanguageDriver;
import com.simple.mybatis.scripting.xmltags.DynamicSqlSource;
import com.simple.mybatis.scripting.xmltags.RawSqlSource;
import com.simple.mybatis.scripting.xmltags.XMLLanguageDriver;
import org.w3c.dom.Document;
import org.w3c.dom.Element;
import org.w3c.dom.Node;
import org.w3c.dom.NodeList;

import javax.xml.parsers.DocumentBuilder;
import javax.xml.parsers.DocumentBuilderFactory;
import java.io.Reader;
import java.util.regex.Pattern;

/**
 * XML Mapper构建器
 */
public class XmlMapperBuilder {
    
    private final Reader reader;
    private final Configuration configuration;
    private final String resource;
    private final LanguageDriver languageDriver;
    
    public XmlMapperBuilder(Reader reader, Configuration configuration, String resource) {
        this.reader = reader;
        this.configuration = configuration;
        this.resource = resource;
        this.languageDriver = new XMLLanguageDriver();
    }
    
    public void parse() {
        try {
            DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
            DocumentBuilder builder = factory.newDocumentBuilder();
            Document document = builder.parse(new org.xml.sax.InputSource(reader));
            Element root = document.getDocumentElement();
            parseMapper(root);
        } catch (Exception e) {
            throw new RuntimeException("Error parsing Mapper XML. Cause: " + e, e);
        }
    }
    
    private void parseMapper(Element root) {
        String namespace = root.getAttribute("namespace");
        if (namespace == null || namespace.isEmpty()) {
            throw new RuntimeException("Mapper's namespace cannot be empty");
        }
        
        // 解析cache
        NodeList caches = root.getElementsByTagName("cache");
        if (caches.getLength() > 0) {
            // 缓存配置解析
        }
        
        // 解析SQL语句
        parseStatements(root, namespace);
    }
    
    private void parseStatements(Element root, String namespace) {
        NodeList children = root.getChildNodes();
        for (int i = 0; i < children.getLength(); i++) {
            Node node = children.item(i);
            if (node instanceof Element) {
                Element element = (Element) node;
                String nodeName = element.getNodeName();
                
                if (isStatementElement(nodeName)) {
                    parseStatement(element, namespace);
                }
            }
        }
    }
    
    private boolean isStatementElement(String nodeName) {
        return "select".equals(nodeName) || "insert".equals(nodeName) || 
               "update".equals(nodeName) || "delete".equals(nodeName);
    }
    
    private void parseStatement(Element element, String namespace) {
        String id = element.getAttribute("id");
        String statementId = namespace + "." + id;
        
        SqlCommandType sqlCommandType = SqlCommandType.valueOf(element.getNodeName().toUpperCase());
        String parameterType = element.getAttribute("parameterType");
        String resultType = element.getAttribute("resultType");
        String resultMap = element.getAttribute("resultMap");
        
        // 解析SQL
        SqlSource sqlSource = createSqlSource(element);
        
        MappedStatement.Builder statementBuilder = new MappedStatement.Builder(configuration, statementId, sqlSource, sqlCommandType);
        
        if (parameterType != null && !parameterType.isEmpty()) {
            Class<?> parameterClass = resolveClass(parameterType);
            statementBuilder.parameterType(parameterClass);
        }
        
        if (resultType != null && !resultType.isEmpty()) {
            Class<?> resultClass = resolveClass(resultType);
            statementBuilder.resultType(resultClass);
        }
        
        // 其他属性设置
        String timeout = element.getAttribute("timeout");
        if (timeout != null && !timeout.isEmpty()) {
            statementBuilder.timeout(Integer.parseInt(timeout));
        }
        
        String useGeneratedKeys = element.getAttribute("useGeneratedKeys");
        if (useGeneratedKeys != null && !useGeneratedKeys.isEmpty()) {
            statementBuilder.useGeneratedKeys(Boolean.parseBoolean(useGeneratedKeys));
        }
        
        MappedStatement mappedStatement = statementBuilder.build();
        configuration.addMappedStatement(mappedStatement);
    }
    
    private SqlSource createSqlSource(Element element) {
        String sql = getSql(element);
        
        // 检查是否包含动态SQL标签
        if (hasDynamicTags(sql)) {
            return languageDriver.createSqlSource(configuration, element, null);
        } else {
            return new RawSqlSource(configuration, sql, null);
        }
    }
    
    private String getSql(Element element) {
        StringBuilder sqlBuilder = new StringBuilder();
        NodeList children = element.getChildNodes();
        for (int i = 0; i < children.getLength(); i++) {
            Node node = children.item(i);
            if (node.getNodeType() == Node.TEXT_NODE) {
                sqlBuilder.append(node.getNodeValue());
            }
        }
        return sqlBuilder.toString().trim();
    }
    
    private boolean hasDynamicTags(String sql) {
        // 简单的动态SQL标签检测
        Pattern pattern = Pattern.compile("<[^>]+>");
        return pattern.matcher(sql).find();
    }
    
    private Class<?> resolveClass(String className) {
        try {
            return Resources.classForName(className);
        } catch (ClassNotFoundException e) {
            throw new RuntimeException("Error resolving class. Cause: " + e, e);
        }
    }
}

// 默认SQL会话实现 - DefaultSqlSession.java
package com.simple.mybatis.session.defaults;

import com.simple.mybatis.cache.CacheKey;
import com.simple.mybatis.config.Configuration;
import com.simple.mybatis.executor.Executor;
import com.simple.mybatis.executor.SimpleExecutor;
import com.simple.mybatis.mapping.MappedStatement;
import com.simple.mybatis.proxy.MapperProxy;
import com.simple.mybatis.session.RowBounds;
import com.simple.mybatis.session.SqlSession;

import java.lang.reflect.Proxy;
import java.sql.Connection;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

/**
 * 默认SQL会话实现
 */
public class DefaultSqlSession implements SqlSession {
    
    private final Configuration configuration;
    private final Executor executor;
    private final boolean autoCommit;
    private boolean dirty;
    
    public DefaultSqlSession(Configuration configuration) {
        this(configuration, true);
    }
    
    public DefaultSqlSession(Configuration configuration, boolean autoCommit) {
        this.configuration = configuration;
        this.executor = new SimpleExecutor(configuration);
        this.autoCommit = autoCommit;
        this.dirty = false;
    }
    
    @Override
    public <T> T selectOne(String statement) {
        return this.selectOne(statement, null);
    }
    
    @Override
    @SuppressWarnings("unchecked")
    public <T> T selectOne(String statement, Object parameter) {
        List<T> list = this.selectList(statement, parameter);
        if (list.size() == 1) {
            return list.get(0);
        } else if (list.size() > 1) {
            throw new RuntimeException("Expected one result (or null) to be returned by selectOne(), but found: " + list.size());
        } else {
            return null;
        }
    }
    
    @Override
    public <E> List<E> selectList(String statement) {
        return this.selectList(statement, null);
    }
    
    @Override
    public <E> List<E> selectList(String statement, Object parameter) {
        return this.selectList(statement, parameter, RowBounds.DEFAULT);
    }
    
    @Override
    public <E> List<E> selectList(String statement, Object parameter, RowBounds rowBounds) {
        try {
            MappedStatement mappedStatement = configuration.getMappedStatement(statement);
            CacheKey cacheKey = executor.createCacheKey(mappedStatement, parameter, rowBounds);
            return executor.query(mappedStatement, parameter, rowBounds, cacheKey);
        } catch (Exception e) {
            throw new RuntimeException("Error executing query for statement: " + statement, e);
        }
    }
    
    @Override
    public <K, V> Map<K, V> selectMap(String statement, String mapKey) {
        return this.selectMap(statement, null, mapKey, RowBounds.DEFAULT);
    }
    
    @Override
    public <K, V> Map<K, V> selectMap(String statement, Object parameter, String mapKey) {
        return this.selectMap(statement, parameter, mapKey, RowBounds.DEFAULT);
    }
    
    @Override
    public <K, V> Map<K, V> selectMap(String statement, Object parameter, String mapKey, RowBounds rowBounds) {
        final List<?> list = selectList(statement, parameter, rowBounds);
        final Map<K, V> map = new HashMap<>();
        for (Object object : list) {
            @SuppressWarnings("unchecked")
            final K key = (K) getValue(object, mapKey);
            @SuppressWarnings("unchecked")
            final V value = (V) object;
            map.put(key, value);
        }
        return map;
    }
    
    private Object getValue(Object object, String property) {
        try {
            java.lang.reflect.Field field = object.getClass().getDeclaredField(property);
            field.setAccessible(true);
            return field.get(object);
        } catch (Exception e) {
            throw new RuntimeException("Error getting property value: " + property, e);
        }
    }
    
    @Override
    public int insert(String statement) {
        return insert(statement, null);
    }
    
    @Override
    public int insert(String statement, Object parameter) {
        return update(statement, parameter);
    }
    
    @Override
    public int update(String statement) {
        return update(statement, null);
    }
    
    @Override
    public int update(String statement, Object parameter) {
        try {
            dirty = true;
            MappedStatement mappedStatement = configuration.getMappedStatement(statement);
            return executor.update(mappedStatement, parameter);
        } catch (Exception e) {
            throw new RuntimeException("Error executing update for statement: " + statement, e);
        }
    }
    
    @Override
    public int delete(String statement) {
        return delete(statement, null);
    }
    
    @Override
    public int delete(String statement, Object parameter) {
        return update(statement, parameter);
    }
    
    @Override
    public void commit() {
        commit(false);
    }
    
    @Override
    public void commit(boolean force) {
        try {
            executor.commit(isCommitOrRollbackRequired(force));
            dirty = false;
        } catch (Exception e) {
            throw new RuntimeException("Error committing transaction", e);
        }
    }
    
    @Override
    public void rollback() {
        rollback(false);
    }
    
    @Override
    public void rollback(boolean force) {
        try {
            executor.rollback(isCommitOrRollbackRequired(force));
            dirty = false;
        } catch (Exception e) {
            throw new RuntimeException("Error rolling back transaction", e);
        }
    }
    
    private boolean isCommitOrRollbackRequired(boolean force) {
        return (!autoCommit && dirty) || force;
    }
    
    @Override
    public void close() {
        try {
            executor.close(isCommitOrRollbackRequired(false));
            dirty = false;
        } catch (Exception e) {
            throw new RuntimeException("Error closing session", e);
        }
    }
    
    @Override
    public void clearCache() {
        executor.clearLocalCache();
    }
    
    @Override
    public Configuration getConfiguration() {
        return configuration;
    }
    
    @Override
    @SuppressWarnings("unchecked")
    public <T> T getMapper(Class<T> type) {
        return (T) Proxy.newProxyInstance(
            type.getClassLoader(),
            new Class[]{type},
            new MapperProxy<>(this, type)
        );
    }
    
    @Override
    public Connection getConnection() {
        try {
            return executor.getTransaction().getConnection();
        } catch (Exception e) {

package com.simple.mybatis.executor;

import com.simple.mybatis.cache.CacheKey;
import com.simple.mybatis.cache.impl.PerpetualCache;
import com.simple.mybatis.mapping.MappedStatement;
import com.simple.mybatis.session.RowBounds;
import com.simple.mybatis.transaction.Transaction;

import java.sql.SQLException;
import java.util.List;

/**
 * 基础执行器
 */
public abstract class BaseExecutor implements Executor {
    
    protected Transaction transaction;
    protected Executor wrapper;
    
    private boolean closed;
    protected final PerpetualCache localCache;
    
    public BaseExecutor(Transaction transaction) {
        this.transaction = transaction;
        this.wrapper = this;
        this.localCache = new PerpetualCache("LocalCache");
    }
    
    @Override
    public <E> List<E> query(MappedStatement ms, Object parameter, RowBounds rowBounds) throws SQLException {
        CacheKey key = createCacheKey(ms, parameter, rowBounds);
        return query(ms, parameter, rowBounds, key);
    }
    
    @Override
    public <E> List<E> query(MappedStatement ms, Object parameter, RowBounds rowBounds, CacheKey key) throws SQLException {
        // 尝试从缓存中获取
        List<E> list = (List<E>) localCache.getObject(key);
        if (list != null) {
            return list;
        }
        
        // 缓存中没有，执行查询
        list = queryFromDatabase(ms, parameter, rowBounds, key);
        return list;
    }
    
    protected <E> List<E> queryFromDatabase(MappedStatement ms, Object parameter, RowBounds rowBounds, CacheKey key) throws SQLException {
        List<E> list;
        localCache.putObject(key, ExecutionPlaceholder.EXECUTION_PLACEHOLDER);
        try {
            list = doQuery(ms, parameter, rowBounds);
        } finally {
            localCache.removeObject(key);
        }
        localCache.putObject(key, list);
        return list;
    }
    
    protected abstract <E> List<E> doQuery(MappedStatement ms, Object parameter, RowBounds rowBounds) throws SQLException;
    
    @Override
    public int update(MappedStatement ms, Object parameter) throws SQLException {
        clearLocalCache();
        return doUpdate(ms, parameter);
    }
    
    protected abstract int doUpdate(MappedStatement ms, Object parameter) throws SQLException;
    
    @Override
    public Transaction getTransaction() {
        return transaction;
    }
    
    @Override
    public void commit(boolean required) throws SQLException {
        if (closed) {
            throw new SQLException("Cannot commit, transaction is already closed");
        }
        clearLocalCache();
        if (required) {
            transaction.commit();
        }
    }
    
    @Override
    public void rollback(boolean required) throws SQLException {
        if (!closed) {
            clearLocalCache();
            if (required) {
                transaction.rollback();
            }
        }
    }
    
    @Override
    public void close(boolean forceRollback) {
        try {
            try {
                rollback(forceRollback);
            } finally {
                transaction.close();
            }
        } catch (SQLException e) {
            // Ignore
        } finally {
            closed = true;
        }
    }
    
    @Override
    public boolean isClosed() {
        return closed;
    }
    
    @Override
    public void clearLocalCache() {
        localCache.clear();
    }
    
    @Override
    public CacheKey createCacheKey(MappedStatement ms, Object parameter, RowBounds rowBounds) {
        CacheKey cacheKey = new CacheKey();
        cacheKey.update(ms.getId());
        cacheKey.update(rowBounds.getOffset());
        cacheKey.update(rowBounds.getLimit());
        cacheKey.update(parameter);
        return cacheKey;
    }
    
    @Override
    public void setExecutorWrapper(Executor executor) {
        this.wrapper = executor;
    }
    
    private static class ExecutionPlaceholder {
        public static final ExecutionPlaceholder EXECUTION_PLACEHOLDER = new ExecutionPlaceholder();
    }
}

 * 简单执行器

    public SimpleExecutor(Configuration configuration, Transaction transaction) {
        super(transaction);
    }
    
    @Override
    protected <E> List<E> doQuery(MappedStatement ms, Object parameter, RowBounds rowBounds) throws SQLException {
        Statement stmt = null;
        try {
            Configuration configuration = ms.getConfiguration();
            BoundSql boundSql = ms.getBoundSql(parameter);
            String sql = boundSql.getSql();
            
            // 处理分页
            if (rowBounds != null && rowBounds != RowBounds.DEFAULT) {
                sql = getPagingSql(sql, rowBounds);
            }
            
            Connection connection = getTransaction().getConnection();
            stmt = prepareStatement(connection, sql);
            setParameters(stmt, boundSql, parameter);
            
            return resultSetHandler.<E>handleResultSets(stmt);
        } finally {
            closeStatement(stmt);
        }
    }
    
    @Override
    protected int doUpdate(MappedStatement ms, Object parameter) throws SQLException {
        Statement stmt = null;
        try {
            Configuration configuration = ms.getConfiguration();
            BoundSql boundSql = ms.getBoundSql(parameter);
            
            Connection connection = getTransaction().getConnection();
            stmt = prepareStatement(connection, boundSql.getSql());
            setParameters(stmt, boundSql, parameter);
            
            return stmt.executeUpdate();
        } finally {
            closeStatement(stmt);
        }
    }
    
    private String getPagingSql(String sql, RowBounds rowBounds) {
        StringBuilder sqlBuilder = new StringBuilder(sql);
        sqlBuilder.append(" LIMIT ").append(rowBounds.getOffset()).append(",").append(rowBounds.getLimit());
        return sqlBuilder.toString();
    }
    
    private Statement prepareStatement(Connection connection, String sql) throws SQLException {
        return connection.prepareStatement(sql);
    }
    
    private void setParameters(Statement stmt, BoundSql boundSql, Object parameter) throws SQLException {
        PreparedStatement ps = (PreparedStatement) stmt;
        // 参数设置逻辑
        // 这里简化实现，实际应该根据parameterType和parameterMappings设置参数
        if (parameter != null) {
            if (parameter instanceof Object[]) {
                Object[] params = (Object[]) parameter;
                for (int i = 0; i < params.length; i++) {
                    ps.setObject(i + 1, params[i]);
                }
            } else {
                ps.setObject(1, parameter);
            }
        }
    }
    
    private void closeStatement(Statement stmt) {
        if (stmt != null) {
            try {
                stmt.close();
            } catch (SQLException e) {
                // ignore
            }
        }

package com.simple.mybatis.cache;

import java.util.ArrayList;
import java.util.List;

/**
 * 缓存键
 */
public class CacheKey {
    
    private static final int DEFAULT_MULTIPLIER = 37;
    private static final int DEFAULT_HASHCODE = 17;
    
    private final int multiplier;
    private int hashcode;
    private long checksum;
    private int count;
    private List<Object> updateList;
    
    public CacheKey() {
        this.hashcode = DEFAULT_HASHCODE;
        this.multiplier = DEFAULT_MULTIPLIER;
        this.count = 0;
        this.updateList = new ArrayList<>();
    }
    
    public void update(Object object) {
        int baseHashCode = object == null ? 1 : object.hashCode();
        
        count++;
        checksum += baseHashCode;
        baseHashCode *= count;
        
        hashcode = multiplier * hashcode + baseHashCode;
        
        updateList.add(object);
    }
    
    @Override
    public boolean equals(Object object) {
        if (this == object) {
            return true;
        }
        if (!(object instanceof CacheKey)) {
            return false;
        }
        
        final CacheKey cacheKey = (CacheKey) object;
        
        if (hashcode != cacheKey.hashcode) {
            return false;
        }
        if (checksum != cacheKey.checksum) {
            return false;
        }
        if (count != cacheKey.count) {
            return false;
        }
        
        for (int i = 0; i < updateList.size(); i++) {
            Object thisObject = updateList.get(i);
            Object thatObject = cacheKey.updateList.get(i);
            if (thisObject == null) {
                if (thatObject != null) {
                    return false;
                }
            } else {
                if (!thisObject.equals(thatObject)) {
                    return false;
                }
            }
        }
        return true;
    }
    
    @Override
    public int hashCode() {
        return hashcode;
    }
}

        }

package com.simple.mybatis.builder;

import com.simple.mybatis.annotations.*;
import com.simple.mybatis.config.Configuration;
import com.simple.mybatis.mapping.MappedStatement;
import com.simple.mybatis.mapping.SqlCommandType;
import com.simple.mybatis.mapping.SqlSource;
import com.simple.mybatis.scripting.xmltags.RawSqlSource;

import java.lang.reflect.Method;

/**
 * 注解配置解析器
 */
public class AnnotationMapperBuilder {
    
    private final Configuration configuration;
    private final Class<?> type;
    
    public AnnotationMapperBuilder(Configuration configuration, Class<?> type) {
        this.configuration = configuration;
        this.type = type;
    }
    
    public void parse() {
        String namespace = type.getName();
        
        Method[] methods = type.getDeclaredMethods();
        for (Method method : methods) {
            parseStatement(method, namespace);
        }
    }
    
    private void parseStatement(Method method, String namespace) {
        String statementId = namespace + "." + method.getName();
        
        SqlCommandType sqlCommandType = null;
        String sql = null;
        
        // 解析注解
        if (method.isAnnotationPresent(Select.class)) {
            sqlCommandType = SqlCommandType.SELECT;
            sql = method.getAnnotation(Select.class).value();
        } else if (method.isAnnotationPresent(Insert.class)) {
            sqlCommandType = SqlCommandType.INSERT;
            sql = method.getAnnotation(Insert.class).value();
        } else if (method.isAnnotationPresent(Update.class)) {
            sqlCommandType = SqlCommandType.UPDATE;
            sql = method.getAnnotation(Update.class).value();
        } else if (method.isAnnotationPresent(Delete.class)) {
            sqlCommandType = SqlCommandType.DELETE;
            sql = method.getAnnotation(Delete.class).value();
        }
        
        if (sqlCommandType != null) {
            SqlSource sqlSource = new RawSqlSource(configuration, sql, method.getParameterTypes()[0]);
            
            MappedStatement.Builder statementBuilder = new MappedStatement.Builder(configuration, statementId, sqlSource, sqlCommandType);
            
            // 设置返回类型
            if (sqlCommandType == SqlCommandType.SELECT) {
                statementBuilder.resultType(method.getReturnType());
            }
            
            MappedStatement mappedStatement = statementBuilder.build();
            configuration.addMappedStatement(mappedStatement);
        }
    }
}
