16. **场景**：扇出 - 扇入模式处理多个任务。
    **答案**：

    go

    

    运行

    

    

    

    

    ```go
    package main
    
    import (
        "fmt"
        "sync"
    )
    
    // 处理单个任务
    func process(id int) int {
        return id * 2
    }
    
    // 扇出：多个Goroutine并发处理
    func fanOut(tasks []int) []<-chan int {
        channels := make([]<-chan int, len(tasks))
        for i, t := range tasks {
            ch := make(chan int)
            channels[i] = ch
            go func(task int, c chan<- int) {
                c <- process(task)
                close(c)
            }(t, ch)
        }
        return channels
    }
    
    // 扇入：汇总结果
    func fanIn(channels []<-chan int) <-chan int {
        out := make(chan int)
        var wg sync.WaitGroup
        
        wg.Add(len(channels))
        for _, ch := range channels {
            go func(c <-chan int) {
                defer wg.Done()
                for res := range c {
                    out <- res
                }
            }(ch)
        }
        
        go func() {
            wg.Wait()
            close(out)
        }()
        return out
    }
    
    func main() {
        tasks := []int{1, 2, 3, 4, 5}
        channels := fanOut(tasks)
        results := fanIn(channels)
        
        for res := range results {
            fmt.Print(res, " ") // 输出：2 4 6 8 10（顺序不定）
        }
    }
    ```

    package main
    
    import (
        "fmt"
        "sync"
    )
    
    func worker(id int, jobs <-chan int, results chan<- int) {
        for j := range jobs {
            fmt.Printf("Worker %d 处理任务 %d\n", id, j)
            results <- j * 2
        }
    }

        const numJobs = 5
        const numWorkers = 2
        
        jobs := make(chan int, numJobs)
        results := make(chan int, numJobs)

        for w := 1; w <= numWorkers; w++ {
            go worker(w, jobs, results)
        }
        
        // 发送任务
        for j := 1; j <= numJobs; j++ {
            jobs <- j
        }
        close(jobs)
        
        // 收集结果
        for a := 1; a <= numJobs; a++ {
            <-results
        }
    }
