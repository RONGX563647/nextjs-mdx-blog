### 场景题：并发任务执行与信号控制的下载管理器

#### 场景描述

你需要设计一个文件下载管理器，该管理器需要支持以下功能：

1. **并发下载**：可以同时从多个 URL 下载文件，充分利用网络资源。
2. **同步控制**：确保多个下载任务之间不会因为竞争共享资源（如磁盘写入、下载进度统计）而出现问题。
3. **原子操作**：准确统计所有下载任务的总字节数、成功 / 失败次数等指标，这些统计需要是原子的，避免并发修改导致的错误。
4. **信号处理**：能够通过系统信号（如 `SIGINT`（Ctrl+C）、`SIGTERM`）优雅地停止所有正在进行的下载任务，释放资源，并输出最终的下载统计信息。

#### 具体要求

1. 并发下载（`sync` 相关）

   ：

   - 使用 `sync.WaitGroup` 等待所有下载任务完成。
   - 使用 `sync.Mutex` 保护对共享资源（如下载进度存储的切片、统计信息的临时计算变量等）的访问。

2. 原子操作（`sync/atomic` 相关）

   ：

   - 使用原子操作来统计总下载字节数、成功下载次数、失败下载次数，确保这些统计在多 goroutine 并发修改时的准确性。

3. 信号处理（`os/signal` 相关）

   ：

   - 监听指定的系统信号，当接收到信号时，停止接收新的下载任务，等待正在进行的下载任务完成，然后输出统计信息并退出程序。

#### 示例问题

1. 如何使用 `sync.WaitGroup` 来等待所有并发的下载任务完成？
2. 为什么统计总下载字节数时需要使用原子操作？如果不使用会出现什么问题？
3. 如何结合 `os/signal` 包来实现优雅退出，确保正在进行的下载任务能正常完成？
4. 请编写代码，实现从多个 URL 下载文件，使用同步机制保护共享资源，原子操作统计下载指标，并能通过信号优雅退出。

### 步骤 2：实现单个文件下载功能

**目标**：编写基础下载函数，从 URL 下载文件到本地。

go

```go
// downloadFile 从URL下载文件到指定目录，返回下载字节数和错误
func downloadFile(url string, saveDir string) (int64, error) {
    // 创建保存目录（如果不存在）
    if err := os.MkdirAll(saveDir, 0755); err != nil {
        return 0, fmt.Errorf("创建目录失败: %v", err)
    }

    // 发送HTTP请求
    resp, err := http.Get(url)
    if err != nil {
        return 0, fmt.Errorf("请求URL失败: %v", err)
    }
    defer resp.Body.Close()

    // 检查响应状态
    if resp.StatusCode != http.StatusOK {
        return 0, fmt.Errorf("请求失败，状态码: %d", resp.StatusCode)
    }

    // 提取文件名（从URL中）
    fileName := filepath.Base(url)
    savePath := filepath.Join(saveDir, fileName)

    // 创建本地文件
    outFile, err := os.Create(savePath)
    if err != nil {
        return 0, fmt.Errorf("创建文件失败: %v", err)
    }
    defer outFile.Close()

    // 写入文件并统计字节数
    written, err := io.Copy(outFile, resp.Body)
    if err != nil {
        return written, fmt.Errorf("写入文件失败: %v", err)
    }

    return written, nil
}
```

### 步骤 3：添加并发控制与同步机制

**目标**：使用 `sync.WaitGroup` 实现多任务并发，用 `sync.Mutex` 保护共享资源。

go

```go
// 定义下载任务函数（供goroutine调用）
func downloadTask(url string, saveDir string, wg *sync.WaitGroup) {
    defer wg.Done() // 任务结束时通知WaitGroup

    // 检查是否需要停止（原子操作读取标志）
    if atomic.LoadUint32(&shouldStop) == 1 {
        mu.Lock()
        results = append(results, fmt.Sprintf("任务取消: %s", url))
        mu.Unlock()
        return
    }

    // 执行下载
    fmt.Printf("开始下载: %s\n", url)
    bytes, err := downloadFile(url, saveDir)

    // 更新共享结果（需要互斥锁保护）
    mu.Lock()
    if err != nil {
        results = append(results, fmt.Sprintf("下载失败: %s, 错误: %v", url, err))
        atomic.AddInt64(&stats.failCount, 1) // 原子更新失败次数
    } else {
        results = append(results, fmt.Sprintf("下载成功: %s, 大小: %d字节", url, bytes))
        atomic.AddInt64(&stats.successCount, 1) // 原子更新成功次数
        atomic.AddInt64(&stats.totalBytes, bytes) // 原子更新总字节数
    }
    mu.Unlock()
}
```

### 步骤 4：实现信号处理与优雅退出

**目标**：监听系统信号（如 Ctrl+C），收到信号后停止新任务，等待已有任务完成。

go

```go
// setupSignalHandler 注册信号处理函数，实现优雅退出
func setupSignalHandler(wg *sync.WaitGroup) {
    sigChan := make(chan os.Signal, 1)
    // 监听SIGINT（Ctrl+C）和SIGTERM信号
    signal.Notify(sigChan, syscall.SIGINT, syscall.SIGTERM)

    // 启动goroutine处理信号
    go func() {
        <-sigChan // 阻塞等待信号
        fmt.Println("\n收到退出信号，正在停止新任务...")
        
        // 原子操作设置停止标志（阻止新任务启动）
        atomic.StoreUint32(&shouldStop, 1)
        
        // 等待所有正在执行的任务完成
        fmt.Println("等待正在进行的下载任务完成...")
        wg.Wait()
        
        // 输出最终统计信息
        printStats()
        os.Exit(0)
    }()
}

// printStats 打印下载统计信息
func printStats() {
    fmt.Println("\n===== 下载统计 =====")
    fmt.Printf("总下载字节数: %d\n", stats.totalBytes)
    fmt.Printf("成功下载: %d 个\n", stats.successCount)
    fmt.Printf("下载失败: %d 个\n", stats.failCount)
    fmt.Println("====================")
}
```

```go
func main() {
    // 配置下载参数

    urls := []string{        // 测试用URL列表（可替换为实际文件URL）
```

### 核心技术点总结

1. **并发控制**：`sync.WaitGroup` 跟踪所有下载任务，确保程序等待所有任务完成。
2. **同步保护**：`sync.Mutex` 保护共享的 `results` 切片，避免并发写入冲突。
3. **原子操作**：`sync/atomic` 确保统计数据（字节数、成功 / 失败次数）在多 goroutine 修改时的准确性。
4. **信号处理**：`os/signal` 监听退出信号，结合原子标志实现优雅退出，避免任务中断导致的数据不一致。

```go
package main

import (
	"fmt"
	"io"
	"net/http"
	"os"
	"path/filepath"
	"sync"
	"sync/atomic"
	"syscall"
	"time"

	"os/signal"
)

// DownloadStats 用于记录下载统计信息，使用原子类型确保并发安全
type DownloadStats struct {
	totalBytes   int64 // 总下载字节数
	successCount int64 // 成功下载的文件数量
	failCount    int64 // 下载失败的文件数量
}

// 全局变量定义
var (
	stats        DownloadStats       // 下载统计信息
	results      []string            // 存储下载结果的共享切片
	mu           sync.Mutex          // 保护results切片的互斥锁
	shouldStop   uint32              // 控制是否停止接收新任务的原子标志(0:运行中, 1:停止)
	maxConcurrent int32 = 3          // 最大并发下载数
	semaphore    = make(chan struct{}, maxConcurrent) // 控制并发数量的信号量
)

// downloadFile 从指定URL下载文件到目标目录，返回下载的字节数和可能的错误
func downloadFile(url string, saveDir string) (int64, error) {
	// 创建保存目录（如果不存在）
	if err := os.MkdirAll(saveDir, 0755); err != nil {
		return 0, fmt.Errorf("创建目录失败: %v", err)
	}

	// 发送HTTP GET请求
	resp, err := http.Get(url)
	if err != nil {
		return 0, fmt.Errorf("请求URL失败: %v", err)
	}
	defer resp.Body.Close()

	// 检查HTTP响应状态码
	if resp.StatusCode != http.StatusOK {
		return 0, fmt.Errorf("请求失败，状态码: %d", resp.StatusCode)
	}

	// 从URL中提取文件名
	fileName := filepath.Base(url)
	savePath := filepath.Join(saveDir, fileName)

	// 创建本地文件
	outFile, err := os.Create(savePath)
	if err != nil {
		return 0, fmt.Errorf("创建文件失败: %v", err)
	}
	defer outFile.Close()

	// 将响应内容写入本地文件，并返回写入的字节数
	written, err := io.Copy(outFile, resp.Body)
	if err != nil {
		// 如果写入失败，尝试删除不完整的文件
		os.Remove(savePath)
		return written, fmt.Errorf("写入文件失败: %v", err)
	}

	return written, nil
}

// downloadTask 下载任务函数，供goroutine调用
func downloadTask(url string, saveDir string, wg *sync.WaitGroup) {
	// 释放信号量并通知WaitGroup任务完成
	defer func() {
		<-semaphore
		wg.Done()
	}()

	// 检查是否需要停止新任务
	if atomic.LoadUint32(&shouldStop) == 1 {
		mu.Lock()
		results = append(results, fmt.Sprintf("任务取消: %s", url))
		mu.Unlock()
		return
	}

	// 执行下载
	fmt.Printf("开始下载: %s\n", url)
	startTime := time.Now()
	bytes, err := downloadFile(url, saveDir)
	duration := time.Since(startTime)

	// 更新共享结果和统计信息
	mu.Lock()
	if err != nil {
		results = append(results, fmt.Sprintf("下载失败: %s (耗时: %v)，错误: %v", url, duration, err))
		atomic.AddInt64(&stats.failCount, 1)
	} else {
		results = append(results, fmt.Sprintf("下载成功: %s (大小: %d字节, 耗时: %v)", url, bytes, duration))
		atomic.AddInt64(&stats.successCount, 1)
		atomic.AddInt64(&stats.totalBytes, bytes)
	}
	mu.Unlock()
}

// setupSignalHandler 注册信号处理函数，实现优雅退出
func setupSignalHandler(wg *sync.WaitGroup) {
	// 创建信号通道
	sigChan := make(chan os.Signal, 1)
	// 监听SIGINT(Ctrl+C)和SIGTERM信号
	signal.Notify(sigChan, syscall.SIGINT, syscall.SIGTERM)

	// 启动goroutine处理信号
	go func() {
		// 等待信号
		sig := <-sigChan
		fmt.Printf("\n收到信号: %v，正在准备优雅退出...\n", sig)

		// 设置停止标志，阻止新任务启动
		atomic.StoreUint32(&shouldStop, 1)

		// 等待所有正在执行的任务完成
		fmt.Println("等待正在进行的下载任务完成...")
		wg.Wait()

		// 输出最终统计信息
		printResults()
		printStats()
		os.Exit(0)
	}()
}

// printStats 打印下载统计信息
func printStats() {
	fmt.Println("\n===== 下载统计 =====")
	fmt.Printf("总下载字节数: %d bytes (%.2f MB)\n", 
		stats.totalBytes, float64(stats.totalBytes)/1024/1024)
	fmt.Printf("成功下载: %d 个文件\n", stats.successCount)
	fmt.Printf("下载失败: %d 个文件\n", stats.failCount)
	fmt.Println("====================")
}

// printResults 打印所有下载结果
func printResults() {
	fmt.Println("\n===== 下载结果 =====")
	for _, res := range results {
		fmt.Println(res)
	}
}

func main() {
	// 配置下载参数
	saveDir := "./downloads" // 文件保存目录
	// 测试用URL列表(可替换为实际文件URL)
	urls := []string{
		"https://example.com",
		"https://httpbin.org/get",
		"https://httpbin.org/image/jpeg",
		"https://httpbin.org/image/png",
		"https://httpbin.org/xml",
		"https://httpbin.org/json",
		"https://httpbin.org/bytes/102400", // 100KB测试文件
		"https://httpbin.org/status/404",   // 测试失败情况
	}

	// 初始化WaitGroup
	var wg sync.WaitGroup

	// 注册信号处理
	setupSignalHandler(&wg)

	// 显示启动信息
	fmt.Printf("下载管理器启动，最大并发数: %d\n", maxConcurrent)
	fmt.Printf("文件将保存到: %s\n", saveDir)
	fmt.Printf("准备下载 %d 个文件...\n", len(urls))

	// 启动并发下载任务
	for _, url := range urls {
		// 检查是否需要停止
		if atomic.LoadUint32(&shouldStop) == 1 {
			fmt.Println("已收到停止信号，不再启动新任务")
			break
		}

		// 获取信号量，控制并发数量
		semaphore <- struct{}{}
		wg.Add(1)
		go downloadTask(url, saveDir, &wg)
	}

	// 等待所有任务完成
	wg.Wait()

	// 检查是否是正常完成(非信号中断)
	if atomic.LoadUint32(&shouldStop) == 0 {
		printResults()
		printStats()
		fmt.Println("所有下载任务已完成")
	}
}
```
