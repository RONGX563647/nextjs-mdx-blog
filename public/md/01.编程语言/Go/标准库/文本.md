```go
package main

import (
	"fmt"
	"regexp"
	"sort"
	"strconv"
	"strings"
	"unicode"
	"unicode/utf8"
)

func main() {
	// 示例文本：包含单词、数字、邮箱、URL、中英文 Unicode 字符
	sampleText := `Hello 世界！This is a test text. There are 123 apples and 45.6 bananas. 
	My email is user@example.com, and the URL is https://example.com/page. 你好，Golang！`

	fmt.Println("===== 文本处理与分析工具 =====")

	// ---------- 1. strings 包：字符串操作（单词频率统计） ----------
	fmt.Println("\n--- 1. 单词频率统计（strings 包） ---")
	wordFreq := countWordFrequency(sampleText)
	// 排序并打印高频单词
	var sortedWords []string
	for word := range wordFreq {
		sortedWords = append(sortedWords, word)
	}
	sort.Strings(sortedWords)
	for i, word := range sortedWords {
		if i >= 5 { // 只打印前 5 个
			break
		}
		fmt.Printf("单词 '%s' 出现 %d 次\n", word, wordFreq[word])
	}

	// ---------- 2. bytes 包：byte slice 便捷操作 ----------
	fmt.Println("\n--- 2. 字节切片替换（bytes 包） ---")
	textBytes := []byte(sampleText)
	// 将 "test" 替换为 "SAMPLE"（字节级操作）
	replacedBytes := bytes.Replace(textBytes, []byte("test"), []byte("SAMPLE"), -1)
	fmt.Println("替换后内容：\n", string(replacedBytes))

	// ---------- 3. strconv 包：字符串与基本类型转换 ----------
	fmt.Println("\n--- 3. 字符串与数值转换（strconv 包） ---")
	// 提取所有数字字符串
	numPattern := regexp.MustCompile(`\d+(\.\d+)?`)
	numStrs := numPattern.FindAllString(sampleText, -1)
	for _, s := range numStrs {
		// 尝试转 int
		if intVal, err := strconv.Atoi(s); err == nil {
			fmt.Printf("整数转换：%s -> %d\n", s, intVal)
		} else if floatVal, err := strconv.ParseFloat(s, 64); err == nil { // 转 float64
			fmt.Printf("浮点数转换：%s -> %f\n", s, floatVal)
		}
	}

	// ---------- 4. regexp 包：正则表达式提取内容 ----------
	fmt.Println("\n--- 4. 正则提取（regexp 包） ---")
	// 提取邮箱
	emailRegex := regexp.MustCompile(`\b[\w.-]+@[\w.-]+\.\w+\b`)
	emails := emailRegex.FindAllString(sampleText, -1)
	fmt.Println("提取的邮箱：", emails)
	// 提取 URL
	urlRegex := regexp.MustCompile(`https?://[^\s]+`)
	urls := urlRegex.FindAllString(sampleText, -1)
	fmt.Println("提取的 URL：", urls)

	// ---------- 5. unicode 包：Unicode 与编码分析 ----------
	fmt.Println("\n--- 5. Unicode 与编码分析（unicode 包） ---")
	chineseCharCount := 0
	totalRuneCount := 0
	for _, r := range sampleText {
		totalRuneCount++
		// 判断是否为中文字符（Unicode 范围：\u4e00-\u9fff）
		if unicode.Is(unicode.Han, r) {
			chineseCharCount++
		}
	}
	fmt.Printf("中文字符数量：%d\n", chineseCharCount)
	// 统计 UTF-8 编码的字符数（与字节长度对比）
	utf8CharCount := utf8.RuneCountInString(sampleText)
	fmt.Printf("UTF-8 字符总数（Rune 数）：%d\n", utf8CharCount)
	fmt.Printf("字符串字节长度：%d\n", len(sampleText))
}

// countWordFrequency 统计文本中单词的频率
func countWordFrequency(text string) map[string]int {
	freq := make(map[string]int)
	// 转小写后，按“非字母数字”分割单词
	words := strings.FieldsFunc(strings.ToLower(text), func(r rune) bool {
		return !unicode.IsLetter(r) && !unicode.IsNumber(r)
	})
	for _, word := range words {
		if word != "" {
			freq[word]++
		}
	}
	return freq
}
```

### 各包用法详解：

1. **`strings` 包**：
   - `strings.ToLower`：将字符串转小写，统一统计逻辑。
   - `strings.FieldsFunc`：自定义分割规则（按 “非字母数字” 字符分割单词）。
   - （扩展）还可使用 `strings.Split`、`strings.Contains`、`strings.Replace` 等函数处理字符串。
2. **`bytes` 包**：
   - `bytes.Replace`：对字节切片 `[]byte` 进行替换操作，类似 `strings.Replace` 但针对字节级数据，适合大文本或二进制内容处理。
3. **`strconv` 包**：
   - `strconv.Atoi`：将字符串转为 `int`（处理整数）。
   - `strconv.ParseFloat`：将字符串转为 `float64`（处理浮点数）。
   - （扩展）`strconv.Itoa`（int 转字符串）、`strconv.FormatFloat`（float 转字符串）等可用于类型转换。
4. **`regexp` 包**：
   - `regexp.MustCompile`：编译正则表达式（若正则语法错误会 panic，适合确定的正则）。
   - `FindAllString`：提取所有匹配正则的子串，适合批量提取（如邮箱、URL、日期等）。
5. **`unicode` 包**：
   - `unicode.Is(unicode.Han, r)`：判断字符 `r` 是否为中文字符（`unicode.Han` 覆盖 Unicode 中 “汉字” 范围）。
   - `unicode/utf8` 子包：`utf8.RuneCountInString` 统计字符串的 Unicode 码点数量（体现 UTF-8 编码下 “字符数” 与 “字节数” 的区别）。

### 场景说明与核心包函数解析

这个**多模式文本内容搜索引擎**可以从指定目录中快速查找包含特定关键词或匹配正则表达式的文本内容，支持多种过滤条件和并发搜索，适用于代码审计、日志分析、文档检索等场景。

#### 核心功能与包函数应用

1. **命令行参数解析（`flag`包）**
   - `flag.String()`/`flag.Bool()`/`flag.Int()`：定义不同类型的命令行参数
   - `flag.Parse()`：解析命令行参数并赋值
   - 作用：实现灵活的搜索配置，支持目录、关键词、正则等多参数组合
2. **文件系统操作（`os`/`filepath`包）**
   - `os.Stat()`：检查目录是否存在
   - `os.Open()`：打开文件准备读取
   - `filepath.Walk()`：递归遍历目录树，收集所有文件
   - `filepath.Ext()`：获取文件扩展名
   - 作用：实现文件系统遍历和过滤，精准定位需要搜索的文件
3. **并发控制（`sync`包）**
   - `sync.WaitGroup`：等待所有工作线程完成搜索
   - `sync.Mutex`：保护共享结果切片的并发写入安全
   - 作用：通过多线程并行处理文件，大幅提高搜索效率
4. **高效文件读取（`bufio`包）**
   - `bufio.Scanner`：创建文件扫描器，支持逐行读取
   - `scanner.Scan()`：读取下一行内容
   - `scanner.Text()`：获取当前行的字符串
   - 作用：高效处理大文件，避免一次性加载整个文件到内存
5. **字符串处理（`strings`包）**
   - `strings.Split()`：分割逗号分隔的扩展名列表
   - `strings.Contains()`：检查行内容是否包含关键词
   - `strings.ToLower()`：实现不区分大小写的匹配
   - `strings.ReplaceAll()`：替换文本实现高亮显示
   - 作用：提供基础的字符串匹配和处理能力
6. **正则匹配（`regexp`包）**
   - `regexp.CompilePOSIX()`：编译 POSIX 兼容的正则表达式
   - `regex.MatchString()`：检查字符串是否匹配正则模式
   - 作用：支持复杂模式的文本匹配，比简单关键词更灵活
7. **结果排序（`sort`包）**
   - `sort.Slice()`：对搜索结果按文件路径和行号排序
   - 作用：使搜索结果更有条理，便于查看
