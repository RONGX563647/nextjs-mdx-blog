### 代码功能与包的常用函数解析

这个文件管理器实现了 8 个常用功能，涵盖了各个包的核心函数：

1. **os 包**：
   - `os.Create()`：创建新文件
   - `os.Open()`：打开文件用于读取
   - `os.OpenFile()`：更灵活地打开文件（支持追加等模式）
   - `os.Stat()`：获取文件信息
   - `os.Remove()`：删除文件
   - `file.Close()`：关闭文件
   - `file.Seek()`：移动文件指针
2. **fmt 包**：
   - `fmt.Println()`：打印到控制台并换行
   - `fmt.Printf()`：格式化输出到控制台
   - `fmt.Fprintln()`：格式化输出到文件
3. **io 包**：
   - `io.Copy()`：复制数据（从一个 Reader 到一个 Writer）
   - `io.EOF`：表示文件结束的错误
4. **ioutil 包**：
   - `ioutil.ReadFile()`：一次性读取整个文件内容
5. **bufio 包**：
   - `bufio.NewScanner()`：逐行扫描文件内容
   - `bufio.NewReader()`：创建带缓冲的读取器
   - `bufio.NewWriter()`：创建带缓冲的写入器
   - `writer.Flush()`：刷新缓冲区，确保数据写入

6. **strings 包**：
   - `strings.Contains()`：检查字符串是否包含子串
   - `strings.Fields()`：分割字符串为单词
   - `strings.TrimSpace()`：去除字符串首尾空白

```go
package main

import (
	"bufio"
	"fmt"
	"io"
	"os"
	"strings"
)

func main() {
	// 程序标题：文本文件管理器
	fmt.Println("===== 文本文件管理器 =====")

	// ===================== 1. 创建并写入示例文件 =====================
	fmt.Println("\n1. 创建并写入示例文件...")
	// os.Create：创建新文件，返回文件对象和错误（若失败则处理）
	file, err := os.Create("example.txt")
	if err != nil {
		fmt.Printf("创建文件失败: %v\n", err)
		return // 错误发生时，直接退出程序（或根据需求做更灵活的错误处理）
	}
	defer file.Close() // 延迟关闭文件：确保函数结束前，文件会被正确关闭

	// 定义要写入的内容
	content := "Go语言是一门简洁高效的编程语言\n" +
		"它非常适合系统编程和网络编程\n" +
		"标准库提供了丰富的功能"
	// fmt.Fprintln：向文件写入内容（带换行），返回写入字节数和错误
	_, err = fmt.Fprintln(file, content)
	if err != nil {
		fmt.Printf("写入文件失败: %v\n", err)
		return
	}
	fmt.Println("→ 已创建 example.txt 并写入初始内容")

	// ===================== 2. 读取整个文件内容（替代弃用的 ioutil.ReadFile） =====================
	fmt.Println("\n2. 读取整个文件内容...")
	// os.ReadFile：一次性读取整个文件到字节切片（Go 1.16+ 推荐，替代 ioutil.ReadFile）
	fileContent, err := os.ReadFile("example.txt")
	if err != nil {
		fmt.Printf("读取文件失败: %v\n", err)
		return
	}
	fmt.Println("→ 文件内容:")
	fmt.Println(string(fileContent)) // 字节切片转字符串后打印

	// ===================== 3. 逐行扫描文件内容（带缓冲，适合大文件） =====================
	fmt.Println("\n3. 逐行扫描并处理文件内容...")
	// os.Open：打开文件用于读取（若失败则处理）
	file, err = os.Open("example.txt")
	if err != nil {
		fmt.Printf("打开文件失败: %v\n", err)
		return
	}
	defer file.Close() // 延迟关闭当前文件

	scanner := bufio.NewScanner(file) // 创建“带缓冲的行扫描器”
	lineNumber := 1
	for scanner.Scan() { // 循环扫描每一行
		line := scanner.Text() // 获取当前行的字符串
		wordCount := len(strings.Fields(line)) // 用 strings.Fields 分割单词并计数
		fmt.Printf("→ 行 %d: %s (单词数: %d)\n", lineNumber, line, wordCount)
		lineNumber++
	}

	// 检查扫描过程中是否出现错误（如 IO 错误）
	if err := scanner.Err(); err != nil {
		fmt.Printf("扫描文件时出错: %v\n", err)
		return
	}

	// ===================== 4. 按关键词搜索文件内容（带缓冲读取） =====================
	fmt.Println("\n4. 按关键词搜索文件内容...")
	searchWord := "编程"
	// 重新打开文件（因为之前的 file 已被关闭，需重新打开）
	file, err = os.Open("example.txt")
	if err != nil {
		fmt.Printf("打开文件失败: %v\n", err)
		return
	}
	defer file.Close()

	_, err = file.Seek(0, 0) // 将文件指针移回开头（确保从文件起始处读取）
	if err != nil {
		fmt.Printf("移动文件指针失败: %v\n", err)
		return
	}

	reader := bufio.NewReader(file) // 创建“带缓冲的读取器”
	foundLines := 0

	for {
		// ReadString('\n')：读取到换行符为止的一行内容，返回字符串和错误
		line, err := reader.ReadString('\n')
		if err != nil {
			if err == io.EOF { // 若为“文件结束”错误
				// 检查最后一行（可能无换行符）
				if strings.Contains(line, searchWord) && line != "" {
					foundLines++
					fmt.Printf("→ 找到匹配行: %s\n", strings.TrimSpace(line))
				}
				break // 结束循环
			}
			// 其他读取错误（如 IO 错误）
			fmt.Printf("读取文件时出错: %v\n", err)
			return
		}

		// 检查行中是否包含关键词
		if strings.Contains(line, searchWord) {
			foundLines++
			fmt.Printf("→ 找到匹配行: %s\n", strings.TrimSpace(line))
		}
	}
	fmt.Printf("→ 共找到 %d 处包含 '%s' 的行\n", foundLines, searchWord)

	// ===================== 5. 复制文件（使用 io.Copy 高效复制） =====================
	fmt.Println("\n5. 复制文件...")
	srcFile, err := os.Open("example.txt") // 打开源文件
	if err != nil {
		fmt.Printf("打开源文件失败: %v\n", err)
		return
	}
	defer srcFile.Close()

	dstFile, err := os.Create("example_backup.txt") // 创建目标文件
	if err != nil {
		fmt.Printf("创建目标文件失败: %v\n", err)
		return
	}
	defer dstFile.Close()

	// io.Copy：从源 Reader 复制到目标 Writer，返回复制的字节数和错误
	bytesWritten, err := io.Copy(dstFile, srcFile)
	if err != nil {
		fmt.Printf("复制文件失败: %v\n", err)
		return
	}
	fmt.Printf("→ 已复制 %d 字节到 example_backup.txt\n", bytesWritten)

	// ===================== 6. 追加内容到文件（使用 os.OpenFile 指定追加模式） =====================
	fmt.Println("\n6. 追加内容到文件...")
	// os.OpenFile：以“追加+只写”模式打开文件，权限 0644（owner 读写，其他只读）
	appendFile, err := os.OpenFile("example.txt", os.O_APPEND|os.O_WRONLY, 0644)
	if err != nil {
		fmt.Printf("打开文件失败: %v\n", err)
		return
	}
	defer appendFile.Close()

	writer := bufio.NewWriter(appendFile) // 创建“带缓冲的写入器”（提升写入效率）
	_, err = writer.WriteString("\n这是追加的新内容\n") // 写入追加内容
	if err != nil {
		fmt.Printf("写入内容失败: %v\n", err)
		return
	}

	// 刷新缓冲区：确保所有数据真正写入文件（缓冲写入需显式 Flush）
	err = writer.Flush()
	if err != nil {
		fmt.Printf("刷新缓冲区失败: %v\n", err)
		return
	}
	fmt.Println("→ 已追加内容到 example.txt")

	// ===================== 7. 获取文件元信息（使用 os.Stat） =====================
	fmt.Println("\n7. 获取文件元信息...")
	fileInfo, err := os.Stat("example.txt") // 获取文件状态信息
	if err != nil {
		fmt.Printf("获取文件信息失败: %v\n", err)
		return
	}
	fmt.Printf("→ 文件名: %s\n", fileInfo.Name())
	fmt.Printf("→ 文件大小: %d 字节\n", fileInfo.Size())
	fmt.Printf("→ 是否为文件: %t\n", !fileInfo.IsDir())
	fmt.Printf("→ 修改时间: %v\n", fileInfo.ModTime())

	// ===================== 8. 清理临时文件（使用 os.Remove） =====================
	fmt.Println("\n8. 清理临时文件...")
	err = os.Remove("example_backup.txt") // 删除备份文件
	if err != nil {
		fmt.Printf("删除文件失败: %v\n", err)
		return
	}
	fmt.Println("→ 已删除 example_backup.txt")

	fmt.Println("\n===== 所有操作完成 =====")
}
```

```go
// LogEntry 表示一条日志的结构化数据

type LogEntry struct {
	Timestamp time.Time              // 日志记录的时间戳

// AnalysisResult 存储日志分析的结果数据

type AnalysisResult struct {

	workers := flag.Int("workers", 4, "并发工作线程数")
	verbose := flag.Bool("verbose", false, "显示详细信息")
	
	// 解析命令行参数
	flag.Parse()
	
	// 将解析结果封装到Config结构体并返回
	return Config{
		InputDir:      *inputDir,
		OutputFile:    *outputFile,
		LogFormat:     *logFormat,
		IncludeLevels: splitCommaList(*includeLevels),
		ExcludeLevels: splitCommaList(*excludeLevels),
		TimeStart:     *timeStart,
		TimeEnd:       *timeEnd,
		Workers:       *workers,
		Verbose:       *verbose,
	}
}

// splitCommaList 将逗号分隔的字符串转换为字符串切片
// 例如将"ERROR,WARN"转换为["ERROR", "WARN"]
func splitCommaList(s string) []string {
	if s == "" {
		return []string{} // 如果输入为空，返回空切片
	}
	parts := strings.Split(s, ",") // 使用逗号分割字符串
	// 去除每个元素的前后空格
	for i, part := range parts {
		parts[i] = strings.TrimSpace(part)
	}
	return parts
}

// validateConfig 验证配置的有效性
// 检查输入目录是否存在、日志格式是否支持、时间格式是否正确等
func validateConfig(config Config) error {
	// 检查输入目录是否存在
	if _, err := os.Stat(config.InputDir); os.IsNotExist(err) {
		return fmt.Errorf("输入目录不存在: %s", config.InputDir)
	}
	
	// 验证日志格式是否支持
	validFormats := map[string]bool{
		"common": true,
		"json":   true,
		"apache": true,
	}
	if !validFormats[config.LogFormat] {
		return fmt.Errorf("无效的日志格式: %s，支持的格式: common, json, apache", config.LogFormat)
	}
	
	// 验证开始时间格式是否正确
	if config.TimeStart != "" {
		if _, err := parseTime(config.TimeStart); err != nil {
			return fmt.Errorf("无效的开始时间格式: %s，应为 YYYY-MM-DD HH:MM:SS", config.TimeStart)
		}
	}
	
	// 验证结束时间格式是否正确
	if config.TimeEnd != "" {
		if _, err := parseTime(config.TimeEnd); err != nil {
			return fmt.Errorf("无效的结束时间格式: %s，应为 YYYY-MM-DD HH:MM:SS", config.TimeEnd)
		}
	}
	
	// 验证工作线程数是否在合理范围内
	if config.Workers <= 0 || config.Workers > 32 {
		return fmt.Errorf("工作线程数必须在 1-32 之间")
	}
	
	return nil // 配置有效，返回nil
}

// findLogFiles 查找指定目录下的所有日志文件
// 使用filepath.Walk遍历目录及其子目录，收集所有符合条件的日志文件
func findLogFiles(dir string) ([]string, error) {
	var logFiles []string // 存储找到的日志文件路径
	
	// 使用filepath.Walk遍历目录
	// Walk函数会递归访问目录下的所有文件和子目录
	err := filepath.Walk(dir, func(path string, info os.FileInfo, err error) error {
		if err != nil {
			return err // 如果访问文件出错，返回错误
		}
		
		// 只处理文件，不处理目录
		if !info.IsDir() {
			// 检查文件扩展名是否为日志相关
			ext := filepath.Ext(path)
			// 日志文件通常以.log或.txt为扩展名，或文件名包含"log"
			if ext == ".log" || ext == ".txt" || strings.Contains(strings.ToLower(info.Name()), "log") {
				logFiles = append(logFiles, path)
			}
		}
		return nil
	})
	
	return logFiles, err
}

// processFiles 并发处理所有日志文件
// 创建一个工作池，使用指定数量的工作线程并发处理文件，提高处理效率
func processFiles(files []string, config Config) error {
	// 创建带缓冲的通道，用于传递文件名给工作线程
	// 缓冲区大小设为文件数量，避免阻塞
	fileChan := make(chan string, len(files))
	
	// 将所有文件添加到通道中
	for _, file := range files {
		fileChan <- file
	}
	close(fileChan) // 关闭通道，通知工作线程没有更多文件
	
	// 创建等待组，用于等待所有工作线程完成
	var wg sync.WaitGroup
	
	// 创建指定数量的工作线程
	for i := 0; i < config.Workers; i++ {
		wg.Add(1) // 增加等待组计数
		// 启动工作线程，传入工作线程ID用于标识
		go func(workerID int) {
			defer wg.Done() // 工作线程结束时，减少等待组计数
			processFilesWorker(fileChan, config, workerID)
		}(i)
	}
	
	wg.Wait() // 等待所有工作线程完成
	return nil
}

// processFilesWorker 工作线程函数，处理通道中的文件
// 从通道中获取文件名，逐个处理，直到通道关闭
func processFilesWorker(fileChan <-chan string, config Config, workerID int) {
	// 从通道中循环获取文件名
	for file := range fileChan {
		if config.Verbose {
			fmt.Printf("工作线程 %d 开始处理文件: %s\n", workerID, file)
		}
		
		// 处理单个文件
		if err := processFile(file, config); err != nil {
			fmt.Printf("处理文件 %s 时出错: %v\n", file, err)
		} else if config.Verbose {
			fmt.Printf("工作线程 %d 完成处理文件: %s\n", workerID, file)
		}
	}
}

// processFile 处理单个日志文件
// 打开文件，逐行读取并解析日志，过滤后进行分析
func processFile(filePath string, config Config) error {
	// 打开文件，返回文件对象
	file, err := os.Open(filePath)
	if err != nil {
		return fmt.Errorf("无法打开文件: %v", err)
	}
	defer file.Close() // 延迟关闭文件，确保资源释放
	
	// 创建扫描器，用于逐行读取文件
	// bufio.Scanner提供了高效的逐行读取功能
	scanner := bufio.NewScanner(file)
	lineNumber := 0 // 记录当前行号，用于错误提示
	
	// 逐行读取文件内容
	for scanner.Scan() {
		lineNumber++
		line := scanner.Text() // 获取当前行的文本内容
		
		// 根据配置的日志格式解析日志行
		entry, err := parseLogLine(line, config.LogFormat, filePath)
		if err != nil {
			if config.Verbose {
				fmt.Printf("文件 %s 第 %d 行解析错误: %v\n", filePath, lineNumber, err)
			}
			continue // 解析错误时，跳过当前行
		}
		
		// 根据配置过滤日志条目
		if !filterLogEntry(entry, config) {
			continue // 不符合过滤条件的日志，跳过
		}
		
		// 处理日志条目，更新统计信息
		processLogEntry(entry)
	}
	
	// 检查扫描过程中是否发生错误
	if err := scanner.Err(); err != nil {
		return fmt.Errorf("扫描文件时出错: %v", err)
	}
	
	return nil
}

// parseLogLine 根据指定格式解析日志行
// 支持三种日志格式：common（通用格式）、json（JSON格式）、apache（Apache日志格式）
func parseLogLine(line string, format string, source string) (LogEntry, error) {
	// 根据日志格式调用相应的解析函数
	switch format {
	case "common":
		return parseCommonLogFormat(line, source)
	case "json":
		return parseJsonLogFormat(line, source)
	case "apache":
		return parseApacheLogFormat(line, source)
	default:
		return LogEntry{}, fmt.Errorf("不支持的日志格式: %s", format)
	}
}

// parseCommonLogFormat 解析通用日志格式
// 通用日志格式示例: 2023-05-15 14:30:45 [INFO] User login successful
func parseCommonLogFormat(line string, source string) (LogEntry, error) {
	// 正则表达式匹配通用日志格式
	// 分组分别匹配：时间戳、日志级别、日志消息
	re := regexp.MustCompile(`^(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) \[([A-Z]+)\] (.*)$`)
	matches := re.FindStringSubmatch(line)
	
	// 检查是否匹配成功
	if len(matches) != 4 {
		return LogEntry{}, fmt.Errorf("不符合通用日志格式")
	}
	
	// 解析时间戳
	timestamp, err := parseTime(matches[1])
	if err != nil {
		return LogEntry{}, fmt.Errorf("时间格式错误: %v", err)
	}
	
	// 初始化详细信息映射
	details := make(map[string]interface{})
	
	// 尝试从消息中提取更多信息
	message := matches[3]
	// 提取响应时间（如果有）
	if strings.Contains(message, "response time:") {
		reRT := regexp.MustCompile(`response time: (\d+)ms`)
		rtMatches := reRT.FindStringSubmatch(message)
		if len(rtMatches) == 2 {
			rt, _ := strconv.Atoi(rtMatches[1])
			details["response_time"] = rt
		}
	}
	
	// 提取请求路径（如果有）
	if strings.Contains(message, "request:") {
		reReq := regexp.MustCompile(`request: (\S+)`)
		reqMatches := reReq.FindStringSubmatch(message)
		if len(reqMatches) == 2 {
			details["request_path"] = reqMatches[1]
		}
	}
	
	// 返回解析后的日志条目
	return LogEntry{
		Timestamp: timestamp,
		Level:     matches[2],
		Message:   message,
		Source:    source,
		Details:   details,
	}, nil
}

// parseJsonLogFormat 解析JSON格式的日志
// JSON日志格式示例: {"time":"2023-05-15T14:30:45Z","level":"INFO","message":"User login successful"}
func parseJsonLogFormat(line string, source string) (LogEntry, error) {
	// 定义一个临时映射存储解析后的JSON数据
	var jsonData map[string]interface{}
	// 使用json.Unmarshal解析JSON字符串
	if err := json.Unmarshal([]byte(line), &jsonData); err != nil {
		return LogEntry{}, fmt.Errorf("JSON解析错误: %v", err)
	}
	
	// 提取必要字段：time、level、message
	timeStr, ok := jsonData["time"].(string)
	if !ok {
		return LogEntry{}, fmt.Errorf("缺少time字段或格式错误")
	}
	
	level, ok := jsonData["level"].(string)
	if !ok {
		return LogEntry{}, fmt.Errorf("缺少level字段或格式错误")
	}
	
	message, ok := jsonData["message"].(string)
	if !ok {
		return LogEntry{}, fmt.Errorf("缺少message字段或格式错误")
	}
	
	// 解析时间戳，尝试多种格式
	timestamp, err := time.Parse(time.RFC3339, timeStr)
	if err != nil {
		// 尝试其他时间格式
		timestamp, err = parseTime(timeStr)
		if err != nil {
			return LogEntry{}, fmt.Errorf("时间格式错误: %v", err)
		}
	}
	
	// 提取详细信息（排除已处理的字段）
	details := make(map[string]interface{})
	for key, value := range jsonData {
		if key != "time" && key != "level" && key != "message" {
			details[key] = value
		}
	}
	
	// 返回解析后的日志条目
	return LogEntry{
		Timestamp: timestamp,
		Level:     level,
		Message:   message,
		Source:    source,
		Details:   details,
	}, nil
}

// parseApacheLogFormat 解析Apache日志格式
// Apache日志格式示例: 127.0.0.1 - - [15/May/2023:14:30:45 +0800] "GET /index.html HTTP/1.1" 200 1234
func parseApacheLogFormat(line string, source string) (LogEntry, error) {
	// 正则表达式匹配Apache日志格式
	// 分组分别匹配：IP地址、时间戳、请求方法、请求路径、状态码、响应大小等
	re := regexp.MustCompile(`^(\S+) \S+ \S+ \[([^\]]+)\] "(\S+) (\S+) \S+" (\d+) (\d+)( "([^"]+)" "([^"]+)")?$`)
	matches := re.FindStringSubmatch(line)
	
	// 检查是否匹配成功
	if len(matches) < 7 {
		return LogEntry{}, fmt.Errorf("不符合Apache日志格式")
	}
	
	// 解析时间戳
	timeStr := matches[2]
	timestamp, err := time.Parse("02/Jan/2006:15:04:05 -0700", timeStr)
	if err != nil {
		return LogEntry{}, fmt.Errorf("时间格式错误: %v", err)
	}
	
	// 构建日志消息
	method := matches[3]
	path := matches[4]
	statusCode := matches[5]
	size := matches[6]
	
	message := fmt.Sprintf("%s %s - 状态码: %s, 大小: %s bytes", 
		method, path, statusCode, size)
	
	// 提取详细信息
	details := make(map[string]interface{})
	details["ip"] = matches[1]
	details["method"] = method
	details["path"] = path
	details["status_code"] = statusCode
	
	// 解析响应大小
	if sizeInt, err := strconv.Atoi(size); err == nil {
		details["size"] = sizeInt
	}
	
	// 解析状态码级别（2xx, 3xx, 4xx, 5xx）
	statusLevel := statusCode[:1]
	switch statusLevel {
	case "2":
		details["status_type"] = "成功"
	case "3":
		details["status_type"] = "重定向"
	case "4":
		details["status_type"] = "客户端错误"
	case "5":
		details["status_type"] = "服务器错误"
	default:
		details["status_type"] = "未知"
	}
	
	// 根据状态码确定日志级别
	level := LevelInfo
	if statusLevel == "4" || statusLevel == "5" {
		level = LevelError
	}
	
	// 返回解析后的日志条目
	return LogEntry{
		Timestamp: timestamp,
		Level:     level,
		Message:   message,
		Source:    source,
		Details:   details,
	}, nil
}

// parseTime 解析时间字符串，尝试多种常见格式
// 提高时间解析的兼容性，支持多种时间表示方式
func parseTime(timeStr string) (time.Time, error) {
	// 尝试多种常见时间格式
	formats := []string{
		"2006-01-02 15:04:05",     // 如：2023-05-15 14:30:45
		"2006-01-02T15:04:05",    // 如：2023-05-15T14:30:45
		"2006-01-02 15:04:05.999", // 带毫秒的格式
		"2006-01-02T15:04:05.999",// 带毫秒和T分隔符的格式
		time.RFC3339,              // RFC3339标准格式
	}
	
	// 逐个尝试格式解析
	for _, format := range formats {
		t, err := time.Parse(format, timeStr)
		if err == nil {
			return t, nil // 解析成功，返回时间
		}
	}
	
	// 所有格式都尝试失败
	return time.Time{}, fmt.Errorf("无法解析时间: %s", timeStr)
}

// filterLogEntry 根据配置过滤日志条目
// 检查日志级别是否符合包含/排除规则，时间是否在指定范围内
func filterLogEntry(entry LogEntry, config Config) bool {
	// 级别过滤：如果指定了包含的级别，则只保留包含的级别
	if len(config.IncludeLevels) > 0 {
		included := false
		for _, level := range config.IncludeLevels {
			if entry.Level == level {
				included = true
				break
			}
		}
		if !included {
			return false // 不在包含列表中，过滤掉
		}
	}
	
	// 级别过滤：如果指定了排除的级别，则排除这些级别
	if len(config.ExcludeLevels) > 0 {
		for _, level := range config.ExcludeLevels {
			if entry.Level == level {
				return false // 在排除列表中，过滤掉
			}
		}
	}
	
	// 时间范围过滤：只保留开始时间之后的日志
	if config.TimeStart != "" {
		startTime, _ := parseTime(config.TimeStart)
		if entry.Timestamp.Before(startTime) {
			return false // 在开始时间之前，过滤掉
		}
	}
	
	// 时间范围过滤：只保留结束时间之前的日志
	if config.TimeEnd != "" {
		endTime, _ := parseTime(config.TimeEnd)
		if entry.Timestamp.After(endTime) {
			return false // 在结束时间之后，过滤掉
		}
	}
	
	return true // 符合所有过滤条件，保留该日志
}

// processLogEntry 处理日志条目，更新全局统计信息
// 使用互斥锁保证多线程并发更新时的数据一致性
func processLogEntry(entry LogEntry) {
	mutex.Lock()         // 加锁，确保并发安全
	defer mutex.Unlock() // 延迟解锁，函数退出时自动释放锁
	
	// 更新总条目计数
	globalResult.TotalEntries++
	
	// 更新级别计数
	globalResult.LevelCount[entry.Level]++
	
	// 错误消息计数：只统计ERROR和FATAL级别的日志
	if entry.Level == LevelError || entry.Level == LevelFatal {
		// 简化错误消息，只保留前100个字符作为键，避免键过长
		key := truncateString(entry.Message, 100)
		globalResult.ErrorMessages[key]++
	}
	
	// 提取请求路径并更新统计
	// 尝试从两个可能的字段中获取请求路径
	if path, ok := entry.Details["request_path"].(string); ok && path != "" {
		globalResult.RequestCount[path]++
	} else if path, ok := entry.Details["path"].(string); ok && path != "" {
		globalResult.RequestCount[path]++
	}
	
	// 提取响应时间并更新统计
	if rt, ok := entry.Details["response_time"].(int); ok {
		if path, ok := entry.Details["request_path"].(string); ok && path != "" {
			globalResult.ResponseTimes[path] = append(globalResult.ResponseTimes[path], rt)
		} else {
			// 无法确定路径时，使用"unknown"作为键
			globalResult.ResponseTimes["unknown"] = append(globalResult.ResponseTimes["unknown"], rt)
		}
	}
	
	// 更新时间分布统计（按小时级别）
	hourKey := entry.Timestamp.Format("2006-01-02 15:00")
	globalResult.TimeDistribution[hourKey]++
}

// truncateString 将字符串截断到指定长度
// 如果字符串长度超过maxLen，则截断并添加省略号
func truncateString(s string, maxLen int) string {
	if len(s) <= maxLen {
		return s
	}
	return s[:maxLen] + "..."
}

// prepareResults 准备分析结果，计算衍生数据
// 对原始统计数据进行处理，如排序、筛选等，为生成报告做准备
func prepareResults() {
	// 将错误消息转换为切片，以便排序
	errorList := make([]ErrorCount, 0, len(globalResult.ErrorMessages))
	for msg, count := range globalResult.ErrorMessages {
		errorList = append(errorList, ErrorCount{Message: msg, Count: count})
	}
	
	// 按错误数量降序排序
	sort.Slice(errorList, func(i, j int) bool {
		return errorList[i].Count > errorList[j].Count
	})
	
	// 取前10个错误作为主要错误
	if len(errorList) > 10 {
		globalResult.TopErrors = errorList[:10]
	} else {
		globalResult.TopErrors = errorList
	}
}

	ext := strings.ToLower(filepath.Ext(outputFile))

		return generateHtmlReport(result, newOutputFile)
	}
}

// generateJsonReport 生成JSON格式的分析报告
// JSON格式适合程序进一步处理和分析
func generateJsonReport(result AnalysisResult, outputFile string) error {
	// 创建输出文件
	file, err := os.Create(outputFile)
	if err != nil {
		return err
	}
	defer file.Close()
	
	// 创建带缓冲的写入器
	writer := bufio.NewWriter(file)
	defer writer.Flush()
	
	// 创建JSON编码器，设置缩进格式，提高可读性
	encoder := json.NewEncoder(writer)
	encoder.SetIndent("", "  ")
	// 将分析结果编码为JSON并写入文件
	return encoder.Encode(result)
}

// generateCsvReport 生成CSV格式的分析报告
// CSV格式适合用Excel等表格软件打开和分析
func generateCsvReport(result AnalysisResult, outputFile string) error {
	// 创建输出文件
	file, err := os.Create(outputFile)
	if err != nil {
		return err
	}
	defer file.Close()
	
	// 创建CSV写入器
	writer := csv.NewWriter(file)
	defer writer.Flush()
	
	// 写入CSV表头
	if err := writer.Write([]string{"报告类型", "项目", "数量", "占比"}); err != nil {
		return err
	}
	
	// 写入级别分布数据
	for level, count := range result.LevelCount {
		percentage := ""
		if result.TotalEntries > 0 {
			percentage = fmt.Sprintf("%.2f%%", float64(count)/float64(result.TotalEntries)*100)
		}
		if err := writer.Write([]string{"级别分布", level, strconv.Itoa(count), percentage}); err != nil {
			return err
		}
	}
	
	// 写入错误消息数据
	for _, errorCount := range result.TopErrors {
		if err := writer.Write([]string{"错误消息", errorCount.Message, strconv.Itoa(errorCount.Count), ""}); err != nil {
			return err
		}
	}
	
	// 写入请求统计数据
	paths := make([]string, 0, len(result.RequestCount))
	for path := range result.RequestCount {
		paths = append(paths, path)
	}
	// 按请求次数排序
	sort.Slice(paths, func(i, j int) bool {
		return result.RequestCount[paths[i]] > result.RequestCount[paths[j]]
	})
	
	// 写入请求路径数据
	for _, path := range paths {
		count := result.RequestCount[path]
		if err := writer.Write([]string{"请求统计", path, strconv.Itoa(count), ""}); err != nil {
			return err
		}
	}
	
	return nil
}

// escapeHtml 转义HTML特殊字符，防止XSS攻击和格式错乱
// 将&、<、>、"、'等特殊字符转换为HTML实体
func escapeHtml(s string) string {
	s = strings.ReplaceAll(s, "&", "&amp;")
	s = strings.ReplaceAll(s, "<", "&lt;")
	s = strings.ReplaceAll(s, ">", "&gt;")
	s = strings.ReplaceAll(s, "\"", "&quot;")
	s = strings.ReplaceAll(s, "'", "&#39;")
	return s
}
```
