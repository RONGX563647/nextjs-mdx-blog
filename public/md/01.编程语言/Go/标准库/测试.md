package main

import (
	"encoding/json"
	"fmt"
	"io"
	"net/http"
	"net/http/httptest"
	"testing"
)

// User 表示用户结构体
type User struct {
	ID   int    `json:"id"`
	Name string `json:"name"`
	Age  int    `json:"age"`
}

// UserService 处理用户相关业务逻辑
type UserService struct {
	users map[int]User
}

// NewUserService 创建 UserService 实例
func NewUserService() *UserService {
	return &UserService{
		users: make(map[int]User),
	}
}

// AddUser 添加用户
func (us *UserService) AddUser(user User) {
	us.users[user.ID] = user
}

// GetUser 获取用户
func (us *UserService) GetUser(id int) (User, bool) {
	user, exists := us.users[id]
	return user, exists
}

// UserHandler 处理用户相关 HTTP 请求
type UserHandler struct {
	service *UserService
}

// NewUserHandler 创建 UserHandler 实例
func NewUserHandler(service *UserService) *UserHandler {
	return &UserHandler{
		service: service,
	}
}

// AddUserHandler 处理添加用户的 HTTP 请求
func (uh *UserHandler) AddUserHandler(w http.ResponseWriter, r *http.Request) {
	body, err := io.ReadAll(r.Body)
	if err != nil {
		http.Error(w, "读取请求体失败", http.StatusBadRequest)
		return
	}
	defer r.Body.Close()

	var user User
	if err := json.Unmarshal(body, &user); err != nil {
		http.Error(w, "解析用户数据失败", http.StatusBadRequest)
		return
	}

	uh.service.AddUser(user)

	w.WriteHeader(http.StatusCreated)
	w.Write([]byte("用户添加成功"))
}

// GetUserHandler 处理获取用户的 HTTP 请求
func (uh *UserHandler) GetUserHandler(w http.ResponseWriter, r *http.Request) {
	id := 0
	// 这里简化处理，假设从请求中获取 ID 为 1
	// 实际场景可能从 URL 参数等获取
	id = 1

	user, exists := uh.service.GetUser(id)
	if !exists {
		http.Error(w, "用户不存在", http.StatusNotFound)
		return
	}

	userJSON, err := json.Marshal(user)
	if err != nil {
		http.Error(w, "序列化用户数据失败", http.StatusInternalServerError)
		return
	}

	w.Header().Set("Content-Type", "application/json")
	w.Write(userJSON)
}

// SetupRouter 设置路由
func SetupRouter(handler *UserHandler) *http.ServeMux {
	mux := http.NewServeMux()
	mux.HandleFunc("/users", handler.AddUserHandler)
	mux.HandleFunc("/users/1", handler.GetUserHandler)
	return mux
}

// 单元测试部分
func TestUserService_AddUser(t *testing.T) {
	service := NewUserService()
	user := User{ID: 1, Name: "张三", Age: 25}
	service.AddUser(user)

	storedUser, exists := service.GetUser(1)
	if !exists {
		t.Fatalf("用户添加后，获取不到，预期存在")
	}
	if storedUser.Name != "张三" || storedUser.Age != 25 {
		t.Errorf("用户信息错误，预期姓名张三，年龄25，实际姓名%v，年龄%v", storedUser.Name, storedUser.Age)
	}
}

func TestUserHandler_AddUserHandler(t *testing.T) {
	service := NewUserService()
	handler := NewUserHandler(service)
	router := SetupRouter(handler)

	reqBody := `{"id": 1, "name": "李四", "age": 30}`
	req, err := http.NewRequest(http.MethodPost, "/users", nil)
	if err != nil {
		t.Fatalf("创建请求失败: %v", err)
	}
	req.Body = io.NopCloser(bytes.NewBufferString(reqBody))

	w := httptest.NewRecorder()
	router.ServeHTTP(w, req)

	if w.Code != http.StatusCreated {
		t.Errorf("添加用户返回状态码错误，预期%d，实际%d", http.StatusCreated, w.Code)
	}
	respBody := w.Body.String()
	if respBody != "用户添加成功" {
		t.Errorf("添加用户返回体错误，预期'用户添加成功'，实际%v", respBody)
	}

	storedUser, exists := service.GetUser(1)
	if !exists {
		t.Fatalf("用户添加后，服务中获取不到，预期存在")
	}
	if storedUser.Name != "李四" || storedUser.Age != 30 {
		t.Errorf("用户信息错误，预期姓名李四，年龄30，实际姓名%v，年龄%v", storedUser.Name, storedUser.Age)
	}
}

func TestUserHandler_GetUserHandler(t *testing.T) {
	service := NewUserService()
	user := User{ID: 1, Name: "王五", Age: 35}
	service.AddUser(user)

	if err != nil {
		t.Fatalf("创建请求失败: %v", err)
	}

	router.ServeHTTP(w, req)

	if w.Code != http.StatusOK {
		t.Errorf("获取用户返回状态码错误，预期%d，实际%d", http.StatusOK, w.Code)
	}
	var respUser User
	if err := json.Unmarshal(w.Body.Bytes(), &respUser); err != nil {
		t.Fatalf("解析返回用户数据失败: %v", err)
	}
	if respUser.Name != "王五" || respUser.Age != 35 {
		t.Errorf("返回用户信息错误，预期姓名王五，年龄35，实际姓名%v，年龄%v", respUser.Name, respUser.Age)
	}
}

// httptest 相关测试，模拟 HTTP 客户端请求
func TestHTTPClient(t *testing.T) {
	// 创建模拟服务器
	service := NewUserService()
	handler := NewUserHandler(service)
	router := SetupRouter(handler)
	server := httptest.NewServer(router)
	defer server.Close()

	// 创建 HTTP 客户端并发送请求
	client := &http.Client{}
	reqBody := `{"id": 1, "name": "客户端测试用户", "age": 28}`
	req, err := http.NewRequest(http.MethodPost, server.URL+"/users", bytes.NewBufferString(reqBody))
	if err != nil {
		t.Fatalf("创建请求失败: %v", err)
	}

	resp, err := client.Do(req)
	if err != nil {
		t.Fatalf("发送请求失败: %v", err)
	}
	defer resp.Body.Close()

	if resp.StatusCode != http.StatusCreated {
		t.Errorf("添加用户返回状态码错误，预期%d，实际%d", http.StatusCreated, resp.StatusCode)
	}

	// 发送获取用户请求
	getReq, err := http.NewRequest(http.MethodGet, server.URL+"/users/1", nil)
	if err != nil {
		t.Fatalf("创建获取请求失败: %v", err)
	}

	getResp, err := client.Do(getReq)
	if err != nil {
		t.Fatalf("发送获取请求失败: %v", err)
	}
	defer getResp.Body.Close()

	if getResp.StatusCode != http.StatusOK {
		t.Errorf("获取用户返回状态码错误，预期%d，实际%d", http.StatusOK, getResp.StatusCode)
	}

	var user User
	if err := json.NewDecoder(getResp.Body).Decode(&user); err != nil {
		t.Fatalf("解析返回用户数据失败: %v", err)
	}
	if user.Name != "客户端测试用户" || user.Age != 28 {
		t.Errorf("返回用户信息错误，预期姓名客户端测试用户，年龄28，实际姓名%v，年龄%v", user.Name, user.Age)
	}
}

### 代码说明

- **业务逻辑部分**：定义了 `User` 结构体表示用户，`UserService` 处理用户的添加和获取业务，`UserHandler` 处理 HTTP 请求，`SetupRouter` 用于设置路由。
- **单元测试**：对 `UserService` 的 `AddUser` 方法、`UserHandler` 的 `AddUserHandler` 和 `GetUserHandler` 方法进行单元测试，验证功能正确性。
- **基准测试**：通过 `BenchmarkUserHandler_AddUserHandler` 和 `BenchmarkUserHandler_GetUserHandler` 测试添加用户和获取用户接口的性能。
- **子测试**：在 `TestUserService` 中使用子测试，分别测试用户添加和不存在用户获取的场景。
- **运行验证示例**：`TestRunAndVerify` 模拟简单的运行验证场景。
- **其他功能测试**：`TestUserHandler_AddUserHandler_InvalidRequest` 测试无效请求的错误处理，`TestHTTPClient` 使用 `httptest` 模拟 HTTP 客户端发送请求并验证。
