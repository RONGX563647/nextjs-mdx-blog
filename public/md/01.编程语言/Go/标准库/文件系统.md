package main

import (
	"fmt"
	"io/fs"
	"os"
	"path/filepath"

// FileStats 存储文件统计信息
// 所有字段名首字母大写以确保可导出和正确引用
type FileStats struct {
	TotalFiles        int64         // 总文件数
	TotalDirs         int64         // 总目录数
	TotalSize         int64         // 总大小（字节）
	LargestFile       string        // 最大文件路径
	LargestFileSize   int64         // 最大文件大小
	NewestFile        string        // 最新文件路径
	NewestFileModTime time.Time     // 最新文件修改时间（修正字段名拼写）
	OldestFile        string        // 最旧文件路径
	OldestFileModTime time.Time     // 最旧文件修改时间（修正字段名拼写）
	ExtensionStats    map[string]int64 // 按扩展名统计的文件大小
	ExtensionCount    map[string]int64  // 按扩展名统计的文件数量
}

// WalkStats 存储遍历过程中的统计信息（支持并发安全更新）
type WalkStats struct {
	FileStats
	mutex sync.Mutex // 保护并发更新的互斥锁
}

// NewWalkStats 创建新的统计对象
// 初始化所有必要的字段，避免nil指针错误
func NewWalkStats() *WalkStats {
	return &WalkStats{
		FileStats: FileStats{
			ExtensionStats:   make(map[string]int64),
			ExtensionCount:   make(map[string]int64),
			NewestFileModTime: time.Unix(0, 0), // 初始化为最早时间
			OldestFileModTime: time.Now(),      // 初始化为当前时间
		},
	}
}

// updateFileStats 并发安全地更新文件统计
// 所有字段引用与结构体定义严格一致
func (ws *WalkStats) updateFileStats(path string, info fs.FileInfo) {
	ws.mutex.Lock()
	defer ws.mutex.Unlock()

	// 更新总文件数或目录数
	if info.IsDir() {
		ws.TotalDirs++
	} else {
		ws.TotalFiles++
		ws.TotalSize += info.Size()

		// 更新扩展名统计
		ext := filepath.Ext(path)
		ws.ExtensionCount[ext]++
		ws.ExtensionStats[ext] += info.Size()

		// 更新最大文件
		if info.Size() > ws.LargestFileSize {
			ws.LargestFile = path
			ws.LargestFileSize = info.Size()
		}

		// 更新最新文件（字段名与结构体一致）
		if info.ModTime().After(ws.NewestFileModTime) {
			ws.NewestFile = path
			ws.NewestFileModTime = info.ModTime()
		}

		// 更新最旧文件（字段名与结构体一致）
		if info.ModTime().Before(ws.OldestFileModTime) {
			ws.OldestFile = path
			ws.OldestFileModTime = info.ModTime()
		}
	}
}

// walkDir 递归遍历目录（使用标准文件系统）
// 支持并发控制，提高大目录遍历效率
func walkDir(path string, stats *WalkStats, concurrency int) error {
	// 创建信号量控制并发数量
	sem := make(chan struct{}, concurrency)
	var wg sync.WaitGroup
	var err error
	errMutex := sync.Mutex{}

	// 定义遍历回调函数
	walkFunc := func(currentPath string, d fs.DirEntry, walkErr error) error {
		if walkErr != nil {
			fmt.Printf("访问路径 %s 出错: %v\n", currentPath, walkErr)
			return nil // 跳过错误路径
		}

		// 获取文件信息
		info, statErr := d.Info()
		if statErr != nil {
			fmt.Printf("获取 %s 信息出错: %v\n", currentPath, statErr)
			return nil
		}

		// 更新统计信息
		stats.updateFileStats(currentPath, info)

		// 如果是目录且不是当前目录，递归处理子目录
		if d.IsDir() && currentPath != "." {
			sem <- struct{}{} // 获取信号量
			wg.Add(1)
			
			// 启动goroutine处理子目录
			go func(dirPath string) {
				defer func() {
					<-sem // 释放信号量
					wg.Done()
				}()
				
				// 递归处理子目录
				if subErr := walkDir(dirPath, stats, concurrency); subErr != nil {
					errMutex.Lock()
					if err == nil {
						err = subErr
					}
					errMutex.Unlock()
				}
			}(currentPath)
			
			// 如果达到最大并发数，等待一个goroutine完成
			if len(sem) == cap(sem) {
				<-sem
				wg.Done()
			}
		}

		return nil
	}

	// 开始遍历
	if walkErr := filepath.WalkDir(path, walkFunc); walkErr != nil {
		return walkErr
	}

	// 等待所有goroutine完成
	wg.Wait()
	
	return err
}

// walkFS 遍历抽象文件系统（演示 io/fs 用法）
// 展示Go 1.16+引入的抽象文件系统接口的使用
func walkFS(fsys fs.FS, root string, stats *WalkStats) error {
	// fs.WalkDir：递归遍历抽象文件系统的目录
	return fs.WalkDir(fsys, ".", func(path string, d fs.DirEntry, err error) error {
		if err != nil {
			fmt.Printf("访问抽象文件系统路径 %s 出错: %v\n", path, err)
			return nil
		}

		info, err := d.Info() // 获取抽象文件的信息
		if err != nil {
			fmt.Printf("获取抽象文件系统 %s 信息出错: %v\n", path, err)
			return nil
		}

		// 拼接完整路径
		fullPath := filepath.Join(root, path)
		stats.updateFileStats(fullPath, info)
		return nil
	})
}

// formatSize 将字节数格式化为易读的单位（KB/MB/GB）
// 方便用户理解文件大小
func formatSize(bytes int64) string {
	const unit = 1024
	if bytes < unit {
		return fmt.Sprintf("%d B", bytes)
	}
	div, exp := int64(unit), 0
	for n := bytes / unit; n >= unit; n /= unit {
		div *= unit
		exp++
	}
	return fmt.Sprintf("%.1f %cB",
		float64(bytes)/float64(div), "KMGTPE"[exp])
}

// printStats 打印文件统计信息
// 格式化输出所有收集到的统计数据
func printStats(stats *WalkStats) {
	fmt.Println("\n===== 文件系统统计结果 =====")
	fmt.Printf("总目录数: %d\n", stats.TotalDirs)
	fmt.Printf("总文件数: %d\n", stats.TotalFiles)
	fmt.Printf("总大小: %s\n", formatSize(stats.TotalSize))

	fmt.Printf("\n最大文件: %s (大小: %s)\n", stats.LargestFile, formatSize(stats.LargestFileSize))
	fmt.Printf("最新文件: %s (修改时间: %v)\n", stats.NewestFile, stats.NewestFileModTime.Format(time.RFC3339))
	fmt.Printf("最旧文件: %s (修改时间: %v)\n", stats.OldestFile, stats.OldestFileModTime.Format(time.RFC3339))

	fmt.Println("\n按扩展名统计:")
	// 排序扩展名（按数量降序）
	var exts []string
	for ext := range stats.ExtensionCount {
		exts = append(exts, ext)
	}
	sort.Slice(exts, func(i, j int) bool {
		return stats.ExtensionCount[exts[i]] > stats.ExtensionCount[exts[j]]
	})

	// 打印每个扩展名的统计信息
	for _, ext := range exts {
		fmt.Printf("  %-6s: %6d 个文件, 总大小 %s\n",
			ext, stats.ExtensionCount[ext], formatSize(stats.ExtensionStats[ext]))
	}
}

// getDirSize 计算指定目录的总大小
func getDirSize(path string) (int64, error) {
	var size int64
	err := filepath.WalkDir(path, func(path string, d fs.DirEntry, err error) error {
		if err != nil {
			return err
		}
		if !d.IsDir() {
			info, err := d.Info()
			if err != nil {
				return err
			}
			size += info.Size()
		}
		return nil
	})
	return size, err
}

// findDuplicateFiles 查找指定目录中的重复文件（基于大小）
// 作为额外功能展示文件系统分析的扩展能力
func findDuplicateFiles(path string) (map[int64][]string, error) {
	sizeMap := make(map[int64][]string)
	
	err := filepath.WalkDir(path, func(path string, d fs.DirEntry, err error) error {
		if err != nil {
			return err
		}
		if !d.IsDir() {
			info, err := d.Info()
			if err != nil {
				return err
			}
			size := info.Size()
			if size > 0 { // 忽略空文件
				sizeMap[size] = append(sizeMap[size], path)
			}
		}
		return nil
	})
	
	// 过滤掉只有一个文件的大小组
	for size, paths := range sizeMap {
		if len(paths) <= 1 {
			delete(sizeMap, size)
		}
	}
	
	return sizeMap, err
}

func main() {
	// 解析命令行参数，获取要分析的目录
	rootDir := "."
	if len(os.Args) > 1 {
		rootDir = os.Args[1]
	}

	// 验证目录是否存在
	if _, err := os.Stat(rootDir); os.IsNotExist(err) {
		fmt.Printf("错误: 目录 '%s' 不存在\n", rootDir)
		os.Exit(1)
	}

	fmt.Printf("开始分析目录: %s\n", rootDir)

	// 1. 使用标准文件系统遍历（os + path/filepath）
	stats := NewWalkStats()
	startTime := time.Now()
	
	// 使用并发数为8的方式遍历目录，提高效率
	if err := walkDir(rootDir, stats, 8); err != nil {
		fmt.Printf("遍历目录出错: %v\n", err)
		return
	}
	
	fmt.Printf("标准遍历耗时: %v\n", time.Since(startTime))

	// 打印标准遍历结果
	printStats(stats)

	// 2. 使用抽象文件系统遍历（演示 io/fs 用法）
	fmt.Println("\n===== 抽象文件系统演示 =====")
	// 创建当前目录的抽象文件系统
	fsys := os.DirFS(rootDir)
	abstractStats := NewWalkStats()
	absStartTime := time.Now()
	
	if err := walkFS(fsys, rootDir, abstractStats); err != nil {
		fmt.Printf("抽象遍历出错: %v\n", err)
		return
	}
	
	fmt.Printf("抽象遍历耗时: %v\n", time.Since(absStartTime))

	// 3. 路径操作演示（path/filepath）
	fmt.Println("\n===== 路径操作演示 =====")
	examplePath := "docs/articles/../notes.txt"
	cleanPath := filepath.Clean(examplePath) // 清理路径中的冗余部分
	absPath, _ := filepath.Abs(cleanPath)   // 获取绝对路径
	base := filepath.Base(absPath)          // 获取基名（最后一个元素）
	dir := filepath.Dir(absPath)            // 获取目录部分
	ext := filepath.Ext(absPath)            // 获取扩展名

	fmt.Printf("原始路径: %s\n", examplePath)
	fmt.Printf("清理后: %s\n", cleanPath)
	fmt.Printf("绝对路径: %s\n", absPath)
	fmt.Printf("基名: %s\n", base)
	fmt.Printf("目录: %s\n", dir)
	fmt.Printf("扩展名: %s\n", ext)

	// 4. 文件操作演示（os）
	fmt.Println("\n===== 文件操作演示 =====")
	tempFile, err := os.CreateTemp("", "fs_demo_*.txt") // 创建临时文件
	if err != nil {
		fmt.Printf("创建临时文件出错: %v\n", err)
		return
	}
	defer os.Remove(tempFile.Name()) // 程序结束时删除临时文件

	tempPath := tempFile.Name()
	fmt.Printf("创建临时文件: %s\n", tempPath)
	tempFile.WriteString("文件系统演示内容") // 写入内容
	tempFile.Close()

	// 检查文件是否存在
	if _, err := os.Stat(tempPath); err == nil {
		fmt.Println("文件存在")
	}

	// 获取文件信息
	info, _ := os.Stat(tempPath)
	fmt.Printf("文件大小: %d 字节\n", info.Size())
	fmt.Printf("修改时间: %v\n", info.ModTime().Format(time.RFC3339))

	// 5. 查找重复文件
	fmt.Println("\n===== 重复文件查找 =====")
	duplicates, err := findDuplicateFiles(rootDir)
	if err != nil {
		fmt.Printf("查找重复文件出错: %v\n", err)
	} else {
		if len(duplicates) == 0 {
			fmt.Println("未发现重复文件")
		} else {
			fmt.Printf("发现 %d 组重复文件:\n", len(duplicates))
			for size, paths := range duplicates {
				fmt.Printf("大小为 %s 的文件:\n", formatSize(size))
				for _, path := range paths {
					fmt.Printf("  - %s\n", path)
				}
			}
		}
	}
}

### 代码功能与包函数详解

#### 1. `os` 包的核心用法

- **`os.DirFS(root string) fs.FS`**：将本地目录封装为**抽象文件系统**（实现 `fs.FS` 接口），让本地文件能以 “抽象 FS” 的方式被操作，也可扩展到 Zip、HTTP 等其他文件系统。
- **`os.CreateTemp(dir, pattern string) (f \*os.File, err error)`**：在临时目录创建临时文件，适合临时数据存储（示例中用于演示文件创建）。
- **`os.Stat(name string) (fs.FileInfo, error)`**：获取文件 / 目录的元信息（如大小、修改时间、是否为目录等），是文件属性查询的核心函数。
- **`os.Remove(name string)`**：删除文件或空目录（示例中通过 `defer` 延迟调用，确保程序结束时清理临时文件）。
- **`os.Args`**：命令行参数切片，用于获取用户指定的目录路径（示例中简化处理，取第一个参数为目标目录）。

#### 2. `path/filepath` 包的核心用法

- **`filepath.WalkDir(root string, walkDirFunc) error`**：递归遍历目录，对每个文件 / 目录执行回调函数。相比旧版 `filepath.Walk`，它基于 `fs.DirEntry` 更高效，减少了不必要的 `stat` 调用。
- **`filepath.Clean(path string) string`**：清理路径中的冗余部分（如 `..`、多余的分隔符），生成规范的路径格式。
- **`filepath.Abs(path string) (string, error)`**：将相对路径转换为绝对路径，便于统一路径表示。
- **`filepath.Base(path string)`**：获取路径的 “基名”（即最后一个元素，类似文件名）。
- **`filepath.Dir(path string)`**：获取路径的 “目录部分”（即去掉基名后的路径）。
- **`filepath.Ext(path string)`**：获取文件的扩展名（如 `.txt`、`.go`）。
- **`filepath.Join(elem ...string) string`**：拼接多个路径元素，自动处理跨平台的路径分隔符（Windows 用 `\`，类 Unix 用 `/`）。

#### 3. `io/fs` 包的核心用法

- **`fs.FS` 接口**：抽象文件系统的核心接口，定义了 `Open(name string) (fs.File, error)` 方法。它让 “本地文件系统”“Zip 包”“HTTP 资源” 等不同存储介质，能以**统一的接口**被操作（示例中用 `os.DirFS` 封装本地目录演示此特性）。
- **`fs.WalkDir(fsys fs.FS, root string, walkDirFunc) error`**：递归遍历**抽象文件系统**的目录，行为与 `filepath.WalkDir` 一致，但作用于 `fs.FS` 而非本地文件系统。这让 “遍历 Zip 内的文件”“遍历网络资源目录” 等操作，能复用相同的遍历逻辑。
- **`fs.DirEntry` 和 `fs.FileInfo`**：分别表示 “目录项” 和 “文件信息” 的抽象接口。`DirEntry` 提供轻量的目录项信息（如 `IsDir()`、`Name()`），`FileInfo` 提供更详细的文件元数据（如 `Size()`、`ModTime()`）。

#### 4. 并发与统计功能

- **`sync.Mutex`**：互斥锁，用于保护**共享的统计数据**（如 `FileStats`）。当多个 goroutine 同时遍历文件时，通过加锁确保统计更新的线程安全。
- **自定义统计结构**：`FileStats` 封装了文件系统的核心统计指标（文件数、总大小、最大 / 最新 / 最旧文件、扩展名分布等）；`WalkStats` 则通过嵌入 `FileStats` 并添加锁，实现并发安全的统计更新。
- **排序与格式化**：使用 `sort.Slice` 对 “扩展名统计结果” 按数量降序排序；通过 `formatSize` 函数将原始字节数转换为易读的单位（KB/MB/GB），提升结果可读性。

### 二、核心模块详解

#### 1. 数据存储设计

- **`FileStats`结构体**：存储所有统计结果，包括：
  - 基础计数（文件数、目录数、总大小）
  - 极值信息（最大 / 最新 / 最旧文件）
  - 扩展名字典（按扩展名统计数量和大小）
- **`WalkStats`结构体**：对`FileStats`进行封装，添加互斥锁`mutex`，实现**并发安全的数据更新**。通过`updateFileStats`方法保证多 goroutine 同时更新时的数据一致性。
- 初始化设计：`NewWalkStats`函数确保所有字段（尤其是 map 类型）被正确初始化，避免 nil 指针错误。

#### 2. 目录遍历引擎

程序实现了两种遍历方式，体现了 Go 语言文件系统处理的灵活性：

- **并发遍历（`walkDir`）**：
  - 使用**信号量模式**（带缓冲的 channel）控制并发数量（默认 8 个 goroutine）
  - 采用递归 + goroutine 池的方式，对子目录进行并行处理
  - 通过`WaitGroup`等待所有子目录处理完成，`errMutex`捕获遍历过程中的错误
- **抽象文件系统遍历（`walkFS`）**：
  - 演示 Go 1.16 + 引入的`io/fs`接口，支持对抽象文件系统（如 zip 包、内存文件系统）的遍历
  - 与`walkDir`形成对比，展示标准库的多态设计

#### 3. 辅助功能模块

- **格式化工具**：`formatSize`将字节数转换为人类可读的单位（B/KB/MB 等）
- **路径处理**：使用`path/filepath`包的`Clean`/`Abs`/`Base`等函数处理路径
- **文件操作**：演示临时文件创建、写入、删除等基础操作
- **重复文件查找**：`findDuplicateFiles`通过文件大小初步识别重复文件（可扩展为基于哈希的精确匹配）

#### 4. 主流程控制

1. 解析命令行参数，获取目标目录
2. 验证目录有效性

7. 执行重复文件查找并展示结果

### 三、设计亮点

1. **并发安全设计**：
   - 使用`sync.Mutex`保护共享数据`FileStats`的并发修改
   - 信号量模式控制 goroutine 数量，避免资源耗尽
   - `WaitGroup`确保所有子任务完成后再继续执行
2. **接口与抽象**：
   - 同时支持`os`包（实际文件系统）和`io/fs`接口（抽象文件系统）
   - 展示了 Go 语言 "面向接口编程" 的特性，提高代码扩展性
3. **错误处理**：
   - 遍历过程中对每个路径的错误单独处理，不影响整体遍历
   - 使用互斥锁收集错误，确保错误信息不丢失
4. **用户体验优化**：
   - 结果格式化输出，增强可读性
   - 提供详细的路径操作演示，帮助理解文件系统概念
   - 兼容默认目录（当前目录）和命令行指定目录
5. **可扩展性**：
   - 结构体设计便于添加新的统计维度
   - 遍历函数与统计函数分离，可单独复用
   - 重复文件查找功能可轻松扩展为基于内容的精确匹配

### 四、总结

该程序充分利用了 Go 语言的并发特性（goroutine、channel、sync 包）和文件系统处理能力，通过模块化设计实现了功能的清晰划分。既展示了基础的文件操作 API 使用，也体现了高级的并发控制和接口设计思想，是一个兼顾实用性和教学性的示例。

程序的设计哲学是 "简洁而不简单"—— 核心功能（目录遍历和统计）实现精炼，同时通过扩展功能和演示代码展示了 Go 语言处理文件系统的多种方式。
